17 // Jacobian transformation matrix
18 template <class data_t>
19 static Tensor<2, data_t> spherical_jacobian(const data_t x, const double y,
20  const double z)
21 {
22  // calculate useful position quantities
23  data_t rho2 = simd_max(x * x + y * y, 1e-12);
24  data_t rho = sqrt(rho2);
25  data_t r2 = simd_max(x * x + y * y + z * z, 1e-12);
26  data_t r = sqrt(r2);
27 
28  // And the sines and cosines of phi and theta
29  data_t cos_phi = x / rho;
30  data_t sin_phi = y / rho;
31 
32  // derivatives for jacobian matrix - drdx etc
33  Tensor<2, data_t> jac;
34  jac[0][0] = x / r;
35  jac[1][0] = cos_phi * z / r2;
36  jac[2][0] = -y / rho2;
37  jac[0][1] = y / r;
38  jac[1][1] = sin_phi * z / r2;
39  jac[2][1] = x / rho2;
40  jac[0][2] = z / r;
41  jac[1][2] = -rho / r2;
42  jac[2][2] = 0.0;
43  return jac;
44 }

46 // Incerse Jacobian
47 template <class data_t>
48 static Tensor<2, data_t>
49 inverse_spherical_jacobian(const data_t x, const double y, const double z)
50 {
51  // calculate useful position quantities
52  data_t rho2 = simd_max(x * x + y * y, 1e-12);
53  data_t rho = sqrt(rho2);
54  data_t r2 = simd_max(x * x + y * y + z * z, 1e-12);
55  data_t r = sqrt(r2);
56 
57  // And the sines and cosines of phi and theta
58  // data_t sin_theta = rho / r;
59  data_t cos_phi = x / rho;
60  data_t sin_phi = y / rho;
61 
62  // derivatives for inverse jacobian matrix - drdx etc
63  Tensor<2, data_t> inv_jac;
64  inv_jac[0][0] = x / r;
65  inv_jac[1][0] = y / r;
66  inv_jac[2][0] = z / r;
67  inv_jac[0][1] = z * cos_phi;
68  inv_jac[1][1] = z * sin_phi;
69  inv_jac[2][1] = -rho;
70  inv_jac[0][2] = -y;
71  inv_jac[1][2] = x;
72  inv_jac[2][2] = 0.0;
73  return inv_jac;
74 }

76 // Convert a Tensor (with two lower indices) in spherical coords to cartesian
77 // coords
78 template <class data_t>
79 static Tensor<2, data_t>
80 spherical_to_cartesian_LL(const Tensor<2, data_t> &spherical_g, const data_t x,
81  const double y, const double z)
82 {
83  Tensor<2, data_t> cartesian_g;
84 
85  // derivatives for jacobian matrix - drdx etc
86  Tensor<2, data_t> jac = spherical_jacobian(x, y, z);
87 
88  // Convert the Tensor to cartesian coords
89  FOR(i, j)
90  {
91  cartesian_g[i][j] = 0.;
92  FOR(k, l)
93  {
94  cartesian_g[i][j] += spherical_g[k][l] * jac[k][i] * jac[l][j];
95  }
96  }
97  return cartesian_g;
98 }
