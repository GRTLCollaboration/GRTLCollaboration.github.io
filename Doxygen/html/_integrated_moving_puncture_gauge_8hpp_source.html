56 // \partial_t shift - advec_coeff * advec.shift = 0
57 // Include in your Example in GRAMRLevel::initial_data as:

61 template <class data_t> void compute(Cell<data_t> current_cell) const
62 {
63 const auto vars = current_cell.template load_vars<Vars>();
64
65 Tensor<1, data_t> B;
66 FOR(i)
67 {
68 B[i] = m_params.shift_Gamma_coeff * vars.Gamma[i] -
69 m_params.eta * vars.shift[i];
70 }
71
72 current_cell.store_vars(B, GRInterval<c_B1, c_B3>());
73 }

75 template <class data_t, template <typename> class vars_t,
76 template <typename> class diff2_vars_t>
77 inline void rhs_gauge(vars_t<data_t> &rhs, const vars_t<data_t> &vars,
78 const vars_t<Tensor<1, data_t>> &d1,
79 const diff2_vars_t<Tensor<2, data_t>> &d2,
80 const vars_t<data_t> &advec) const
81 {
82 rhs.lapse = m_params.lapse_advec_coeff * advec.lapse -
83 m_params.lapse_coeff *
84 pow(vars.lapse, m_params.lapse_power) *
85 (vars.K - 2 * vars.Theta);
86 FOR(i)
87 {
88 rhs.shift[i] = m_params.shift_advec_coeff * advec.shift[i] +
89 m_params.shift_Gamma_coeff * vars.Gamma[i] -
90 m_params.eta * vars.shift[i] - vars.B[i];
91 rhs.B[i] = 0.; // static, stays the same to save initial condition
92 }
