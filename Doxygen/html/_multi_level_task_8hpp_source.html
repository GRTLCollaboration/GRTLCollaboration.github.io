19template <class level_t = GRAMRLevel>
20class MultiLevelTask : public Scheduler::PeriodicFunction
21{
22 std::function<void(level_t *)> m_func;
23 bool m_reverse_levels;
24
25 // Use default condstructor of PeriodicFunction
26 using PeriodicFunction::PeriodicFunction;
27
28 AMR *m_amr_ptr;
29
30 public:
31 MultiLevelTask(std::function<void(level_t *)> a_func,
32 bool a_reverse_levels = true)
33 : m_func(a_func), m_reverse_levels(a_reverse_levels)
34 {
35 }
36
37 // required from Scheduler::PeriodicFunction
38 virtual void setUp(AMR &a_AMR, int a_interval = -1) override
39 {
40 m_amr_ptr = &a_AMR;
41 }
42
43 // required from Scheduler::PeriodicFunction
44 virtual void operator()(int a_step = 0, Real a_time = 0.) override
45 {
46 auto amr_level_ptrs = m_amr_ptr->getAMRLevels().stdVector();
47
48 // need to reverse this vector so that m_func is called in order of
49 // finest level to coarsest. This is important for example for
50 // 'specificPostTimeStep', which is always run in reverse order of
51 // levels
52 if (m_reverse_levels)
53 std::reverse(std::begin(amr_level_ptrs), std::end(amr_level_ptrs));
54
55 for (AMRLevel *amr_level_ptr : amr_level_ptrs)
56 m_func(level_t::gr_cast(amr_level_ptr));
57 }
58};

75 int a_interval = std::numeric_limits<int>::max())
76 : RefCountedPtr<Scheduler>(new Scheduler),
77 m_ptr_to_func(new MultiLevelTask<level_t>(a_func, a_reverse_levels))
78 // the two 'new' pointers are deleted by RefCountedPtr, no memory leak
79 {
80 if (a_interval <= 0) // the user probably means "never again"
81 a_interval = std::numeric_limits<int>::max();
82 (*this)->schedule(m_ptr_to_func, a_interval);
83 }
84
85 // run immediately!
