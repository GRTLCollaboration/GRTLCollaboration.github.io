42template <class data_t>
43void EMTensor<matter_t>::compute(Cell<data_t> current_cell) const
44{
45 const auto vars = current_cell.template load_vars<Vars>();
46 const auto d1 = m_deriv.template diff1<Vars>(current_cell);
47
48 using namespace TensorAlgebra;
49
50 const auto h_UU = compute_inverse_sym(vars.h);
51 const auto chris = compute_christoffel(d1.h, h_UU);
52
53 const auto emtensor = m_matter.compute_emtensor(vars, d1, h_UU, chris.ULL);
54
55 if (m_c_rho >= 0)
56 {
57 current_cell.store_vars(emtensor.rho, m_c_rho);
58 }
59
60 if (m_c_Si.size() > 0)
61 {
62#if DEFAULT_TENSOR_DIM == 3
63 FOR(i) { current_cell.store_vars(emtensor.Si[i], m_c_Si.begin() + i); }
64#endif
65 }
66
67 if (m_c_Sij.size() > 0)
68 {
69#if DEFAULT_TENSOR_DIM == 3
70 current_cell.store_vars(emtensor.Sij[0][0], m_c_Sij.begin());
71 current_cell.store_vars(emtensor.Sij[0][1], m_c_Sij.begin() + 1);
72 current_cell.store_vars(emtensor.Sij[0][2], m_c_Sij.begin() + 2);
73 current_cell.store_vars(emtensor.Sij[1][1], m_c_Sij.begin() + 3);
74 current_cell.store_vars(emtensor.Sij[1][2], m_c_Sij.begin() + 4);
75 current_cell.store_vars(emtensor.Sij[2][2], m_c_Sij.begin() + 5);
76#endif
77 }
78}
