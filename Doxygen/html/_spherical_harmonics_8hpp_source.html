24 // Calculates the spin weight es, el, em spherical harmonic
25 template <class data_t>
26 Y_lm_t<data_t> spin_Y_lm(const data_t x, const double y, const double z,
27  const int es, const int el, const int em)
28 {
29 
30  CH_assert((el >= 0) && (el >= abs(em)));
31 
32  Y_lm_t<data_t> Y_lm;
33 
34  // calculate useful position quantities
35  data_t r = simd_max(sqrt(x * x + y * y + z * z), 1e-6);
36  data_t theta = acos(z / r);
37  data_t phi = atan2(y, x);
38 
39  using namespace Combinatorics;
40  double coefficient = pow(-1.0, es) * sqrt((2.0 * el + 1.0) / (4.0 * M_PI));
41  coefficient *= sqrt(factorial(el + em) * factorial(el - em) /
42  factorial(el + es) / factorial(el - es));
43 
44  data_t sum = 0.0;
45  int lower_limit = em + es > 0 ? em + es : 0;
46  int upper_limit = el + em < el + es ? el + em : el + es;
47 
48  for (int i = lower_limit; i <= upper_limit; i++)
49  {
50  double temp = n_choose_r(el + es, i) * n_choose_r(el - es, i - es - em);
51  sum += temp * pow(-1.0, i) *
52  pow(cos(theta / 2.0), 2 * (el - i) + es + em) *
53  pow(sin(theta / 2.0), 2 * i - em - es);
54  }
55 
56  Y_lm.Real = coefficient * sum * cos(em * phi);
57  Y_lm.Im = coefficient * sum * sin(em * phi);
58  Y_lm.magnitude = sqrt(Y_lm.Real * Y_lm.Real + Y_lm.Im * Y_lm.Im);
59 
60  return Y_lm;
61 }
