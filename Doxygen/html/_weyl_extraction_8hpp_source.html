56  double r, double, double) {
57  // here the std::vector<double> passed will just have
58  // the real and imaginary parts of the Weyl4 scalar as its

60  return std::make_pair(r * Weyl4_reim_parts[0],
61  r * Weyl4_reim_parts[1]);
