21template <int N_DIMS>
22void QuinticConvolution<N_DIMS>::setup(
23 const std::array<int, N_DIMS> &deriv,
24 const std::array<double, N_DIMS> &eval_index)
25{
26 m_interp_points.clear();
27 m_interp_weights.clear();
28
29 std::array<double, N_DIMS> dxs = m_source.get_dxs();
30
31 double weights_1d[N_DIMS][6];
32
33 for (int dim = 0; dim < N_DIMS; ++dim)
34 {
35 double s = eval_index[dim] - floor(eval_index[dim]);
36
37 if (deriv[dim] == 0)
38 {
39 weights_1d[dim][0] =
40 s *
41 (0.046875 +
42 s * (-0.1875 + s * (0.28125 + (-0.1875 + 0.046875 * s) * s)));
43 weights_1d[dim][1] =
44 s *
45 (-0.59375 +
46 s * (1.25 + s * (-0.5625 + (-0.296875 + 0.203125 * s) * s)));
47 weights_1d[dim][2] =
48 1. + (s * s) * (-2.125 + (1.96875 - 0.84375 * s) * (s * s));
49 weights_1d[dim][3] =
50 s * (0.59375 +
51 s * (1.25 + s * (0.5625 + (-2.25 + 0.84375 * s) * s)));
52 weights_1d[dim][4] =
53 s *
54 (-0.046875 +
55 s * (-0.1875 + s * (-0.28125 + (0.71875 - 0.203125 * s) * s)));
56 weights_1d[dim][5] =
57 (0.046875 - 0.046875 * s) * ((s * s) * (s * s));
58 }
59 else if (deriv[dim] == 1)
60 {
61 weights_1d[dim][0] =
62 (0.046875 +
63 s * (-0.375 + s * (0.84375 + (-0.75 + 0.234375 * s) * s))) /
64 dxs[dim];
65 weights_1d[dim][1] =
66 (-0.59375 +
67 s * (2.5 + s * (-1.6875 + s * (-1.1875 + 1.015625 * s)))) /
68 dxs[dim];
69 weights_1d[dim][2] =
70 (s * (-4.25 + (7.875 - 4.21875 * s) * (s * s))) / dxs[dim];
71 weights_1d[dim][3] =
72 (0.59375 + s * (2.5 + s * (1.6875 + s * (-9. + 4.21875 * s)))) /
73 dxs[dim];
74 weights_1d[dim][4] =
75 (-0.046875 +
76 s * (-0.375 + s * (-0.84375 + (2.875 - 1.015625 * s) * s))) /
77 dxs[dim];
78 weights_1d[dim][5] =
79 ((0.1875 - 0.234375 * s) * s * (s * s)) / dxs[dim];
80 }
81 else if (deriv[dim] == 2)
82 {
83 weights_1d[dim][0] =
84 (-0.375 + s * (1.6875 + (-2.25 + 0.9375 * s) * s)) /
85 (dxs[dim] * dxs[dim]);
86 weights_1d[dim][1] =
87 (2.5 + s * (-3.375 + s * (-3.5625 + 4.0625 * s))) /
88 (dxs[dim] * dxs[dim]);
89 weights_1d[dim][2] = (-4.25 + (23.625 - 16.875 * s) * (s * s)) /
90 (dxs[dim] * dxs[dim]);
91 weights_1d[dim][3] = (2.5 + s * (3.375 + s * (-27. + 16.875 * s))) /
92 (dxs[dim] * dxs[dim]);
93 weights_1d[dim][4] =
94 (-0.375 + s * (-1.6875 + (8.625 - 4.0625 * s) * s)) /
95 (dxs[dim] * dxs[dim]);
96 weights_1d[dim][5] =
97 ((0.5625 - 0.9375 * s) * (s * s)) / (dxs[dim] * dxs[dim]);
98 }
99 else
100 {
101 MayDay::Error("Quintic convolution algorithm only supports up to "
102 "second derivative");
103 }
104 }
105
106 std::array<double, N_DIMS> interp_index;
107
108#if N_DIMS >= 3
109 for (int z = 0; z < 6; ++z)
110 {
111 interp_index[2] = floor(eval_index[2]) + z - 2;
112#endif
113#if N_DIMS >= 2
114 for (int y = 0; y < 6; ++y)
115 {
116 interp_index[1] = floor(eval_index[1]) + y - 2;
117#endif
118
119 for (int x = 0; x < 6; ++x)
120 {
121 interp_index[0] = floor(eval_index[0]) + x - 2;
122 CH_assert(m_source.contains(interp_index));
123
124 m_interp_points.push_back(IntVect(D_DECL6(
125 interp_index[0], interp_index[1], interp_index[2],
126 interp_index[3], interp_index[4], interp_index[5])));
127#if N_DIMS >= 3
128 m_interp_weights.push_back(weights_1d[0][x] * weights_1d[1][y] *
129 weights_1d[2][z]);
130#elif N_DIMS >= 2
131 m_interp_weights.push_back(weights_1d[0][x] * weights_1d[1][y]);
132#else
133 m_interp_weights.push_back(weights_1d[0][x]);
134#endif
135 }
136#if N_DIMS >= 2
137 }
138#endif
139#if N_DIMS >= 3
140 }
141#endif
142}

145template <class GeneralArrayBox>
146double QuinticConvolution<N_DIMS>::interpData(const GeneralArrayBox &fab,
147 int comp)
148{
149 long double accum = 0.0;
150
151 for (int i = 0; i < m_interp_points.size(); ++i)
152 {
153 double data = m_interp_weights[i] * fab.get(m_interp_points[i], comp);
154 accum += data;
155 }
156
157 return accum;
