83template <class vars_t, typename value_t>
84ALWAYS_INLINE void assign(vars_t &vars, const value_t &value)
85{
86 // The template magic below is needed to make sure that we can write
87 // assign(vars, 0.) and 0. gets correctly cast from double to simd<double>
88 // if necessary.
89 using data_t = typename strip_nested_template<vars_t>::type;
90 vars.enum_mapping([&value](const int &ivar, data_t &var)
91 { var = static_cast<data_t>(value); });
92}
