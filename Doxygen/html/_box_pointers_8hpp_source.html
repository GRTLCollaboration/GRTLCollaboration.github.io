46 BoxPointers(const FArrayBox &in, FArrayBox &out)
47 : m_in_lo(in.loVect()), m_in_hi(in.hiVect()), m_out_lo(out.loVect()),
48 m_out_hi(out.hiVect()), m_in_num_comps(in.nComp()),
49 m_out_num_comps(out.nComp())
50 {
51 m_in_ptr.resize(m_in_num_comps);
52 m_out_ptr.resize(m_out_num_comps);
53 // dataPtr in Chombo does CH_assert bound check
54 // which we don't want to do in a loop
55 for (int i = 0; i < m_in_num_comps; ++i)
56 m_in_ptr[i] = in.dataPtr(i);
57 // If the output FArrayBox doesn't have all components, just don't set
58 // the pointers
59 for (int i = 0; i < m_out_num_comps; ++i)
60 m_out_ptr[i] = out.dataPtr(i);
61
62 m_in_stride[0] = 1;
63 m_in_stride[1] = m_in_hi[0] - m_in_lo[0] + 1;
64#if CH_SPACEDIM >= 3
65 m_in_stride[2] = (m_in_hi[1] - m_in_lo[1] + 1) * m_in_stride[1];
66#endif
67
68 m_out_stride[0] = 1;
69 m_out_stride[1] = m_out_hi[0] - m_out_lo[0] + 1;
70#if CH_SPACEDIM >= 3
71 m_out_stride[2] = (m_out_hi[1] - m_out_lo[1] + 1) * m_out_stride[1];
72#endif
73 }
74
75 CellIndexIn get_in_index(IntVect integer_coords) const
76 {
77 return (m_in_stride[2] * (integer_coords[2] - m_in_lo[2]) +
78 m_in_stride[1] * (integer_coords[1] - m_in_lo[1]) +
79 (integer_coords[0] - m_in_lo[0]));
80 }
81
82 CellIndexOut get_out_index(IntVect integer_coords) const
83 {
84 return (m_out_stride[2] * (integer_coords[2] - m_out_lo[2]) +
85 m_out_stride[1] * (integer_coords[1] - m_out_lo[1]) +
86 (integer_coords[0] - m_out_lo[0]));
87 }
