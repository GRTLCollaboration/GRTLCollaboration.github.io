86 void add_mode_integrand(
87 int es, int el, int em, const complex_function_t &a_function,
88 std::pair<std::vector<double>, std::vector<double>> &out_integrals,
89 const IntegrationMethod &a_method_theta = IntegrationMethod::simpson,
90 const IntegrationMethod &a_method_phi = IntegrationMethod::trapezium,
91 const bool a_broadcast_integral = false)
92 {
93 // {x, y, z}
94 SphericalGeometry::UP_DIR up_dir = this->m_geom.get_up_dir();
95 std::array<int, 3> dirs = {(up_dir + 1) % 3, (up_dir + 2) % 3, up_dir};
96 std::array<double, 3> center;
97 for (int i = 0; i < 3; ++i)
98 center[i] = (dirs[i] < CH_SPACEDIM ? m_center[dirs[i]] : 0.);
99
100 auto integrand_re = [dirs, center, &geom = this->m_geom, es, el, em,
101 &a_function](std::vector<double> &a_data_here,
102 double r, double theta, double phi)
103 {
104 // note that spin_Y_lm requires the coordinates with the center
105 // at the origin
106 double x = geom.get_grid_coord(dirs[0], r, theta, phi) - center[0];
107 double y = geom.get_grid_coord(dirs[1], r, theta, phi) - center[1];
108 double z = geom.get_grid_coord(dirs[2], r, theta, phi) - center[2];
109 SphericalHarmonics::Y_lm_t<double> Y_lm =
110 SphericalHarmonics::spin_Y_lm(x, y, z, es, el, em);
111 auto function_here = a_function(a_data_here, r, theta, phi);
112 return (function_here.first * Y_lm.Real +
113 function_here.second * Y_lm.Im) /
114 (r * r);
115 };
116 this->add_integrand(integrand_re, out_integrals.first, a_method_theta,
117 a_method_phi, a_broadcast_integral);
118
119 auto integrand_im = [dirs, center, &geom = this->m_geom, es, el, em,
120 &a_function](std::vector<double> &a_data_here,
121 double r, double theta, double phi)
122 {
123 // note that spin_Y_lm requires the coordinates with the center
124 // at the origin
125 double x = geom.get_grid_coord(dirs[0], r, theta, phi) - center[0];
126 double y = geom.get_grid_coord(dirs[1], r, theta, phi) - center[1];
127 double z = geom.get_grid_coord(dirs[2], r, theta, phi) - center[2];
128 SphericalHarmonics::Y_lm_t<double> Y_lm =
129 SphericalHarmonics::spin_Y_lm(x, y, z, es, el, em);
130 auto function_here = a_function(a_data_here, r, theta, phi);
131 return (function_here.second * Y_lm.Real -
132 function_here.first * Y_lm.Im) /
133 (r * r);
134 };
135 this->add_integrand(integrand_im, out_integrals.second, a_method_theta,
136 a_method_phi, a_broadcast_integral);
137 }

141 std::pair<std::vector<double>, std::vector<double>> integrate_mode(
142 int es, int el, int em, complex_function_t a_function,
143 const IntegrationMethod &a_method_theta = IntegrationMethod::simpson,
144 const IntegrationMethod &a_method_phi = IntegrationMethod::trapezium)
145 {
146 this->m_integrands.clear();
147 this->m_integration_methods.clear();
148 this->m_integrals.clear();
149
150 std::pair<std::vector<double>, std::vector<double>> integrals;
151 add_mode_integrand(es, el, em, a_function, integrals, a_method_theta,
152 a_method_phi);
153 this->integrate();
154 return integrals;
155 }
