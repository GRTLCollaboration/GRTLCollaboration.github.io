27static bool directory_exists(const std::string &path)
28{
29 struct stat stat_struct;
30 return (path == "") || (stat(path.c_str(), &stat_struct) == 0 &&
31 S_ISDIR(stat_struct.st_mode));
32}

34static bool mkdir_recursive(const std::string &path)
35{
36#ifdef CH_MPI
37 // all processes should get here at the same time
38 // e.g. if doing:
39 // if (!directory_exists(path))
40 // mkdir_recursive(path);
41 // We want to make sure the directory is not created until all
42 // ranks have passed the 'directory_exists'
43 MPI_Barrier(Chombo_MPI::comm);
44#endif
45
46 bool success = true;
47 if (procID() == 0)
48 {
49 // For Windows backslash, use delimeters = "/\\"
50 static const std::string delimeter = "/";
51 std::size_t found = path.find_first_of(delimeter);
52
53 // Even though this might look like RWX permissions for everyone,
54 // this will be masked by the system user mask (see 'man umask' or
55 // https://man7.org/linux/man-pages/man2/umask.2.html)
56 mode_t permissions = S_IRWXU | S_IRWXG | S_IRWXO;
57
58 // NB: this would be very beautiful recursively, but let's not do it
59 // because the function involves MPI_Barrier's
60 while (success && found != std::string::npos)
61 {
62 std::string subpath;
63 subpath = path.substr(0, found);
64
65 if (subpath != "." && subpath != "")
66 {
67 // success if created or if not created because it already
68 // exists
69 success &= (mkdir(subpath.c_str(), permissions) == 0 ||
70 errno == EEXIST);
71 }
72 found = path.find_first_of(delimeter, found + 1);
73 }
74 // if path doesn't finish with "/", one more to do
75 if (found != path.size() - 1)
76 success &=
77 (mkdir(path.c_str(), permissions) == 0 || errno == EEXIST);
78 }
79
80#ifdef CH_MPI
81 // all processes should wait to make sure directory structure is well set
82 // for everyone
83 MPI_Barrier(Chombo_MPI::comm);
84#endif
85
86 return success;
87}
