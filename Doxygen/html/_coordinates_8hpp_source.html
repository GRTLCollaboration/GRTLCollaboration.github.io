28 Coordinates(IntVect integer_coords, double dx,

42#ifdef CH_SPACEDIM

44#endif
45#endif

48 ALWAYS_INLINE static void compute_coord(double &out, int position,
49 double dx,
50 double center_distance = 0)
51 {
52 out = (position + 0.5) * dx - center_distance;
53 }

55 static void // typename std::enable_if_t<(simd_traits<double>::simd_len >
56 // 1), void>
57 compute_coord(simd<double> &out, int position, double dx,
58 double center_distance = 0)
59 {
60 double out_arr[simd_traits<double>::simd_len];
61 for (int i = 0; i < simd_traits<double>::simd_len; ++i)
62 {
63 out_arr[i] = (position + i + 0.5) * dx - center_distance;
64 }
65 out = simd<double>::load(out_arr);

70 data_t get_radius() const
71 {
72 // Note that this is not currently dimension independent
73 data_t r = sqrt(x * x + y * y + z * z);
74
75 const double minimum_r = 1e-6;
76 return simd_max(r, minimum_r);
77 }

81 static data_t get_radius(IntVect integer_coords, double dx,
82 std::array<double, CH_SPACEDIM> center = {0})
83 {
84 data_t xx;
85 double yy;
86 double zz;
87
88 // Note that this is not currently dimension independent
89 compute_coord(xx, integer_coords[0], dx, center[0]);
90 compute_coord(yy, integer_coords[1], dx, center[1]);
91 compute_coord(zz, integer_coords[2], dx, center[2]);
92
93 data_t r = sqrt(xx * xx + yy * yy + zz * zz);
94
95 const double minimum_r = 1e-6;
96 return simd_max(r, minimum_r);
97 }
