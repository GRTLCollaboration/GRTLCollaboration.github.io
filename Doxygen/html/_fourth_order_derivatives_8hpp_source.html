30 template <class data_t>
31 ALWAYS_INLINE data_t diff1(const double *in_ptr, const int idx,
32 const int stride) const
33 {
34 auto in = SIMDIFY<data_t>(in_ptr);
35
36 data_t weight_far = 8.33333333333333333333e-2;
37 data_t weight_near = 6.66666666666666666667e-1;
38
39 // NOTE: if you have been sent here by the debugger because of
40 // EXC_BAD_ACCESS or something similar you might be trying to take
41 // derivatives without ghost points.
42 return (weight_far * in[idx - 2 * stride] -
43 weight_near * in[idx - stride] +
44 weight_near * in[idx + stride] -
45 weight_far * in[idx + 2 * stride]) *
46 m_one_over_dx;
47 }

114

157 template <class data_t>
158 ALWAYS_INLINE data_t mixed_diff2(const double *in_ptr, const int idx,
159 const int stride1, const int stride2) const
160 {
161 auto in = SIMDIFY<data_t>(in_ptr);
162
163 data_t weight_far_far = 6.94444444444444444444e-3;
164 data_t weight_near_far = 5.55555555555555555556e-2;
165 data_t weight_near_near = 4.44444444444444444444e-1;
166
167 return (weight_far_far * in[idx - 2 * stride1 - 2 * stride2] -
168 weight_near_far * in[idx - 2 * stride1 - stride2] +
169 weight_near_far * in[idx - 2 * stride1 + stride2] -
170 weight_far_far * in[idx - 2 * stride1 + 2 * stride2]
171
172 - weight_near_far * in[idx - stride1 - 2 * stride2] +
173 weight_near_near * in[idx - stride1 - stride2] -
174 weight_near_near * in[idx - stride1 + stride2] +
175 weight_near_far * in[idx - stride1 + 2 * stride2]
176
177 + weight_near_far * in[idx + stride1 - 2 * stride2] -
178 weight_near_near * in[idx + stride1 - stride2] +
179 weight_near_near * in[idx + stride1 + stride2] -
180 weight_near_far * in[idx + stride1 + 2 * stride2]
181
182 - weight_far_far * in[idx + 2 * stride1 - 2 * stride2] +
183 weight_near_far * in[idx + 2 * stride1 - stride2] -
184 weight_near_far * in[idx + 2 * stride1 + stride2] +
185 weight_far_far * in[idx + 2 * stride1 + 2 * stride2]) *
186 m_one_over_dx2;
187 }

232 template <template <typename> class vars_t, class data_t>
233 auto diff2(const Cell<data_t> &current_cell) const
234 {
235 vars_t<Tensor<2, data_t>> d2;
236 const auto in_index = current_cell.get_in_index();
237 const auto strides = current_cell.get_box_pointers().m_in_stride;
238 d2.enum_mapping([&](const int &ivar, Tensor<2, data_t> &var) {
239 FOR(dir1) // First calculate the repeated derivatives
240 {
241 var[dir1][dir1] = diff2<data_t>(
242 current_cell.get_box_pointers().m_in_ptr[ivar], in_index,
243 strides[dir1]);
244 for (int dir2 = 0; dir2 < dir1; ++dir2)
245 {
246 auto tmp = mixed_diff2<data_t>(
247 current_cell.get_box_pointers().m_in_ptr[ivar],
248 in_index, strides[dir1], strides[dir2]);
249 var[dir1][dir2] = tmp;
250 var[dir2][dir1] = tmp;
251 }
252 }
253 });
254 return d2;
255 }

259 template <class data_t, class mask_t>
260 ALWAYS_INLINE data_t advection_term(const double *in_ptr, const int idx,
261 const data_t &vec_comp,
262 const int stride,
263 const mask_t shift_positive) const
264 {
265 const auto in = SIMDIFY<data_t>(in_ptr);
266 const data_t in_left = in[idx - stride];
267 const data_t in_centre = in[idx];
268 const data_t in_right = in[idx + stride];
269
270 data_t weight_0 = -2.50000000000000000000e-1;
271 data_t weight_1 = -8.33333333333333333333e-1;
272 data_t weight_2 = +1.50000000000000000000e+0;
273 data_t weight_3 = -5.00000000000000000000e-1;
274 data_t weight_4 = +8.33333333333333333333e-2;
275
276 data_t upwind;
277 upwind = vec_comp *
278 (weight_0 * in_left + weight_1 * in_centre +
279 weight_2 * in_right + weight_3 * in[idx + 2 * stride] +
280 weight_4 * in[idx + 3 * stride]) *
281 m_one_over_dx;
282
283 data_t downwind;
284 downwind = vec_comp *
285 (-weight_4 * in[idx - 3 * stride] -
286 weight_3 * in[idx - 2 * stride] - weight_2 * in_left -
287 weight_1 * in_centre - weight_0 * in_right) *
288 m_one_over_dx;
289
290 return simd_conditional(shift_positive, upwind, downwind);
291 }

346 template <class data_t>
347 ALWAYS_INLINE data_t dissipation_term(const double *in_ptr, const int idx,
348 const int stride) const
349 {
350 const auto in = SIMDIFY<data_t>(in_ptr);
351 data_t weight_vfar = 1.56250e-2;
352 data_t weight_far = 9.37500e-2;
353 data_t weight_near = 2.34375e-1;
354 data_t weight_local = 3.12500e-1;
355
356 return (weight_vfar * in[idx - 3 * stride] -
357 weight_far * in[idx - 2 * stride] +
358 weight_near * in[idx - stride] - weight_local * in[idx] +
359 weight_near * in[idx + stride] -
360 weight_far * in[idx + 2 * stride] +
361 weight_vfar * in[idx + 3 * stride]) *
362 m_one_over_dx;
363 }
