20 data_t BoostedBH::psi_minus_one(Coordinates<data_t> coords) const
21 {
22  const data_t r = center_dist(coords);
23  const data_t cos_theta = (coords.z - m_params.center[2]) / r;
24  const data_t P_squared = pow(m_params.momentum[0], 2) +

31 template <class data_t>
32 Tensor<2, data_t> BoostedBH::Aij(Coordinates<data_t> coords) const
33 {
34  const data_t r = center_dist(coords);
35  const data_t l[3] = {(coords.x - m_params.center[0]) / r,
36  (coords.y - m_params.center[1]) / r,
37  (coords.z - m_params.center[2]) / r};
38  const data_t l_dot_p = l[0] * m_params.momentum[0] +
39  l[1] * m_params.momentum[1] +
40  l[2] * m_params.momentum[2];
41 
42  Tensor<2, data_t> out;
43 
44  FOR(i, j)
45  {
46  const double delta = (i == j) ? 1 : 0;
47  out[i][j] = 1.5 *
48  (m_params.momentum[i] * l[j] + m_params.momentum[j] * l[i] -
49  (delta - l[i] * l[j]) * l_dot_p) /
50  (r * r);
51  }
52  return out;
53 }

58 data_t BoostedBH::center_dist(Coordinates<data_t> coords) const
59 {

79 {
