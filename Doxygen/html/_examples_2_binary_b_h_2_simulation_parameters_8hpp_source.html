44#ifdef USE_TWOPUNCTURES
45 void read_tp_params(GRParmParse &pp)
46 {
47 tp_params.verbose = (verbosity > 0);
48 // check whether to calculate the target ADM masses or use provided bare
49 // masses
50 bool calculate_target_masses;
51 pp.load("TP_calculate_target_masses", calculate_target_masses, false);
52 tp_params.give_bare_mass = !calculate_target_masses;
53
54 // masses
55 if (calculate_target_masses)
56 {
57 pp.load("TP_target_mass_plus", tp_params.target_M_plus);
58 pp.load("TP_target_mass_minus", tp_params.target_M_minus);
59 pp.load("TP_adm_tol", tp_params.adm_tol, 1e-10);
60 pout() << "The black holes have target ADM masses of "
61 << tp_params.target_M_plus << " and "
62 << tp_params.target_M_minus << "\n";
63 bh1_params.mass = tp_params.target_M_minus;
64 bh2_params.mass = tp_params.target_M_plus;
65 }
66 else
67 {
68 pp.load("TP_mass_plus", tp_params.par_m_plus);
69 pp.load("TP_mass_minus", tp_params.par_m_minus);
70 bh1_params.mass = tp_params.par_m_plus;
71 bh2_params.mass = tp_params.par_m_minus;
72 pout() << "The black holes have bare masses of "
73 << std::setprecision(16) << tp_params.par_m_plus << " and "
74 << tp_params.par_m_minus << "\n";
75 // reset precision
76 pout() << std::setprecision(6);
77 }
78
79 // BH spin and momenta
80 std::array<double, CH_SPACEDIM> spin_minus, spin_plus;
81 pp.load("TP_momentum_minus", bh1_params.momentum);
82 pp.load("TP_momentum_plus", bh2_params.momentum);
83 pp.load("TP_spin_plus", spin_plus);
84 pp.load("TP_spin_minus", spin_minus);
85 FOR(i)
86 {
87 tp_params.par_P_minus[i] = bh1_params.momentum[i];
88 tp_params.par_P_plus[i] = bh2_params.momentum[i];
89 tp_params.par_S_minus[i] = spin_minus[i];
90 tp_params.par_S_plus[i] = spin_plus[i];
91 }
92
93 pout() << "The corresponding momenta are:";
94 pout() << "\nP_plus = ";
95 FOR(i) { pout() << tp_params.par_P_plus[i] << " "; }
96 pout() << "\nP_minus = ";
97 FOR(i) { pout() << tp_params.par_P_minus[i] << " "; }
98
99 pout() << "\nThe corresponding spins are:";
100 pout() << "\nS_plus = ";
101 FOR(i) { pout() << tp_params.par_S_plus[i] << " "; }
102 pout() << "\nS_minus = ";
103 FOR(i) { pout() << tp_params.par_S_minus[i] << " "; }
104 pout() << "\n";
105
106 // interpolation type
107 bool use_spectral_interpolation;
108 pp.load("TP_use_spectral_interpolation", use_spectral_interpolation,
109 false);
110 tp_params.grid_setup_method =
111 (use_spectral_interpolation) ? "evaluation" : "Taylor expansion";
112
113 // initial_lapse (default to psi^n)
114 pp.load("TP_initial_lapse", tp_params.initial_lapse,
115 std::string("psi^n"));
116 if (tp_params.initial_lapse != "twopunctures-antisymmetric" &&
117 tp_params.initial_lapse != "twopunctures-averaged" &&
118 tp_params.initial_lapse != "psi^n" &&
119 tp_params.initial_lapse != "brownsville")
120 {
121 std::string message = "Parameter: TP_initial_lapse: ";
122 message += tp_params.initial_lapse;
123 message += " invalid";
124 MayDay::Error(message.c_str());
125 }
126 if (tp_params.initial_lapse == "psi^n")
127 {
128 pp.load("TP_initial_lapse_psi_exponent",
129 tp_params.initial_lapse_psi_exponent, -2.0);
130 }
131
132 // Spectral grid parameters
133 pp.load("TP_npoints_A", tp_params.npoints_A, 30);
134 pp.load("TP_npoints_B", tp_params.npoints_B, 30);
135 pp.load("TP_npoints_phi", tp_params.npoints_phi, 16);
136 if (tp_params.npoints_phi % 4 != 0)
137 {
138 MayDay::Error("TP_npoints_phi must be a multiple of 4");
139 }
140
141 // Solver parameters and tolerances
142 pp.load("TP_Newton_tol", tp_params.Newton_tol, 1e-10);
143 pp.load("TP_Newton_maxit", tp_params.Newton_maxit, 5);
144 pp.load("TP_epsilon", tp_params.TP_epsilon, 1e-6);
145 pp.load("TP_Tiny", tp_params.TP_Tiny, 0.0);
146 pp.load("TP_Extend_Radius", tp_params.TP_Extend_Radius, 0.0);
147
148 // BH positions
149 pp.load("TP_offset_minus", tp_offset_minus);
150 pp.load("TP_offset_plus", tp_offset_plus);
151 bh1_params.center = center;
152 bh2_params.center = center;
153 bh1_params.center[0] += tp_offset_minus;
154 bh2_params.center[0] += tp_offset_plus;
155 double center_offset_x = 0.5 * (tp_offset_plus + tp_offset_minus);
156 tp_params.center_offset[0] = center_offset_x;
157 // par_b is half the distance between BH_minus and BH_plus
158 tp_params.par_b = 0.5 * (tp_offset_plus - tp_offset_minus);
159 pp.load("TP_swap_xz", tp_params.swap_xz, false);
160
161 // Debug output
162 pp.load("TP_do_residuum_debug_output",
163 tp_params.do_residuum_debug_output, false);
164 pp.load("TP_do_initial_debug_output", tp_params.do_initial_debug_output,
165 false);
166
167 // Irrelevant parameters set to default value
168 tp_params.keep_u_around = false;
169 tp_params.use_sources = false;
170 tp_params.rescale_sources = true;
171 tp_params.use_external_initial_guess = false;
172 tp_params.multiply_old_lapse = false;
173 tp_params.schedule_in_ADMBase_InitialData = true;
174 tp_params.solve_momentum_constraint = false;
175 tp_params.metric_type = "something else";
176 tp_params.conformal_storage = "not conformal at all";
177 tp_params.conformal_state = 0;
178 tp_params.mp = 0;
179 tp_params.mm = 0;
180 tp_params.mp_adm = 0;
181 tp_params.mm_adm = 0;
182 }
183#else
185 void read_bh_params(GRParmParse &pp)
186 {
187 // Initial data
188 pp.load("massA", bh1_params.mass);
189 pp.load("momentumA", bh1_params.momentum);
190 pp.load("massB", bh2_params.mass);
191 pp.load("momentumB", bh2_params.momentum);
192
193 // Get the centers of the BHs either explicitly or as
194 // an offset (not both, or they will be offset from center
195 // provided)
196 std::array<double, CH_SPACEDIM> centerA, centerB;
197 std::array<double, CH_SPACEDIM> offsetA, offsetB;
198 pp.load("centerA", centerA, center);
199 pp.load("centerB", centerB, center);
200 pp.load("offsetA", offsetA, {0.0, 0.0, 0.0});
201 pp.load("offsetB", offsetB, {0.0, 0.0, 0.0});
202 FOR(idir)
203 {
204 bh1_params.center[idir] = centerA[idir] + offsetA[idir];
205 bh2_params.center[idir] = centerB[idir] + offsetB[idir];
206 }
207 }
208#endif /* USE_TWOPUNCTURES */

210 void check_params()
211 {
212#ifdef USE_TWOPUNCTURES
213 // These checks are mostly taken from the Einstein Toolkit thorn
214 // documentation:
215 // https://einsteintoolkit.org/thornguide/EinsteinInitialData/TwoPunctures/documentation.html
216 std::string mass_plus_name, mass_minus_name;
217 if (tp_params.give_bare_mass)
218 {
219 mass_minus_name = "TP_mass_minus";
220 mass_plus_name = "TP_mass_plus";
221 }
222 else
223 {
224 mass_minus_name = "TP_target_mass_minus";
225 mass_plus_name = "TP_target_mass_plus";
226 check_parameter("TP_adm_tol", tp_params.adm_tol,
227 tp_params.adm_tol > 0., "must be > 0.0");
228 }
229 check_parameter(mass_minus_name, bh1_params.mass, bh1_params.mass >= 0.,
230 "mustd be >= 0.0");
231 check_parameter(mass_plus_name, bh2_params.mass, bh2_params.mass >= 0.,
232 "must be >= 0.0");
233
234 int offset_dir = (!tp_params.swap_xz) ? 0 : 2;
235 warn_parameter("TP_offset_minus", tp_offset_minus,
236 tp_offset_minus < (ivN[offset_dir] + 1) * coarsest_dx -
237 center[offset_dir],
238 "should be within the computational domain");
239 warn_parameter("TP_offset_plus", tp_offset_plus,
240 tp_offset_plus < (ivN[offset_dir] + 1) * coarsest_dx -
241 center[offset_dir],
242 "should be within the computational domain");
243 check_parameter("TP_npoints_A", tp_params.npoints_A,
244 tp_params.npoints_A >= 4, "must be >= 4");
245 check_parameter("TP_npoints_B", tp_params.npoints_B,
246 tp_params.npoints_B >= 4, "must be >= 4");
247 check_parameter("TP_npoints_phi", tp_params.npoints_phi,
248 tp_params.npoints_phi >= 4 &&
249 tp_params.npoints_phi % 2 == 0,
250 "must be >= 4 and divisible by 2");
251 check_parameter("TP_Newton_maxit", tp_params.Newton_maxit,
252 tp_params.Newton_maxit >= 0, "must be >= 0");
253 check_parameter("TP_Newton_tol", tp_params.Newton_tol,
254 tp_params.Newton_tol >= 0., "must be >= 0.0");
255 check_parameter("TP_epsilon", tp_params.TP_epsilon,
256 tp_params.TP_epsilon >= 0., "must be >= 0.0");
257 check_parameter("TP_Tiny", tp_params.TP_Tiny, tp_params.TP_Tiny >= 0.,
258 "must be >= 0.0");
259 check_parameter("TP_Extend_Radius", tp_params.TP_Extend_Radius,
260 tp_params.TP_Extend_Radius >= 0., "must be >= 0.0");
261#else
262 warn_parameter("massA", bh1_params.mass, bh1_params.mass >= 0,
263 "should be >= 0");
264 warn_parameter("massB", bh2_params.mass, bh2_params.mass >= 0,
265 "should be >= 0");
266 warn_array_parameter(
267 "momentumA", bh1_params.momentum,
268 std::sqrt(ArrayTools::norm2(bh1_params.momentum)) <
269 0.3 * bh1_params.mass,
270 "approximation used for boosted BH only valid for small boosts");
271 warn_array_parameter(
272 "momentumB", bh2_params.momentum,
273 std::sqrt(ArrayTools::norm2(bh2_params.momentum)) <
274 0.3 * bh1_params.mass,
275 "approximation used for boosted BH only valid for small boosts");
276 FOR(idir)
277 {
278 std::string nameA = "centerA[" + std::to_string(idir) + "]";
279 std::string nameB = "centerB[" + std::to_string(idir) + "]";
280 double center_A_dir = bh1_params.center[idir];
281 double center_B_dir = bh2_params.center[idir];
282 warn_parameter(nameA, center_A_dir,
283 (center_A_dir >= 0.0) &&
284 (center_A_dir <= (ivN[idir] + 1) * coarsest_dx),
285 "should be within the computational domain");
286 warn_parameter(nameB, center_B_dir,
287 (center_B_dir >= 0.0) &&
288 (center_B_dir <= (ivN[idir] + 1) * coarsest_dx),
289 "should be within the computational domain");
290 }
291#endif /* USE_TWOPUNCTURES */
292 check_parameter("puncture_tracking_level", puncture_tracking_level,
293 (puncture_tracking_level >= 0) &&
294 (puncture_tracking_level <= max_level),
295 "must be between 0 and max_level (inclusive)");
296 }
