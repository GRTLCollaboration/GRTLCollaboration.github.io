32 template <class data_t>
33 ALWAYS_INLINE static data_t
34 compute_z_terms(const int i, const int j,
35 const Tensor<1, data_t> &Z_over_chi,
36 const Tensor<2, data_t> &h, const Tensor<1, data_t> &d1_chi)
37 {
38 data_t out = 0.;
39 FOR(k)
40 {
41 out += Z_over_chi[k] * (h[i][k] * d1_chi[j] + h[j][k] * d1_chi[i] -
42 h[i][j] * d1_chi[k]);
43 }
44 return out;

48 template <class data_t, template <typename> class vars_t,
49 template <typename> class diff2_vars_t>
50 static ricci_t<data_t>
51 compute_ricci_Z(const vars_t<data_t> &vars,
52 const vars_t<Tensor<1, data_t>> &d1,
53 const diff2_vars_t<Tensor<2, data_t>> &d2,
54 const Tensor<2, data_t> &h_UU, const chris_t<data_t> &chris,
55 const Tensor<1, data_t> &Z_over_chi)
56 {
57 ricci_t<data_t> out;
58
59 Tensor<2, data_t> covdtilde2chi;
60 FOR(k, l)
61 {
62 covdtilde2chi[k][l] = d2.chi[k][l];
63 FOR(m) { covdtilde2chi[k][l] -= chris.ULL[m][k][l] * d1.chi[m]; }
64 }
65
66 Tensor<3, data_t> chris_LLU = {0.};
67 data_t boxtildechi = 0.;
68 data_t dchi_dot_dchi = 0.;
69 FOR(i, j)
70 {
71 boxtildechi += covdtilde2chi[i][j] * h_UU[i][j];
72 dchi_dot_dchi += d1.chi[i] * d1.chi[j] * h_UU[i][j];
73 FOR(k, l) { chris_LLU[i][j][k] += h_UU[k][l] * chris.LLL[i][j][l]; }
74 }
75
76 FOR(i, j)
77 {
78 data_t ricci_hat = 0;
79 FOR(k)
80 {
81 // We call this ricci_hat rather than ricci_tilde as we have
82 // replaced what should be \tilde{Gamma} with \hat{Gamma} in
83 // order to avoid adding terms that cancel later on
84 ricci_hat += 0.5 * (vars.h[k][i] * d1.Gamma[k][j] +
85 vars.h[k][j] * d1.Gamma[k][i]);
86 ricci_hat += 0.5 * vars.Gamma[k] * d1.h[i][j][k];
87 FOR(l)
88 {
89 ricci_hat += -0.5 * h_UU[k][l] * d2.h[i][j][k][l] +
90 (chris.ULL[k][l][i] * chris_LLU[j][k][l] +
91 chris.ULL[k][l][j] * chris_LLU[i][k][l] +
92 chris.ULL[k][i][l] * chris_LLU[k][j][l]);
93 }
94 }
95
96 data_t ricci_chi =
97 0.5 * ((GR_SPACEDIM - 2) * covdtilde2chi[i][j] +
98 vars.h[i][j] * boxtildechi -
99 ((GR_SPACEDIM - 2) * d1.chi[i] * d1.chi[j] +
100 GR_SPACEDIM * vars.h[i][j] * dchi_dot_dchi) /
101 (2 * vars.chi));
102
103 data_t z_terms = compute_z_terms(i, j, Z_over_chi, vars.h, d1.chi);
104
105 out.LL[i][j] =
106 (ricci_chi + vars.chi * ricci_hat + z_terms) / vars.chi;
107 }
108
109 out.scalar = vars.chi * TensorAlgebra::compute_trace(out.LL, h_UU);
110
111 return out;
112 }

114 template <class data_t>
115 static Tensor<2, data_t>
116 compute_d1_chris_contracted(const Tensor<2, data_t> &h_UU,
117 const Tensor<2, Tensor<1, data_t>> &d1_h,
118 const Tensor<2, Tensor<2, data_t>> &d2_h)
119 {
120 Tensor<2, data_t> d1_chris_contracted = 0.0;
121 FOR(i, j)
122 {
123 FOR(m, n, p)
124 {
125 data_t d1_terms = 0.0;
126 FOR(q, r)
127 {
128 d1_terms += -h_UU[q][r] * (d1_h[n][q][j] * d1_h[m][p][r] +
129 d1_h[m][n][j] * d1_h[p][q][r]);
130 }
131 d1_chris_contracted[i][j] +=
132 h_UU[i][m] * h_UU[n][p] * (d2_h[m][n][j][p] + d1_terms);
133 }
134 }
135 return d1_chris_contracted;
136 }

138 // This function allows adding arbitrary multiples of D_{(i}Z_{j)}
139 // to the Ricci scalar rather than the default of 2 in compute_ricci_Z
140 template <class data_t, template <typename> class vars_t,
141 template <typename> class diff2_vars_t>
142 static ricci_t<data_t>
143 compute_ricci_Z_general(const vars_t<data_t> &vars,
144 const vars_t<Tensor<1, data_t>> &d1,
145 const diff2_vars_t<Tensor<2, data_t>> &d2,
146 const Tensor<2, data_t> &h_UU,
147 const chris_t<data_t> &chris, const double dZ_coeff)
148 {
149 // get contributions from conformal metric and factor with zero Z vector
150 Tensor<1, data_t> Z0 = 0.;
151 auto ricci = compute_ricci_Z(vars, d1, d2, h_UU, chris, Z0);
152
153 // need to add term to correct for d1.Gamma (includes Z contribution)
154 // and Gamma in ricci_hat
155 auto d1_chris_contracted =
156 compute_d1_chris_contracted(h_UU, d1.h, d2.h);
157 Tensor<1, data_t> Z_over_chi;
158 FOR(i) { Z_over_chi[i] = 0.5 * (vars.Gamma[i] - chris.contracted[i]); }
159 FOR(i, j)
160 {
161 FOR(m)
162 {
163 // This corrects for the \hat{Gamma}s in ricci_hat
164 ricci.LL[i][j] +=
165 (1. - 0.5 * dZ_coeff) * 0.5 *
166 (vars.h[m][i] *
167 (d1_chris_contracted[m][j] - d1.Gamma[m][j]) +
168 vars.h[m][j] *
169 (d1_chris_contracted[m][i] - d1.Gamma[m][i]) +
170 (chris.contracted[m] - vars.Gamma[m]) * d1.h[i][j][m]);
171 }
172 data_t z_terms = compute_z_terms(i, j, Z_over_chi, vars.h, d1.chi);
173 ricci.LL[i][j] += 0.5 * dZ_coeff * z_terms / vars.chi;
174 }
175 ricci.scalar = vars.chi * TensorAlgebra::compute_trace(ricci.LL, h_UU);
176 return ricci;
177 }

179 // This function returns the pure Ricci scalar with no contribution from the
180 // Z vector - used e.g. in the constraint calculations.
181 template <class data_t, template <typename> class vars_t,
182 template <typename> class diff2_vars_t>
183 static ricci_t<data_t>
184 compute_ricci(const vars_t<data_t> &vars,
185 const vars_t<Tensor<1, data_t>> &d1,
186 const diff2_vars_t<Tensor<2, data_t>> &d2,
187 const Tensor<2, data_t> &h_UU, const chris_t<data_t> &chris)
188 {
189 return compute_ricci_Z_general(vars, d1, d2, h_UU, chris, 0.0);
190 }
