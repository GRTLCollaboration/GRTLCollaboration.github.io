28 void read_params(GRParmParse &pp)
29 {
30 // Initial scalar field data
31 initial_params.center =
32 center; // already read in SimulationParametersBase
33 pp.load("G_Newton", G_Newton,
34 0.0); // for now the example neglects backreaction
35 pp.load("scalar_amplitude", initial_params.amplitude, 0.1);
36 pp.load("scalar_width", initial_params.width, 1.0);
37 pp.load("scalar_mass", potential_params.scalar_mass, 0.1);
38
39 // Initial Kerr data
40 pp.load("kerr_mass", kerr_params.mass, 1.0);
41 pp.load("kerr_spin", kerr_params.spin, 0.0);
42 pp.load("kerr_center", kerr_params.center, center);
43 }
44
45 void check_params()
46 {
47 warn_parameter("scalar_mass", potential_params.scalar_mass,
48 potential_params.scalar_mass <
49 0.2 / coarsest_dx / dt_multiplier,
50 "oscillations of scalar field do not appear to be "
51 "resolved on coarsest level");
52 warn_parameter("scalar_width", initial_params.width,
53 initial_params.width < 0.5 * L,
54 "is greater than half the domain size");
55 warn_parameter("kerr_mass", kerr_params.mass, kerr_params.mass >= 0.0,
56 "should be >= 0.0");
57 check_parameter("kerr_spin", kerr_params.spin,
58 std::abs(kerr_params.spin) <= kerr_params.mass,
59 "must satisfy |a| <= M = " +
60 std::to_string(kerr_params.mass));
61 FOR(idir)
62 {
63 std::string name = "kerr_center[" + std::to_string(idir) + "]";
64 warn_parameter(
65 name, kerr_params.center[idir],
66 (kerr_params.center[idir] >= 0) &&
67 (kerr_params.center[idir] <= (ivN[idir] + 1) * coarsest_dx),
68 "should be within the computational domain");
69 }
70 }
71
72 // Initial data for matter and potential and BH
73 double G_Newton;
74 InitialScalarData::params_t initial_params;
75 Potential::params_t potential_params;
76 KerrBH::params_t kerr_params;
