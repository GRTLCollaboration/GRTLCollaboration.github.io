18void QuinticConvolution::setup(const std::array<int, CH_SPACEDIM> &deriv,
19 const std::array<double, CH_SPACEDIM> &dx,
20 const std::array<double, CH_SPACEDIM> &evalCoord,
21 const IntVect &nearest)
22{
23 m_interp_points.clear();
24 m_interp_weights.clear();
25
26 double weights_1d[CH_SPACEDIM][6];
27
28 for (int dim = 0; dim < CH_SPACEDIM; ++dim)
29 {
30 double s = evalCoord[dim] - floor(evalCoord[dim]);
31
32 if (deriv[dim] == 0)
33 {
34 weights_1d[dim][0] =
35 s *
36 (0.046875 +
37 s * (-0.1875 + s * (0.28125 + (-0.1875 + 0.046875 * s) * s)));
38 weights_1d[dim][1] =
39 s *
40 (-0.59375 +
41 s * (1.25 + s * (-0.5625 + (-0.296875 + 0.203125 * s) * s)));
42 weights_1d[dim][2] =
43 1. + (s * s) * (-2.125 + (1.96875 - 0.84375 * s) * (s * s));
44 weights_1d[dim][3] =
45 s * (0.59375 +
46 s * (1.25 + s * (0.5625 + (-2.25 + 0.84375 * s) * s)));
47 weights_1d[dim][4] =
48 s *
49 (-0.046875 +
50 s * (-0.1875 + s * (-0.28125 + (0.71875 - 0.203125 * s) * s)));
51 weights_1d[dim][5] =
52 (0.046875 - 0.046875 * s) * ((s * s) * (s * s));
53 }
54 else if (deriv[dim] == 1)
55 {
56 weights_1d[dim][0] =
57 (0.046875 +
58 s * (-0.375 + s * (0.84375 + (-0.75 + 0.234375 * s) * s))) /
59 dx[dim];
60 weights_1d[dim][1] =
61 (-0.59375 +
62 s * (2.5 + s * (-1.6875 + s * (-1.1875 + 1.015625 * s)))) /
63 dx[dim];
64 weights_1d[dim][2] =
65 (s * (-4.25 + (7.875 - 4.21875 * s) * (s * s))) / dx[dim];
66 weights_1d[dim][3] =
67 (0.59375 + s * (2.5 + s * (1.6875 + s * (-9. + 4.21875 * s)))) /
68 dx[dim];
69 weights_1d[dim][4] =
70 (-0.046875 +
71 s * (-0.375 + s * (-0.84375 + (2.875 - 1.015625 * s) * s))) /
72 dx[dim];
73 weights_1d[dim][5] =
74 ((0.1875 - 0.234375 * s) * s * (s * s)) / dx[dim];
75 }
76 else if (deriv[dim] == 2)
77 {
78 weights_1d[dim][0] =
79 (-0.375 + s * (1.6875 + (-2.25 + 0.9375 * s) * s)) /
80 (dx[dim] * dx[dim]);
81 weights_1d[dim][1] =
82 (2.5 + s * (-3.375 + s * (-3.5625 + 4.0625 * s))) /
83 (dx[dim] * dx[dim]);
84 weights_1d[dim][2] =
85 (-4.25 + (23.625 - 16.875 * s) * (s * s)) / (dx[dim] * dx[dim]);
86 weights_1d[dim][3] = (2.5 + s * (3.375 + s * (-27. + 16.875 * s))) /
87 (dx[dim] * dx[dim]);
88 weights_1d[dim][4] =
89 (-0.375 + s * (-1.6875 + (8.625 - 4.0625 * s) * s)) /
90 (dx[dim] * dx[dim]);
91 weights_1d[dim][5] =
92 ((0.5625 - 0.9375 * s) * (s * s)) / (dx[dim] * dx[dim]);
93 }
94 else
95 {
96 MayDay::Error("Quintic convolution algorithm only supports up to "
97 "second derivative");
98 }
99 }
100
101 std::array<double, CH_SPACEDIM> interp_coord;
102
103#if CH_SPACEDIM >= 3
104 for (int z = 0; z < 6; ++z)
105 {
106 interp_coord[2] = floor(evalCoord[2]) + z - 2;
107#endif
108 for (int y = 0; y < 6; ++y)
109 {
110 interp_coord[1] = floor(evalCoord[1]) + y - 2;
111
112 for (int x = 0; x < 6; ++x)
113 {
114 interp_coord[0] = floor(evalCoord[0]) + x - 2;
115 CH_assert(m_source.contains(interp_coord));
116
117 m_interp_points.push_back(IntVect(D_DECL6(
118 interp_coord[0], interp_coord[1], interp_coord[2],
119 interp_coord[3], interp_coord[4], interp_coord[5])));
120 m_interp_weights.push_back(D_TERM6(weights_1d[0][x],
121 *weights_1d[1][y],
122 *weights_1d[2][z], , , ));
123 }
124 }
125#if CH_SPACEDIM >= 3
126 }
127#endif
