24template <class matter_t, class gauge_t, class deriv_t>
25template <class data_t>
26void MatterCCZ4RHS<matter_t, gauge_t, deriv_t>::compute(
27 Cell<data_t> current_cell) const
28{
29 // copy data from chombo gridpoint into local variables
30 const auto matter_vars = current_cell.template load_vars<Vars>();
31 const auto d1 = this->m_deriv.template diff1<Vars>(current_cell);
32 const auto d2 = this->m_deriv.template diff2<Diff2Vars>(current_cell);
33 const auto advec =
34 this->m_deriv.template advection<Vars>(current_cell, matter_vars.shift);
35
36 // Call CCZ4 RHS - work out RHS without matter, no dissipation
37 Vars<data_t> matter_rhs;
38 this->rhs_equation(matter_rhs, matter_vars, d1, d2, advec);
39
40 // add RHS matter terms from EM Tensor
41 add_emtensor_rhs(matter_rhs, matter_vars, d1);
42
43 // add evolution of matter fields themselves
44 my_matter.add_matter_rhs(matter_rhs, matter_vars, d1, d2, advec);
45
46 // Add dissipation to all terms
47 this->m_deriv.add_dissipation(matter_rhs, current_cell, this->m_sigma);
48
49 // Write the rhs into the output FArrayBox
50 current_cell.store_vars(matter_rhs);
51}

53// Function to add in EM Tensor matter terms to CCZ4 rhs
54template <class matter_t, class gauge_t, class deriv_t>
55template <class data_t>
56void MatterCCZ4RHS<matter_t, gauge_t, deriv_t>::add_emtensor_rhs(
57 Vars<data_t> &matter_rhs, const Vars<data_t> &matter_vars,
58 const Vars<Tensor<1, data_t>> &d1) const
59{
60 using namespace TensorAlgebra;
61
62 const auto h_UU = compute_inverse_sym(matter_vars.h);
63 const auto chris = compute_christoffel(d1.h, h_UU);
64
65 // Calculate elements of the decomposed stress energy tensor
66 const auto emtensor =
67 my_matter.compute_emtensor(matter_vars, d1, h_UU, chris.ULL);
68
69 // Update RHS for K and Theta depending on formulation
70 if (this->m_formulation == CCZ4RHS<>::USE_BSSN)
71 {
72 matter_rhs.K += 4.0 * M_PI * m_G_Newton * matter_vars.lapse *
73 (emtensor.S + emtensor.rho);
74 matter_rhs.Theta += 0.0;
75 }
76 else
77 {
78 matter_rhs.K += 4.0 * M_PI * m_G_Newton * matter_vars.lapse *
79 (emtensor.S - 3 * emtensor.rho);
80 matter_rhs.Theta +=
81 -8.0 * M_PI * m_G_Newton * matter_vars.lapse * emtensor.rho;
82 }
83
84 // Update RHS for other variables
85 Tensor<2, data_t> Sij_TF = emtensor.Sij;
86 make_trace_free(Sij_TF, matter_vars.h, h_UU);
87
88 FOR(i, j)
89 {
90 matter_rhs.A[i][j] += -8.0 * M_PI * m_G_Newton * matter_vars.chi *
91 matter_vars.lapse * Sij_TF[i][j];
92 }
93
94 FOR(i)
95 {
96 data_t matter_term_Gamma = 0.0;
97 FOR(j)
98 {
99 matter_term_Gamma += -16.0 * M_PI * m_G_Newton * matter_vars.lapse *
100 h_UU[i][j] * emtensor.Si[j];
101 }
102
103 matter_rhs.Gamma[i] += matter_term_Gamma;
104 matter_rhs.B[i] += matter_term_Gamma;
105 }
106}
