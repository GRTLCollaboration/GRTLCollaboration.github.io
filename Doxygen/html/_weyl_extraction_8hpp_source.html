42 void execute_query(AMRInterpolator<Lagrange<4>> *a_interpolator)
43 {
44 // extract the values of the Weyl scalars on the spheres
45 extract(a_interpolator);
46
47 if (m_params.write_extraction)
48 write_extraction(m_params.extraction_file_prefix);
49
50 // now calculate and write the requested spherical harmonic modes
51 std::vector<std::pair<std::vector<double>, std::vector<double>>>
52 mode_integrals(m_num_modes);
53
54 // note that this is normalised by multiplying by radius
55 auto normalised_Weyl4_complex = [](std::vector<double> Weyl4_reim_parts,
56 double r, double, double) {
57 // here the std::vector<double> passed will just have
58 // the real and imaginary parts of the Weyl4 scalar as its
59 // only components
60 return std::make_pair(r * Weyl4_reim_parts[0],
61 r * Weyl4_reim_parts[1]);
62 };
63
64 // add the modes that will be integrated
65 for (int imode = 0; imode < m_num_modes; ++imode)
66 {
67 const auto &mode = m_modes[imode];
68 constexpr int es = -2;
69 add_mode_integrand(es, mode.first, mode.second,
70 normalised_Weyl4_complex, mode_integrals[imode]);
71 }
72
73 // do the integration over the surface
74 integrate();
75
76 // write the integrals
77 for (int imode = 0; imode < m_num_modes; ++imode)
78 {
79 const auto &mode = m_modes[imode];
80 std::string integrals_filename = m_params.integral_file_prefix +
81 std::to_string(mode.first) +
82 std::to_string(mode.second);
83 std::vector<std::vector<double>> integrals_for_writing = {
84 std::move(mode_integrals[imode].first),
85 std::move(mode_integrals[imode].second)};
86 std::vector<std::string> labels = {"integral Re", "integral Im"};
87 write_integrals(integrals_filename, integrals_for_writing, labels);
88 }
89 }
90};
