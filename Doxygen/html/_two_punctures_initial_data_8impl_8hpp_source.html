13 void TwoPuncturesInitialData::compute(Cell<double> current_cell) const
14 {
15  Vars<double> vars;
16  // Set only the non-zero components explicitly below
17  VarsTools::assign(vars, 0.);
18 
19  Coordinates<double> coords(current_cell, m_dx, m_center);
20  Tensor<2, double> h_phys, K_tensor;
21  Tensor<1, double> shift, Z3;
22  double lapse, Theta;
23 
24  interpolate_tp_vars(coords, h_phys, K_tensor, lapse, shift, Theta, Z3);
25 
26  using namespace TensorAlgebra;
27  // analytically set Bowen-York properties below (e.g. conformal flatness,
28  // tracefree K)
29 
30  // metric variables
31  vars.chi = pow(compute_determinant_sym(h_phys), -1. / 3.);
32  FOR(i)
33  {
34  // Bowen-York data is conformally flat
35  vars.h[i][i] = 1.0;
36  }
37 
38  // extrinsic curvature
39  FOR(i, j) { vars.A[i][j] = vars.chi * K_tensor[i][j]; }
40  // conformal flatness means h_UU = h
41  make_trace_free(vars.A, vars.h, vars.h);
42 
43  // gauge
44  vars.lapse = lapse;
45 
46  current_cell.store_vars(vars);
47 }
