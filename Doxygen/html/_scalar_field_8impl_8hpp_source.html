14template <class potential_t>

39// Calculate the stress energy tensor elements
40template <class potential_t>
41template <class data_t, template <typename> class vars_t>
42void ScalarField<potential_t>::emtensor_excl_potential(
43 emtensor_t<data_t> &out, const vars_t<data_t> &vars,
44 const vars_t<Tensor<1, data_t>> &d1, const Tensor<2, data_t> &h_UU,
45 const Tensor<3, data_t> &chris_ULL)
46{
47 // Useful quantity Vt
48 data_t Vt = -vars.Pi * vars.Pi;
49 FOR(i, j) { Vt += vars.chi * h_UU[i][j] * d1.phi[i] * d1.phi[j]; }
50
51 // Calculate components of EM Tensor
52 // S_ij = T_ij
53 FOR(i, j)
54 {
55 out.Sij[i][j] =
56 -0.5 * vars.h[i][j] * Vt / vars.chi + d1.phi[i] * d1.phi[j];
57 }
58
59 // S = Tr_S_ij
60 out.S = vars.chi * TensorAlgebra::compute_trace(out.Sij, h_UU);
61
62 // S_i (note lower index) = - n^a T_ai
63 FOR(i) { out.Si[i] = -d1.phi[i] * vars.Pi; }
64
65 // rho = n^a n^b T_ab
66 out.rho = vars.Pi * vars.Pi + 0.5 * Vt;
67}

96// the RHS excluding the potential terms
97template <class potential_t>
98template <class data_t, template <typename> class vars_t,
99 template <typename> class diff2_vars_t,
100 template <typename> class rhs_vars_t>
101void ScalarField<potential_t>::matter_rhs_excl_potential(
102 rhs_vars_t<data_t> &rhs, const vars_t<data_t> &vars,
103 const vars_t<Tensor<1, data_t>> &d1,
104 const diff2_vars_t<Tensor<2, data_t>> &d2, const vars_t<data_t> &advec)
105{
106 using namespace TensorAlgebra;
107
108 const auto h_UU = compute_inverse_sym(vars.h);
109 const auto chris = compute_christoffel(d1.h, h_UU);
110
111 // evolution equations for scalar field and (minus) its conjugate momentum
112 rhs.phi = vars.lapse * vars.Pi + advec.phi;
113 rhs.Pi = vars.lapse * vars.K * vars.Pi + advec.Pi;
114
115 FOR(i, j)
116 {
117 // includes non conformal parts of chris not included in chris_ULL
118 rhs.Pi += h_UU[i][j] * (-0.5 * d1.chi[j] * vars.lapse * d1.phi[i] +
119 vars.chi * vars.lapse * d2.phi[i][j] +
120 vars.chi * d1.lapse[i] * d1.phi[j]);
121 FOR(k)
122 {
123 rhs.Pi += -vars.chi * vars.lapse * h_UU[i][j] * chris.ULL[k][i][j] *
124 d1.phi[k];
125 }
126 }
127}
