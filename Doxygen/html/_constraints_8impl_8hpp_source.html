36 template <class data_t, template <typename> class vars_t,
37  template <typename> class diff2_vars_t>
38 Constraints::Vars<data_t> Constraints::constraint_equations(
39  const vars_t<data_t> &vars, const vars_t<Tensor<1, data_t>> &d1,
40  const diff2_vars_t<Tensor<2, data_t>> &d2) const
41 {
42  Vars<data_t> out;
43 
44  const data_t chi_regularised = simd_max(1e-6, vars.chi);
45 
46  auto h_UU = TensorAlgebra::compute_inverse_sym(vars.h);
47  auto chris = TensorAlgebra::compute_christoffel(d1.h, h_UU);
48 
49  auto ricci = CCZ4Geometry::compute_ricci(vars, d1, d2, h_UU, chris);
50 
51  auto A_UU = TensorAlgebra::raise_all(vars.A, h_UU);
52  data_t tr_A2 = TensorAlgebra::compute_trace(vars.A, A_UU);
53 
54  out.Ham = ricci.scalar +
55  (GR_SPACEDIM - 1.) * vars.K * vars.K / GR_SPACEDIM - tr_A2;
56  out.Ham -= 2 * m_cosmological_constant;
57 
58  Tensor<2, data_t> covd_A[CH_SPACEDIM];
59  FOR(i, j, k)
60  {
61  covd_A[i][j][k] = d1.A[j][k][i];
62  FOR(l)
63  {
64  covd_A[i][j][k] += -chris.ULL[l][i][j] * vars.A[l][k] -
65  chris.ULL[l][i][k] * vars.A[l][j];
66  }
67  }
68 
69  FOR(i) { out.Mom[i] = -(GR_SPACEDIM - 1.) * d1.K[i] / GR_SPACEDIM; }
70  FOR(i, j, k)
71  {
72  out.Mom[i] += h_UU[j][k] *
73  (covd_A[k][j][i] - GR_SPACEDIM * vars.A[i][j] *
74  d1.chi[k] / (2 * chi_regularised));
75  }
76 
77  return out;
78 }
