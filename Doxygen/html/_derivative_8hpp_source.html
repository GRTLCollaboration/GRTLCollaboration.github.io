68 bool operator<(const Derivative &rhs) const
69 {
70 int derivs = 0;
71 int rhs_derivs = 0;
72
73 for (int i = 0; i < CH_SPACEDIM; ++i)
74 {
75 derivs += (*this)[i];
76 rhs_derivs += rhs[i];
77 }
78
79 if (derivs < rhs_derivs)
80 {
81 return true;
82 }
83 else if (derivs > rhs_derivs)
84 {
85 return false;
86 }
87 else
88 {
89 for (int i = 0; i < CH_SPACEDIM; ++i)
90 {
91 // This is counterintuitive but is actually the ordering the we
92 // want in order to generalise to arbitrary #dims
93 if ((*this)[i] > rhs[i])
94 {
95 return true;
96 }
97 else if ((*this)[i] < rhs[i])
98 {
99 return false;
100 }
101 }
102
103 return false;
104 }
105 }
