24 template <class data_t>
25 ALWAYS_INLINE data_t compute_determinant_sym(const Tensor<2, data_t, 3> &matrix)
26 {
27  data_t det = matrix[0][0] * matrix[1][1] * matrix[2][2] +
28  2 * matrix[0][1] * matrix[0][2] * matrix[1][2] -
29  matrix[0][0] * matrix[1][2] * matrix[1][2] -
30  matrix[1][1] * matrix[0][2] * matrix[0][2] -
31  matrix[2][2] * matrix[0][1] * matrix[0][1];
32 
33  return det;
34 }

38 template <class data_t>
39 ALWAYS_INLINE data_t compute_determinant(const Tensor<2, data_t, 3> &matrix)
40 {
41  data_t det =
42  matrix[0][0] *
43  (matrix[1][1] * matrix[2][2] - matrix[1][2] * matrix[2][1]) -
44  matrix[0][1] *
45  (matrix[2][2] * matrix[1][0] - matrix[1][2] * matrix[2][0]) +
46  matrix[0][2] *
47  (matrix[1][0] * matrix[2][1] - matrix[1][1] * matrix[2][0]);
48  return det;

52 template <class data_t>
53 Tensor<2, data_t> compute_inverse_sym(const Tensor<2, data_t, 3> &matrix)
54 {
55  data_t deth = compute_determinant_sym(matrix);
56  data_t deth_inverse = 1. / deth;
57  Tensor<2, data_t> h_UU;
58  h_UU[0][0] = (matrix[1][1] * matrix[2][2] - matrix[1][2] * matrix[1][2]) *
59  deth_inverse;
60  h_UU[0][1] = (matrix[0][2] * matrix[1][2] - matrix[0][1] * matrix[2][2]) *
61  deth_inverse;
62  h_UU[0][2] = (matrix[0][1] * matrix[1][2] - matrix[0][2] * matrix[1][1]) *
63  deth_inverse;
64  h_UU[1][1] = (matrix[0][0] * matrix[2][2] - matrix[0][2] * matrix[0][2]) *
65  deth_inverse;
66  h_UU[1][2] = (matrix[0][1] * matrix[0][2] - matrix[0][0] * matrix[1][2]) *
67  deth_inverse;
68  h_UU[2][2] = (matrix[0][0] * matrix[1][1] - matrix[0][1] * matrix[0][1]) *
69  deth_inverse;
70  h_UU[1][0] = h_UU[0][1];
71  h_UU[2][0] = h_UU[0][2];
72  h_UU[2][1] = h_UU[1][2];
73 
74  return h_UU;
75 }

164 ALWAYS_INLINE void make_trace_free(Tensor<2, data_t> &tensor_LL,
165  const Tensor<2, data_t> &metric,
166  const Tensor<2, data_t> &inverse_metric)
167 {

287 template <class data_t>
288 chris_t<data_t>
289 compute_christoffel(const Tensor<2, Tensor<1, data_t>> &d1_metric,
290  const Tensor<2, data_t> &h_UU)
291 {
292  chris_t<data_t> out;
293 
294  FOR(i, j, k)
295  {
296  out.LLL[i][j][k] = 0.5 * (d1_metric[j][i][k] + d1_metric[k][i][j] -
297  d1_metric[j][k][i]);
298  }
299  FOR(i, j, k)
300  {
301  out.ULL[i][j][k] = 0;
302  FOR(l) { out.ULL[i][j][k] += h_UU[i][l] * out.LLL[l][j][k]; }
303  }
304  FOR(i)
305  {
306  out.contracted[i] = 0;
307  FOR(j, k) { out.contracted[i] += h_UU[j][k] * out.ULL[i][j][k]; }
308  }
309 
310  return out;
311 }

313 template <class data_t>
314 Tensor<3, data_t> compute_phys_chris(const Tensor<1, data_t> &d1_chi,
315  const data_t &vars_chi,
316  const Tensor<2, data_t> &vars_h,
317  const Tensor<2, data_t> &h_UU,
318  const Tensor<3, data_t> &chris_ULL)
319 {
320  Tensor<3, data_t> chris_phys;
321  FOR(i, j, k)
322  {
323  chris_phys[i][j][k] =
324  chris_ULL[i][j][k] -
325  0.5 / vars_chi *
326  (delta(i, k) * d1_chi[j] + delta(i, j) * d1_chi[k]);
327  FOR(m)
328  {
329  chris_phys[i][j][k] +=
330  0.5 / vars_chi * vars_h[j][k] * h_UU[i][m] * d1_chi[m];
331  }
332  }
333  return chris_phys;
