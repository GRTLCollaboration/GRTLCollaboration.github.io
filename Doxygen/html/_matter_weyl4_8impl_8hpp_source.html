14template <class data_t>
15void MatterWeyl4<matter_t>::compute(Cell<data_t> current_cell) const
16{
17
18 // copy data from chombo gridpoint into local variables
19 const auto vars = current_cell.template load_vars<Vars>();
20 const auto d1 = m_deriv.template diff1<Vars>(current_cell);
21 const auto d2 = m_deriv.template diff2<Diff2Vars>(current_cell);
22
23 // Get the coordinates
24 const Coordinates<data_t> coords(current_cell, m_dx, m_center);
25
26 // Compute the inverse metric
27 using namespace TensorAlgebra;
28 const auto h_UU = compute_inverse_sym(vars.h);
29 const auto chris = compute_christoffel(d1.h, h_UU);
30
31 // Compute the spatial volume element
32 const auto epsilon3_LUU = compute_epsilon3_LUU(vars, h_UU);
33
34 // Compute the E and B fields
35 EBFields_t<data_t> ebfields =
36 compute_EB_fields(vars, d1, d2, epsilon3_LUU, h_UU, chris);
37
38 // Add in matter terms to E and B fields
39 add_matter_EB(ebfields, vars, d1, epsilon3_LUU, h_UU, chris);
40
41 // work out the Newman Penrose scalar
42 NPScalar_t<data_t> out =
43 compute_Weyl4(ebfields, vars, d1, d2, h_UU, coords);
44
45 // Write the rhs into the output FArrayBox
46 current_cell.store_vars(out.Real, c_Weyl4_Re);
47 current_cell.store_vars(out.Im, c_Weyl4_Im);
48}

52void MatterWeyl4<matter_t>::add_matter_EB(EBFields_t<data_t> &ebfields,
53 const Vars<data_t> &vars,
54 const Vars<Tensor<1, data_t>> &d1,
55 const Tensor<3, data_t> &epsilon3_LUU,
56 const Tensor<2, data_t> &h_UU,
57 const chris_t<data_t> &chris) const
58{
59 // Calculate decomposed energy momentum tensor components
60 const auto emtensor = m_matter.compute_emtensor(vars, d1, h_UU, chris.ULL);
61
62 Tensor<2, data_t> Sij_TF = emtensor.Sij;
63 TensorAlgebra::make_trace_free(Sij_TF, vars.h, h_UU);
64
65 // as we made the vacuum expression of Bij explictly symmetric and Eij
66 // explictly trace-free, only Eij has matter terms
67 FOR(i, j) { ebfields.E[i][j] += -4.0 * M_PI * m_G_Newton * Sij_TF[i][j]; }
68}
