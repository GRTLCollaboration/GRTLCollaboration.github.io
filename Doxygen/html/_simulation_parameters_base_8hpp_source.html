102  pp.load("num_points_theta", extraction_params.num_points_theta, 5);

112  if (pp.contains("modes"))

114  pp.load("num_modes", extraction_params.num_modes);

123  extraction_modes_vect[2 * i];
124  extraction_params.modes[i].second =
125  extraction_modes_vect[2 * i + 1];

127  }
128  else

130  // by default extraction (l,m) = (2,0), (2,1) and (2,2)
131  extraction_params.num_modes = 3;

133  for (int i = 0; i < 3; ++i)
134  {
135  extraction_params.modes[i].first = 2;
136  extraction_params.modes[i].second = i;
137  }
138  }
139 
140  pp.load("write_extraction", extraction_params.write_extraction,
141  false);

145  if (!extraction_path.empty() && extraction_path.back() != '/')
146  extraction_path += "/";

163  void check_params()
164  {
165  check_parameter("dt_multiplier", dt_multiplier, dt_multiplier < 1.0,
166  "must be < 1.0 for stability");
167  warn_parameter("dt_multiplier", dt_multiplier, dt_multiplier <= 0.5,
168  "is unlikely to be stable for > 0.5");
169 
170  check_parameter("sigma", sigma,
171  (sigma >= 0.0) && (sigma <= 2.0 / dt_multiplier),
172  "must be >= 0.0 and <= 2 / dt_multiplier for stability "
173  "(see Alcubierre p344)");
174  warn_parameter("nan_check", nan_check, nan_check,
175  "should not normally be disabled");
176  // not sure these are necessary hence commented out
177  // check_parameter("min_chi", min_chi, (min_chi >= 0.0), "must be >=
178  // 0.0"); check_parameter("min_lapse", min_lapse, (min_lapse >= 0.0)
179  // "must be >= 0.0");
180  check_parameter("formulation", formulation,
181  (formulation == CCZ4RHS<>::USE_CCZ4) ||
182  (formulation == CCZ4RHS<>::USE_BSSN),
183  "must be 0 or 1");
184  if (formulation == CCZ4RHS<>::USE_CCZ4)
185  {
186  warn_parameter(
187  "kappa1", ccz4_params.kappa1, ccz4_params.kappa1 > 0.0,
188  "should be greater than 0.0 to damp constraints (see "
189  "arXiv:1106.2254).");
190  warn_parameter("kappa2", ccz4_params.kappa2,
191  ccz4_params.kappa2 > -1.0,
192  "should be greater than -1.0 to damp constraints "
193  "(see arXiv:1106.2254)");
194  }
195  else if (formulation == CCZ4RHS<>::USE_BSSN)
196  {
197  // maybe we should just set these to zero and print a warning
198  // in the BSSN case
199  warn_parameter("kappa1", ccz4_params.kappa1,
200  ccz4_params.kappa1 == 0.0,
201  "setting to 0.0 as required for BSSN");
202  warn_parameter("kappa2", ccz4_params.kappa2,
203  ccz4_params.kappa2 == 0.0,
204  "setting to 0.0 as required for BSSN");
205  warn_parameter("kappa3", ccz4_params.kappa3,
206  ccz4_params.kappa3 == 0.0,
207  "setting to 0.0 as required for BSSN");
208  // no warning necessary for ccz4_params.covariantZ4
209  ccz4_params.kappa1 = 0.0;
210  ccz4_params.kappa2 = 0.0;
211  ccz4_params.kappa3 = 0.0;
212  }
213 
214  // only warn for gauge parameters as there are legitimate cases you may
215  // want to deviate from the norm
216  warn_parameter("lapse_advec_coeff", ccz4_params.lapse_advec_coeff,
217  min(abs(ccz4_params.lapse_advec_coeff),
218  abs(ccz4_params.lapse_advec_coeff - 1.0)) <
219  std::numeric_limits<double>::epsilon(),
220  "usually set to 0.0 or 1.0");
221  warn_parameter("lapse_power", ccz4_params.lapse_power,
222  abs(ccz4_params.lapse_power - 1.0) <
223  std::numeric_limits<double>::epsilon(),
224  "set to 1.0 for 1+log slicing");
225  warn_parameter("lapse_coeff", ccz4_params.lapse_coeff,
226  abs(ccz4_params.lapse_coeff - 2.0) <
227  std::numeric_limits<double>::epsilon(),
228  "set to 2.0 for 1+log slicing");
229  warn_parameter("shift_Gamma_coeff", ccz4_params.shift_Gamma_coeff,
230  abs(ccz4_params.shift_Gamma_coeff - 0.75) <
231  std::numeric_limits<double>::epsilon(),
232  "usually set to 0.75");
233  warn_parameter("shift_advec_coeff", ccz4_params.shift_advec_coeff,
234  min(abs(ccz4_params.shift_advec_coeff),
235  abs(ccz4_params.shift_advec_coeff - 1.0)) <
236  std::numeric_limits<double>::epsilon(),
237  "usually set to 0.0 or 1.0");
238  warn_parameter("eta", ccz4_params.eta,
239  ccz4_params.eta > 0.1 && ccz4_params.eta < 10,
240  "usually O(1/M_ADM) so typically O(1) in code units");
241 
242  // Now extraction parameters
243  if (activate_extraction)
244  {
245  check_parameter(
246  "num_extraction_radii", extraction_params.num_extraction_radii,
247  extraction_params.num_extraction_radii > 0,
248  "must be bigger than 0 when activate_extraction = 1");
249 
250  FOR(idir)
251  {
252  std::string center_name =
253  "extraction_center[" + std::to_string(idir) + "]";
254  double center_in_dir = extraction_params.center[idir];
255  check_parameter(
256  center_name, center_in_dir,
257  (center_in_dir >= reflective_domain_lo[idir]) &&
258  (center_in_dir <= reflective_domain_hi[idir]),
259  "must be in the computational domain after "
260  "applying reflective symmetry");
261  for (int iradius = 0;
262  iradius < extraction_params.num_extraction_radii;
263  ++iradius)
264  {
265  std::string radius_name =
266  "extraction_radii[" + std::to_string(iradius) + "]";
267  double radius = extraction_params.extraction_radii[iradius];
268  if (idir == 0)
269  check_parameter(radius_name, radius, radius >= 0.0,
270  "must be >= 0.0");
271  check_parameter(
272  radius_name, radius,
273  (center_in_dir - radius >=
274  reflective_domain_lo[idir]) &&
275  (center_in_dir + radius <=
276  reflective_domain_hi[idir]),
277  "extraction sphere must lie within the computational "
278  "domain after applying reflective symmetry");
279  }
280  }
281  for (int imode = 0; imode < extraction_params.num_modes; ++imode)
282  {
283  auto &mode = extraction_params.modes[imode];
284  int l = mode.first;
285  int m = mode.second;
286  std::string mode_name = "modes[" + std::to_string(imode) + "]";
287  std::string value_str = "(" + std::to_string(mode.first) +
288  ", " + std::to_string(mode.second) +
289  ")";
290  check_parameter(
291  mode_name, value_str, (l >= 2) && (abs(m) <= l),
292  "l must be >= 2 and m must satisfy -l <= m <= l");
293  }
294  }
295  }
