42 ChiAndPhiTaggingCriterion(const double dx, const double threshold_chi,

44 : m_dx(dx), m_deriv(dx), m_threshold_chi(threshold_chi),

46
47 template <class data_t> void compute(Cell<data_t> current_cell) const
48 {
49 const auto d2 = m_deriv.template diff2<MatterVars>(current_cell);
50 const auto d2chi = m_deriv.template diff2<Vars>(current_cell);
51
52 data_t mod_d2_chi = 0;
53 data_t mod_d2_phi = 0;
54
55 FOR(idir, jdir)
56 {
57 mod_d2_chi += d2chi.chi[idir][jdir] * d2chi.chi[idir][jdir];
58
59 mod_d2_phi += d2.Pi[idir][jdir] * d2.Pi[idir][jdir] +
60 d2.phi[idir][jdir] * d2.phi[idir][jdir];
61 }
62
63 data_t criterion_chi = m_dx / m_threshold_chi * sqrt(mod_d2_chi);
64
65 data_t criterion_phi = m_dx / m_threshold_phi * sqrt(mod_d2_phi);
66
67 data_t criterion = simd_max(criterion_chi, criterion_phi);
68
69 // Write back into the flattened Chombo box
70 current_cell.store_vars(criterion, 0);
71 }
72};
