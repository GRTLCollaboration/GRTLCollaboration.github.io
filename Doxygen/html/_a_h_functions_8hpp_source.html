120 ExpansionFunction(const AHVarsData<int, double> &a_data,
121 const Tensor<1, double> &a_coords,
122 const Tensor<1, double> &a_coords_cartesian)
123 {
124 CH_TIME("ExpansionFunction::calculate_data");
125
126 f = a_coords[CH_SPACEDIM - 1];
127
128 // * ---------------------------
129 // * GR-RELATED DATA
130 // * ---------------------------
131
132 int h[CH_SPACEDIM][CH_SPACEDIM];
133 int A[CH_SPACEDIM][CH_SPACEDIM];
134
135 int comp_h = c_h11;
136 int comp_A = c_A11;
137 for (int i = 0; i < CH_SPACEDIM; ++i)
138 {
139 for (int j = i; j < CH_SPACEDIM; ++j)
140 {
141 h[i][j] = comp_h;
142 A[i][j] = comp_A;
143 if (i != j)
144 {
145 h[j][i] = comp_h;
146 A[j][i] = comp_A;
147 }
148 ++comp_h;
149 ++comp_A;
150 }
151 }
152
153 const double chi = a_data.vars.at(c_chi);
154 trK = a_data.vars.at(c_K);
155
156 // INVERSE METRIC
157 Tensor<2, double, CH_SPACEDIM> h_DD;
158 FOR(i, j) { h_DD[i][j] = a_data.vars.at(h[i][j]); }
159
160 Tensor<2, double, CH_SPACEDIM> h_UU =
161 TensorAlgebra::compute_inverse_sym(h_DD);
162
163 // Reconstructing ADM variables
164 Tensor<1, double, CH_SPACEDIM> dchi;
165
166 FOR(i) { dchi[i] = a_data.d1.at(c_chi)[i]; }
167
168 for (int i = 0; i < CH_SPACEDIM; ++i)
169 {
170 for (int j = i; j < CH_SPACEDIM; ++j)
171 {
172 {
173 const double gij = h_DD[i][j] / chi;
174 g[i][j] = gij;
175 g[j][i] = gij;
176 }
177 {
178 const double g_UUij = h_UU[i][j] * chi;
179 g_UU[i][j] = g_UUij;
180 g_UU[j][i] = g_UUij;
181 }
182
183 {
184 const double Aij = a_data.vars.at(A[i][j]);
185 const double Kij = Aij / chi + trK * g[i][j] / GR_SPACEDIM;
186 K[i][j] = Kij;
187 K[j][i] = Kij;
188 }
189
190 for (int k = 0; k < CH_SPACEDIM; ++k)
191 {
192 {
193 const double dhij = a_data.d1.at(h[i][j])[k];
194 const double dgijk =
195 (dhij - (h_DD[i][j] * dchi[k]) / chi) / chi;
196 dg[i][j][k] = dgijk;
197 dg[j][i][k] = dgijk;
198 }
199 }
200 }
201 }
202
203 // part for higher dimensions that use Cartoon Method
204 // Uli's paper does it for 3+1D (from 5D) - arxiv 1808.05834
205#if GR_SPACEDIM != CH_SPACEDIM
206 int comp_hww =
207 c_K - 1; // c_K-1 expected to be c_hww (is c_hww direct better?)
208 int comp_Aww = c_Aww; // is (c_Theta-1) more general?
209 Tensor<1, double, CH_SPACEDIM> dhww;
210
211 coords = a_coords_cartesian;
212
213 double hww = a_data.vars.at(comp_hww);
214 FOR(a) { dhww[a] = a_data.d1.at(comp_hww)[a]; }
215 double Aww = a_data.vars.at(comp_Aww);
216
217 g_hd = hww / chi;
218 FOR(a) { dg_hd[a] = (dhww[a] - (hww * dchi[a]) / chi) / chi; }
219
220 Kww = Aww / chi + trK * g_hd / GR_SPACEDIM;
221#endif
222 }

234 double get(const AHGeometryData &geo_data, const AHDerivData &deriv,
235 const params &a_params) const
236 {
237 Tensor<1, double> s_L = get_level_function_derivative(geo_data, deriv);
238 Tensor<2, double> Ds =
239 get_level_function_2nd_covariant_derivative(geo_data, deriv, s_L);
240 double norm_s;
241 Tensor<1, double> S_U;
242 get_spatial_normal_U_and_norm(S_U, norm_s, s_L);
243
244 // calculate D_i S^i and S^i S^j K_ij
245 double DiSi = 0.;
246 double Kij_dot_Si_Sj = 0.;
247 FOR(a, b)
248 {
249 DiSi += (g_UU[a][b] - S_U[a] * S_U[b]) * Ds[a][b] / norm_s;
250 Kij_dot_Si_Sj += S_U[a] * S_U[b] * K[a][b];
251 }
252
253 // Calculation of expansion - as in (6.7.9 / 6.7.13) of Alcubierre
254 double expansion = DiSi - trK + Kij_dot_Si_Sj;
255
256 // part from extra dimensions in the case of Cartoon methods
257#if GR_SPACEDIM != CH_SPACEDIM
258 expansion += (GR_SPACEDIM - CH_SPACEDIM) * S_U[CH_SPACEDIM - 1] /
259 coords[CH_SPACEDIM - 1];
260 FOR(a)
261 {
262 expansion +=
263 (GR_SPACEDIM - CH_SPACEDIM) * 0.5 * dg_hd[a] / g_hd * S_U[a];
264 }
265#endif
266
267 // using "r * Expansion" significantly improves the convergence
268 // (making a Schw. BH converge for any radius >~ 0.5*r_AH instead of
269 // only up to ~ 3 * r_AH as it happens just with the expansion)
270 // (see arXiv:gr-qc/0512169, around Fig 3.4)
271 return expansion * pow(f, a_params.expansion_radius_power);
272 }

275 Tensor<1, double>
276 get_level_function_derivative(const AHGeometryData &geo_data,
277 const AHDerivData &deriv) const
278 {
279 // calculate D_a L of 6.7.12 of Alcubierre for some level function
280 // L picking L = f - F(u,v)
281 // D_a L = d_a f - dF/du * du/dx^a - dF/dv * dv/dx^a
282
283 Tensor<1, double> s_L = {0.}; // not normalized, just D_a L
284 FOR(a)
285 {
286 s_L[a] = D_TERM(geo_data.df[a], -(deriv.duF * geo_data.du[a]),
287 -(deriv.dvF * geo_data.dv[a]));
288 }
289
290 return s_L;
291 }
292 Tensor<1, double> get_spatial_normal_U(const Tensor<1, double> &s_L) const
293 {
294 Tensor<1, double> S_U;
295 double norm_s;
296 get_spatial_normal_U_and_norm(S_U, norm_s, s_L);
297 return S_U;
298 }
299 void get_spatial_normal_U_and_norm(Tensor<1, double> &S_U, double &norm_s,
300 const Tensor<1, double> &s_L) const
301 {
302 // calculate S_U = the real 's' of Alcubierre
303
304 // norm of s_L = | D_a L| (the 'u' in 6.7.12 of Alcubierre)
305 norm_s = 0.0;
306 FOR(a, b) { norm_s += g_UU[a][b] * s_L[a] * s_L[b]; }
307 norm_s = sqrt(norm_s);
308
309 // raise s_L
310 Tensor<1, double> s_U = {0.};
311 FOR(a, b) { s_U[a] += g_UU[a][b] * s_L[b]; }
312
313 FOR(a) { S_U[a] = s_U[a] / norm_s; }
314 }
315
316 Tensor<2, double> get_level_function_2nd_covariant_derivative(
317 const AHGeometryData &geo_data, const AHDerivData &deriv,
318 const Tensor<1, double> &s_L) const
319 {
320 // calculates D_a D_b L, required for 6.7.13 of Alcubierre
321 // for the level function L = f - F(u,v)
322
323 Tensor<2, double> ds = {0.};
324 FOR(a, b)
325 {
326 ds[a][b] =
327 D_TERM(geo_data.ddf[a][b],
328 -(deriv.duF * geo_data.ddu[a][b]) -
329 (deriv.duduF * geo_data.du[a] * geo_data.du[b]),
330 -(deriv.dvF * geo_data.ddv[a][b]) -
331 (deriv.dvdvF * geo_data.dv[a] * geo_data.dv[b]) -
332 (deriv.dudvF * (geo_data.du[a] * geo_data.dv[b] +
333 geo_data.du[b] * geo_data.dv[a])));
334 }
335
336 // calculate Christoffels on this point (u,v,f)
337 Tensor<3, double> chris = {0.};
338 FOR(a, b, c, d)
339 {
340 chris[a][b][c] +=
341 0.5 * g_UU[a][d] * (dg[b][d][c] + dg[c][d][b] - dg[b][c][d]);
342 }
343
344 // covariant derivatrive of s_a to use for DS
345 Tensor<2, double> Ds = {0.};
346 FOR(a, b)
347 {
348 Ds[a][b] = ds[a][b];
349 FOR(c) { Ds[a][b] -= chris[c][a][b] * s_L[c]; }
350 }
351
352 return Ds;
353 }
