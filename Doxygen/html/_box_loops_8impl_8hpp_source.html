18 template <typename... compute_ts>
19 void BoxLoops::innermost_loop(const ComputePack<compute_ts...> &compute_pack,
20  const BoxPointers &box_pointers, const int iy,
21  const int iz, const int loop_lo_x,
22  const int loop_hi_x)
23 {
24  int simd_width = simd<double>::simd_len;
25  int x_simd_max =
26  loop_lo_x +
27  simd_width * (((loop_hi_x - loop_lo_x + 1) / simd_width) - 1);
28 // SIMD LOOP
29 #ifdef __INTEL_COMPILER
30 #pragma novector
31 #else
32 #pragma omp simd safelen(1)
33 #endif /* __INTEL_COMPILER */
34  for (int ix = loop_lo_x; ix <= x_simd_max; ix += simd_width)
35  {
36  compute_pack.call_compute(
37  Cell<simd<double>>(IntVect(ix, iy, iz), box_pointers));
38  }
39 
40  // REMAINDER LOOP
41  for (int ix = x_simd_max + simd<double>::simd_len; ix <= loop_hi_x; ++ix)
42  {
43  compute_pack.call_compute(
44  Cell<double>(IntVect(ix, iy, iz), box_pointers));
45  }
46 }

61 template <typename... compute_ts, typename... simd_info>
62 void BoxLoops::loop(const ComputePack<compute_ts...> &compute_pack,
63  const FArrayBox &in, FArrayBox &out, const Box &loop_box,
64  simd_info... info)
65 {
66  // Makes sure we are not requesting data outside the box of 'out'
67  CH_assert(out.box().contains(loop_box));
68 
69  BoxPointers box_pointers(in, out);
70 
71  const int *loop_lo = loop_box.loVect();
72  const int *loop_hi = loop_box.hiVect();
73 
74 #pragma omp parallel for default(shared) collapse(CH_SPACEDIM - 1)
75 #if CH_SPACEDIM >= 3
76  for (int iz = loop_lo[2]; iz <= loop_hi[2]; ++iz)
77 #endif
78  for (int iy = loop_lo[1]; iy <= loop_hi[1]; ++iy)
79  {
80 #ifdef EQUATION_DEBUG_MODE
81  innermost_loop(compute_pack, box_pointers, iy, iz, loop_lo[0],
82  loop_hi[0], disable_simd());
83 #else
84  innermost_loop(compute_pack, box_pointers, iy, iz, loop_lo[0],
85  loop_hi[0], std::forward<simd_info>(info)...);
86 #endif
87  }
88 }

115 template <typename... compute_ts, typename... simd_info>
116 void BoxLoops::loop(const ComputePack<compute_ts...> &compute_pack,
117  const LevelData<FArrayBox> &in, LevelData<FArrayBox> &out,
118  bool fill_ghosts, simd_info... info)
119 {
120  DataIterator dit0 = in.dataIterator();
121  int nbox = dit0.size();
122  for (int ibox = 0; ibox < nbox; ++ibox)
123  {
124  DataIndex di = dit0[ibox];
125  const FArrayBox &in_fab = in[di];
126  FArrayBox &out_fab = out[di];
127 
128  Box out_box;
129  if (fill_ghosts)
130  out_box = out_fab.box();
131  else
132  out_box = in.disjointBoxLayout()[di];
133 
134  loop(compute_pack, in_fab, out_fab, out_box,
135  std::forward<simd_info>(info)...);
136  }
137 }
