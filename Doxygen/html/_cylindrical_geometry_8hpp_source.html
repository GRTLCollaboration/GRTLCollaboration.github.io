22class CylindricalGeometry
23{
24 private:

28 public:
29 CylindricalGeometry(const std::array<double, CH_SPACEDIM> &a_center,
30 double a_z_length)
31 : m_center(a_center), m_z_length(a_z_length)
32 {
33 }
34
36 inline double du(int a_num_points_z) const
37 {
38 return m_z_length / (a_num_points_z - 1);
39 }
40
42 inline double dv(int a_num_points_phi) const
43 {
44 return 2.0 * M_PI / ((double)a_num_points_phi);
45 }
46
48 inline double u(int a_iz, int a_num_points_z) const
49 {
50 return a_iz * du(a_num_points_z) - (m_z_length / 2.0);
51 }
52
54 inline double v(int a_iphi, int a_num_points_phi) const
55 {
56 return a_iphi * dv(a_num_points_phi);
57 }
58
59 inline bool is_u_periodic() const { return false; }
60 inline bool is_v_periodic() const { return true; }
63 inline double get_grid_coord(int a_dir, double a_radius, double a_z,
64 double a_phi) const
65 {
66 switch (a_dir)
67 {
68 case (0):
69 return m_center[0] + a_radius * cos(a_phi);
70 case (1):
71 return m_center[1] + a_radius * sin(a_phi);
72 case (2):
73 return m_center[2] + a_z;
74 default:
75 MayDay::Error("CylindricalGeometry: Direction not supported");
76 }
77 }
78
81 inline double area_element(double a_radius, double a_z, double a_phi) const
82 {
83 return a_radius;
84 }
85

87
88 inline std::string u_name() const { return "z"; }
89
90 inline std::string v_name() const { return "phi"; }
