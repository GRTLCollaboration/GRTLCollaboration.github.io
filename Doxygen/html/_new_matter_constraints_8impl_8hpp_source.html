27 void MatterConstraints<matter_t>::compute(Cell<data_t> current_cell) const
28 {
29  // Load local vars and calculate derivs
30  const auto vars = current_cell.template load_vars<BSSNMatterVars>();
31  const auto d1 = m_deriv.template diff1<BSSNMatterVars>(current_cell);
32  const auto d2 = m_deriv.template diff2<BSSNMatterVars>(current_cell);
33 
34  // Inverse metric and Christoffel symbol
35  const auto h_UU = TensorAlgebra::compute_inverse_sym(vars.h);
36  const auto chris = TensorAlgebra::compute_christoffel(d1.h, h_UU);
37 
38  // Get the non matter terms for the constraints
39  Vars<data_t> out = constraint_equations(vars, d1, d2, h_UU, chris);
40 
41  // Energy Momentum Tensor
42  const auto emtensor = my_matter.compute_emtensor(vars, d1, h_UU, chris.ULL);
43 
44  // Hamiltonian constraint
45  if (m_c_Ham >= 0 || m_c_Ham_abs_terms >= 0)
46  {
47  out.Ham += -16.0 * M_PI * m_G_Newton * emtensor.rho;
48  out.Ham_abs_terms += 16.0 * M_PI * m_G_Newton * abs(emtensor.rho);
49  }
50 
51  // Momentum constraints
52  if (m_c_Moms.size() > 0 || m_c_Moms_abs_terms.size() > 0)
53  {
54  FOR(i)
55  {
56  out.Mom[i] += -8.0 * M_PI * m_G_Newton * emtensor.Si[i];
57  out.Mom_abs_terms[i] +=
58  8.0 * M_PI * m_G_Newton * abs(emtensor.Si[i]);
59  }
60  }
61  // Write the constraints into the output FArrayBox
62  store_vars(out, current_cell);
63 }
