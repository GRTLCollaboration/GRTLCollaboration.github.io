80  static data_t get_radius(IntVect integer_coords, double dx,
81  std::array<double, CH_SPACEDIM> center = {0})
82  {
83  data_t xx;
84  double yy;

86 
87  // Note that this is not currently dimension independent
88  compute_coord(xx, integer_coords[0], dx, center[0]);
89  compute_coord(yy, integer_coords[1], dx, center[1]);
90  compute_coord(zz, integer_coords[2], dx, center[2]);
91 
92  data_t r = sqrt(xx * xx + yy * yy + zz * zz);
93
