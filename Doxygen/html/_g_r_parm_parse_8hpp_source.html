51 template <class data_t, long unsigned int n_comp>
52 void load(const char *name, std::array<data_t, n_comp> &array) const
53 {
54 getarr(name, ParmParseTranslator<data_t>::pp_type, array.data(), 0,
55 n_comp, -1);
56 }
57
59 template <class data_t>
60 void load(const char *name, std::vector<data_t> &vector,
61 const int num_comp) const
62 {
63 getarr(name, vector, 0, num_comp);
64 }

67 template <class data_t>
68 typename std::enable_if_t<
69 !std::is_enum<data_t>::value> // Can't use for enum types
70 load(const char *name, data_t &parameter) const

72 get(name, parameter);
73 }

76 template <typename enum_type>
77 typename std::enable_if_t<
78 std::is_enum<enum_type>::value> // Only enabled for enum types
79 load(const char *name, enum_type &parameter) const

81 int iparam;
82 get(name, iparam);
83 parameter = static_cast<enum_type>(iparam);

88 template <class data_t>
89 void load(const char *name, data_t &parameter,
90 const data_t default_value) const
91 {
92 if (contains(name))
93 {
94 load(name, parameter);
95 }
96 else
97 {
98 parameter = default_value;

143 std::enable_if_t<
144 ArrayTools::is_std_array_or_vector<data_t>::value,
145 bool> = true> // use this code for std::arrays and vectors
146 void default_message(const char *name, const data_t &default_value) const
147 {
148 pout() << "Parameter: " << name << " not found in parameter file. "
149 << "It has been set to its default "
150 "value =";
151 for (auto elem : default_value)

153 pout() << " " << elem;

155 pout() << "." << std::endl;
