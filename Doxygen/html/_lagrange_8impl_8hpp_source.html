12/* Finite difference weight generation algorithm
13 *
14 * Translated from the F77 code found in
15 * Bengt Fornberg: "Calculation of Weights in Finite Difference Formulas"
16 * SIAM Review 40, 3 (1998): 685-691
17 *
18 * Here we restrict ourselves to integer grid. Original code allows for
19 * arbitrary grid locations specified by grid[i]. To recover the general
20 * routine: - change type of c1,c2,c3 to 'double' - replace '0', 'i', 'j' in the
21 * annotated lines with 'grid[0]', 'grid[i]', 'grid[j]'.
22 */
23template <int Order>
24Lagrange<Order>::Stencil::Stencil(int width, int deriv, double dx,
25 double point_offset)
26 : m_width(width), m_deriv(deriv), m_dx(dx), m_point_offset(point_offset)
27{
28 int c1 = 1;
29 int c2;
30 int c3;
31 double c4 = 0 - m_point_offset; /* replace for general grid */
32 double c5;
33
34 std::vector<double> tmp_weights(width * (deriv + 1), 0);
35
36 tmp_weights[0] = 1.0;
37
38 for (int i = 1; i < m_width; ++i)
39 {
40 int mn = (i < m_deriv) ? i : m_deriv;
41
42 c2 = 1;
43 c5 = c4;
44 c4 = i - m_point_offset; /* replace for general grid */
45
46 for (int j = 0; j < i; ++j)
47 {
48 c3 = i - j; /* replace for general grid */
49 c2 = c2 * c3;
50
51 if (j == i - 1)
52 {
53 for (int k = mn; k > 0; --k)
54 {
55 tmp_weights[k * m_width + i] =
56 c1 *
57 (k * tmp_weights[(k - 1) * m_width + (i - 1)] -
58 c5 * tmp_weights[k * m_width + (i - 1)]) /
59 c2;
60 }
61 tmp_weights[i] = -c1 * c5 * tmp_weights[i - 1] / c2;
62 }
63
64 for (int k = mn; k > 0; --k)
65 {
66 tmp_weights[k * m_width + j] =
67 (c4 * tmp_weights[k * m_width + j] -
68 k * tmp_weights[(k - 1) * m_width + j]) /
69 c3;
70 }
71 tmp_weights[j] = c4 * tmp_weights[j] / c3;
72 }
73
74 c1 = c2;
75 }
76
77 if (deriv > 0)
78 {
79 const double dx_factor = pow(m_dx, deriv);
80 m_weights.resize(m_width);
81 for (int i = 0; i < m_width; ++i)
82 {
83 m_weights[i] = tmp_weights[m_width * m_deriv + i] / dx_factor;
84 }
85 }
86 else
87 {
88 m_weights = std::move(tmp_weights);
89 }
90
91 if (false)
92 {
93 pout() << TAG << "Created a stencil for deriv " << m_deriv
94 << " of width " << m_width << " for point " << m_point_offset
95 << endl;
96 pout() << " Weights = { ";
97 for (int i = 0; i < m_width; ++i)
98 {
99 pout() << m_weights[i] << " ";
100 }
101 pout() << "}" << endl;
102 }
103}

123template <int Order>
124typename Lagrange<Order>::Stencil
125Lagrange<Order>::getStencil(int width, int deriv, double dx,
126 double point_offset)
127{
128 for (typename stencil_collection_t::iterator it =
129 m_memoized_stencils.begin();
130 it != m_memoized_stencils.end(); ++it)
131 {
132 if (it->isSameAs(width, deriv, dx, point_offset))
133 {
134 // Make a copy, lest std::vector decides to move our stencil during
135 // growth op
136 return *it;
137 }
138 }
139
140 // We have to insert a new stencil.
141 return *m_memoized_stencils.insert(m_memoized_stencils.end(),
142 Stencil(width, deriv, dx, point_offset));
143}

186template <int Order>
187double Lagrange<Order>::interpData(const FArrayBox &fab, int comp)
188{
189 /*
190 m_interp_neg.clear();
191 m_interp_pos.clear();
192
193 // We are adding 200+ numbers at roughly the same magnitudes but alternating
194 signs.
195 // Let's keep track of positive and negative terms separately to make sure
196 we don't run into trouble. for (int i = 0; i < m_interp_points.size(); ++i)
197 {
198 double data = m_interp_weights[i] * fab.get(m_interp_points[i], comp);
199 if (data > 0)
200 {
201 m_interp_pos.insert(data);
202 }
203 else
204 {
205 m_interp_neg.insert(data);
206 }
207 }
208
209 // Add positive terms from smallest to largest
210 double pos = 0;
211 for (typename multiset<double>::iterator it = m_interp_pos.begin(); it !=
212 m_interp_pos.end(); ++it)
213 {
214 pos += *it;
215 }
216
217 // Largest negative term has smallest magnitude. Use reverse iterator.
218 double neg = 0;
219 for (typename multiset<double>::reverse_iterator it = m_interp_neg.rbegin();
220 it != m_interp_neg.rend(); ++it)
221 {
222 neg += *it;
223 }
224
225 return pos + neg;
226 */
227
228 long double accum = 0.0;
229
230 for (int i = 0; i < m_interp_points.size(); ++i)
231 {
232 double data = m_interp_weights[i] * fab.get(m_interp_points[i], comp);
233 accum += data;
234 }
235
236 return accum;
237}

239template <int Order>
240pair<std::vector<IntVect>, std::vector<double>>
241Lagrange<Order>::generateStencil(
242 const std::array<int, CH_SPACEDIM> &deriv,
243 const std::array<double, CH_SPACEDIM> &dx,
244 const std::array<double, CH_SPACEDIM> &evalCoord, const IntVect &nearest,
245 int dim)
246{
247 std::vector<IntVect> out_points;
248 std::vector<double> out_weights;
249
250 /*
251 * SCAN ALONG THIS DIRECTION TO FIND THE LARGEST CONTIGUOUS CHUNK OF VALID
252 * POINTS
253 */
254
255 // Allocate a std::vector twice as big as we can possibly need
256 // This way we insert to either direction without shifting/growing
257 std::vector<int> my_points(2 * (Order + deriv[dim]));
258
259 enum
260 {
261 DOWN,
262 UP
263 };
264
265 bool can_grow[2] = {true, true};
266 int points_min = Order + deriv[dim];
267 int points_max = Order + deriv[dim];
268
269 std::array<double, CH_SPACEDIM> interp_coord = evalCoord;
270 int candidate = nearest[dim];
271 int grown_direction = (nearest[dim] - evalCoord[dim] < 0) ? DOWN : UP;
272
273 while ((can_grow[DOWN] || can_grow[UP]) &&
274 (points_max - points_min < Order + deriv[dim]))
275 {
276 interp_coord[dim] = candidate;
277
278 if (m_source.contains(interp_coord))
279 {
280 int idx =
281 (grown_direction == DOWN) ? (--points_min) : (points_max++);
282 my_points[idx] = candidate;
283 }
284 else
285 {
286 can_grow[grown_direction] = false;
287 }
288
289 // Flip "grow" direction if we can do so
290 if (can_grow[1 - grown_direction])
291 {
292 grown_direction = 1 - grown_direction;
293 }
294
295 candidate = (grown_direction) ? (my_points[points_max - 1] + 1)
296 : (my_points[points_min] - 1);
297 }
298
299 int stencil_width = points_max - points_min;
300 CH_assert(stencil_width > 0);
301
302 const Stencil my_weights =
303 getStencil(stencil_width, deriv[dim], dx[dim],
304 evalCoord[dim] - my_points[points_min]);
305
306 if (m_verbosity)
307 {
308 pout() << TAG << "Stencil: dim = " << dim
309 << ", coord = " << evalCoord[dim] << ", points = { ";
310 for (int i = points_min; i < points_max; ++i)
311 {
312 pout() << my_points[i] << " ";
313 }
314 pout() << "}, weights = { ";
315 for (int i = 0; i < stencil_width; ++i)
316 {
317 pout() << my_weights[i] << " ";
318 }
319 pout() << "}" << endl;
320 }
321
322 // There is going to be potentially a LOT of temporary std::vectors getting
323 // allocated in here. If things get slow this will be a good place to look
324 // first.
325 for (int i = 0; i < stencil_width; ++i)
326 {
327 interp_coord[dim] = my_points[i + points_min];
328
329 if (dim > 0)
330 {
331 // Descend to the next dimension
332 pair<std::vector<IntVect>, std::vector<double>> sub_result =
333 generateStencil(deriv, dx, interp_coord, nearest, dim - 1);
334 std::vector<IntVect> &sub_points = sub_result.first;
335 std::vector<double> &sub_weights = sub_result.second;
336
337 // Take tensor product weights
338 for (int j = 0; j < sub_points.size(); ++j)
339 {
340 if (my_weights[i] != 0)
341 {
342 out_points.push_back(sub_points[j]);
343 out_weights.push_back(my_weights[i] * sub_weights[j]);
344 }
345 }
346 }
347 else
348 {
349 // "Terminal" dimension, just push back our own stuff
350 if (my_weights[i] != 0)
351 {
352 out_points.push_back(IntVect(D_DECL6(
353 interp_coord[0], interp_coord[1], interp_coord[2],
354 interp_coord[3], interp_coord[4], interp_coord[5])));
355 out_weights.push_back(my_weights[i]);
356 }
357 }
358 }
359
360 return pair<std::vector<IntVect>, std::vector<double>>(
361 std::move(out_points), std::move(out_weights));
362}
