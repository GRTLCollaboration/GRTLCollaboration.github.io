36  void read_params(GRParmParse &pp)
37  {
38  // must be before any pout() in the code to setPoutBaseName
39  read_filesystem_params(pp);
40 
41  pp.load("verbosity", verbosity, 0);
42  // Grid setup
43  pp.load("max_spatial_derivative_order", max_spatial_derivative_order,
44  4);
45  pp.load("num_ghosts", num_ghosts,
46  (max_spatial_derivative_order == 6) ? 4 : 3);
47  pp.load("tag_buffer_size", tag_buffer_size, 3);
48  pp.load("grid_buffer_size", grid_buffer_size, 8);
49  pp.load("dt_multiplier", dt_multiplier, 0.25);
50  pp.load("fill_ratio", fill_ratio, 0.7);
51 
52  // Periodicity and boundaries
53  boundary_params.read_params(pp);
54 
55  // L's, N's and center
56  read_grid_params(pp);
57 
58 #ifdef CH_USE_HDF5
59  pp.load("ignore_checkpoint_name_mismatch",
60  ignore_checkpoint_name_mismatch, false);
61 #endif
62 
63  pp.load("max_level", max_level, 0);
64  // the reference ratio is hard coded to 2
65  // in principle it can be set to other values, but this is
66  // not recommended since we do not test GRChombo with other
67  // refinement ratios - use other values at your own risk
68  ref_ratios.resize(max_level + 1);
69  ref_ratios.assign(2);
70  pp.getarr("regrid_interval", regrid_interval, 0, max_level);
71  // Regridding on max_level does nothing but Chombo's AMR class
72  // expects this Vector to be of length max_level + 1
73  // so just set the final value to 0.
74  regrid_interval.resize(max_level + 1);
75  regrid_interval[max_level] = 0;
76 
77  if (pp.contains("regrid_thresholds"))
78  {
79  pout() << "Using multiple regrid thresholds." << std::endl;
80  // As for regrid_interval, the last element is irrelevant
81  pp.getarr("regrid_thresholds", regrid_thresholds, 0, max_level);
82  regrid_thresholds.resize(max_level + 1);
83  regrid_thresholds[max_level] = regrid_thresholds[max_level - 1];
84  }
85  else
86  {
87  pout() << "Using single regrid threshold." << std::endl;
88  double regrid_threshold;
89  pp.load("regrid_threshold", regrid_threshold, 0.5);
90  regrid_thresholds = Vector<double>(max_level + 1, regrid_threshold);
91  }
92 
93  // time stepping outputs and regrid data
94  pp.load("checkpoint_interval", checkpoint_interval, 1);
95  pp.load("plot_interval", plot_interval, 0);
96  pp.load("stop_time", stop_time, 1.0);
97  pp.load("max_steps", max_steps, 1000000);
98 #ifdef CH_USE_HDF5
99  pp.load("write_plot_ghosts", write_plot_ghosts, false);
100 
101  // load vars to write to plot files
102  UserVariables::load_vars_to_vector(pp, "plot_vars", "num_plot_vars",
103  plot_vars, num_plot_vars);
104 #endif
105  // alias the weird chombo names to something more descriptive
106  // for these box params, and default to some reasonable values
107  if (pp.contains("max_grid_size"))
108  {
109  pp.load("max_grid_size", max_grid_size);
110  }
111  else
112  {
113  pp.load("max_box_size", max_grid_size, 64);
114  }
115  if (pp.contains("block_factor"))
116  {
117  pp.load("block_factor", block_factor);
118  }
119  else
120  {
121  pp.load("min_box_size", block_factor, 8);
122  }
123 
124  if (pp.contains("check_params"))
125  just_check_params = true;
126 
127  pp.load("print_progress_only_to_rank_0", print_progress_only_to_rank_0,
128  false);
129  }

207  void read_grid_params(GRParmParse &pp)
208  {
209  // Grid N
210  std::array<int, CH_SPACEDIM> Ni_full;
211  std::array<int, CH_SPACEDIM> Ni;
212  ivN = IntVect::Unit;
213 
214  // cannot contain both
215  if ((pp.contains("N_full") && pp.contains("N")))
216  MayDay::Error("Please only provide 'N' or 'N_full', not both");
217 
218  int N_full = -1;
219  int N = -1;
220  if (pp.contains("N_full"))
221  pp.load("N_full", N_full);
222  else if (pp.contains("N"))
223  pp.load("N", N);
224 
225  // read all options (N, N_full, Ni_full and Ni) and then choose
226  // accordingly
227  FOR(dir)
228  {
229  std::string name = ("N" + std::to_string(dir + 1));
230  std::string name_full = ("N" + std::to_string(dir + 1) + "_full");
231  Ni_full[dir] = -1;
232  Ni[dir] = -1;
233 
234  // only one of them exists - this passes if none of the 4 exist, but
235  // that is asserted below
236  if (!((N_full > 0 || N > 0) && !pp.contains(name.c_str()) &&
237  !pp.contains(name_full.c_str())) &&
238  !((N_full < 0 && N < 0) && !(pp.contains(name.c_str()) &&
239  pp.contains(name_full.c_str()))))
240  error("Please provide 'N' or 'N_full' or a set of "
241  "'N1/N1_full', 'N2/N2_full', 'N3/N3_full'");
242 
243  if (N_full < 0 && N < 0)
244  {
245  if (pp.contains(name_full.c_str()))
246  pp.load(name_full.c_str(), Ni_full[dir]);
247  else
248  pp.load(name.c_str(), Ni[dir]);
249  }
250  if (N < 0 && N_full < 0 && Ni[dir] < 0 &&
251  Ni_full[dir] < 0) // sanity check
252  error("Please provide 'N' or 'N_full' or a set of "
253  "'N1/N1_full', 'N2/N2_full', 'N3/N3_full'");
254 
255  if (N_full > 0)
256  Ni_full[dir] = N_full;
257  else if (N > 0)
258  Ni[dir] = N;
259 
260  if (Ni[dir] > 0)
261  {
262  if (boundary_params.lo_boundary[dir] ==
263  BoundaryConditions::REFLECTIVE_BC ||
264  boundary_params.hi_boundary[dir] ==
265  BoundaryConditions::REFLECTIVE_BC)
266 
267  Ni_full[dir] = Ni[dir] * 2;
268  else
269  Ni_full[dir] = Ni[dir];
270  }
271  else
272  {
273  if (boundary_params.lo_boundary[dir] ==
274  BoundaryConditions::REFLECTIVE_BC ||
275  boundary_params.hi_boundary[dir] ==
276  BoundaryConditions::REFLECTIVE_BC)
277  {
278  check_parameter("N" + std::to_string(dir) + "_full",
279  Ni_full[dir], Ni_full[dir] % 2 == 0,
280  "must be a multiple of 2");
281 
282  Ni[dir] = Ni_full[dir] / 2;
283  }
284  else
285  Ni[dir] = Ni_full[dir];
286  }
287  ivN[dir] = Ni[dir] - 1;
288  }
289  int max_N_full = *std::max_element(Ni_full.begin(), Ni_full.end());
290  int max_N = ivN.max() + 1;
291 
292  // Grid L
293  // cannot contain both
294  if ((pp.contains("L_full") && pp.contains("L")))
295  error("Please only provide 'L' or 'L_full', not both");
296 
297  double L_full = -1.;
298  if (pp.contains("L_full"))
299  pp.load("L_full", L_full);
300  else
301  pp.load("L", L, 1.0);
302 
303  if (L_full > 0.)
304  // necessary for some reflective BC cases, as 'L' is the
305  // length of the longest side of the box
306  L = (L_full * max_N) / max_N_full;
307 
308  coarsest_dx = L / max_N;
309  coarsest_dt = coarsest_dx * dt_multiplier;
310 
311  // grid spacing params
312  dx.fill(coarsest_dx);
313  origin.fill(coarsest_dx / 2.0);
314 
315  // These aren't parameters but used in parameter checks
316  FOR(idir)
317  {
318  reflective_domain_lo[idir] = ((boundary_params.lo_boundary[idir] ==
319  BoundaryConditions::REFLECTIVE_BC)
320  ? -1.0
321  : 0.0) *
322  (ivN[idir] + 1) * coarsest_dx;
323  reflective_domain_hi[idir] = ((boundary_params.hi_boundary[idir] ==
324  BoundaryConditions::REFLECTIVE_BC)
325  ? 2.0
326  : 1.0) *
327  (ivN[idir] + 1) * coarsest_dx;
328  }
329 
330  // First work out the default center ignoring reflective BCs
331  // but taking into account different grid lengths in each direction
332  std::array<double, CH_SPACEDIM> default_center;
333 #if CH_SPACEDIM == 3
334  default_center = {0.5 * Ni[0] * coarsest_dx, 0.5 * Ni[1] * coarsest_dx,
335  0.5 * Ni[2] * coarsest_dx};
336 #elif CH_SPACEDIM == 2
337  default_center = {0.5 * Ni[0] * coarsest_dx, 0.5 * Ni[1] * coarsest_dx};
338 #endif
339  // Now take into account reflective BCs
340  FOR(idir)
341  {
342  if ((boundary_params.lo_boundary[idir] ==
343  BoundaryConditions::REFLECTIVE_BC) &&
344  (boundary_params.hi_boundary[idir] !=
345  BoundaryConditions::REFLECTIVE_BC))
346  default_center[idir] = 0.;
347  else if ((boundary_params.hi_boundary[idir] ==
348  BoundaryConditions::REFLECTIVE_BC) &&
349  (boundary_params.lo_boundary[idir] !=
350  BoundaryConditions::REFLECTIVE_BC))
351  default_center[idir] = coarsest_dx * Ni[idir];
352  }
353 
354  pp.load("center", center, default_center); // default to center
355  }

357  void check_params()
358  {
359  check_parameter("L", L, L > 0.0, "must be > 0.0");
360  check_parameter("max_level", max_level, max_level >= 0, "must be >= 0");
361  check_parameter("max_spatial_derivative_order",
362  max_spatial_derivative_order,
363  max_spatial_derivative_order == 4 ||
364  max_spatial_derivative_order == 6,
365  "only 4 and 6 are supported");
366  // the following check assumes you will be taking one-sided derivatives
367  // of the order given by max_spatial_derivative_order
368  check_parameter(
369  "num_ghosts", num_ghosts,
370  (num_ghosts >= ((max_spatial_derivative_order == 6) ? 4 : 3)) &&
371  (num_ghosts <= block_factor),
372  "must be >= 3 (4th order derivatives) or 4 (6th order derivatives) "
373  "and <= min_box_size (aka block_factor)");
374  check_parameter("tag_buffer_size", tag_buffer_size,
375  tag_buffer_size >= 0, "must be >= 0");
376  // assume ref_ratio is always 2
377  check_parameter(
378  "grid_buffer_size", grid_buffer_size,
379  grid_buffer_size >= ceil(num_ghosts / 2.0),
380  "must be >= ceil(num_ghosts/max_ref_ratio) for proper nesting");
381 
382  // check the restart_file exists and can be read if restarting from a
383  // checkpoint
384 #ifdef CH_USE_HDF5
385  if (restart_from_checkpoint)
386  {
387  bool restart_file_exists =
388  (access((hdf5_path + restart_file).c_str(), R_OK) == 0);
389  check_parameter("restart_file", restart_file, restart_file_exists,
390  "file cannot be opened for reading");
391  }
392 #endif
393 
394  check_parameter("dt_multiplier", dt_multiplier, dt_multiplier > 0.0,
395  "must be > 0.0");
396  check_parameter("max_grid_size/max_box_size", max_grid_size,
397  max_grid_size >= 0, "must be >= 0");
398  check_parameter("block_factor/min_box_size", block_factor,
399  block_factor >= 1, "must be >= 1");
400  check_parameter("block_factor/min_box_size", block_factor,
401  Misc::isPower2(block_factor), "must be a power of 2");
402  // note that this also enforces block_factor <= max_grid_size
403  // if max_grid_size > 0
404  check_parameter("block_factor/min_box_size", block_factor,
405  max_grid_size % block_factor == 0,
406  "must divide max_grid_size/max_box_size = " +
407  std::to_string(max_grid_size));
408  FOR(idir)
409  {
410  std::string Ni_string = "N" + std::to_string(idir + 1);
411  std::string invalid_message = "must divide " + Ni_string;
412  if (boundary_params.reflective_boundaries_exist)
413  {
414  invalid_message += " (or " + Ni_string + "_full/2)";
415  }
416  invalid_message += " = " + std::to_string(ivN[idir] + 1);
417  check_parameter("block_factor/min_box_size", block_factor,
418  (ivN[idir] + 1) % block_factor == 0,
419  invalid_message);
420  }
421  check_parameter("fill_ratio", fill_ratio,
422  (fill_ratio > 0.0) && (fill_ratio <= 1.0),
423  "must be > 0 and <= 1");
424  // (MR); while this would technically work (any plot files would just
425  // overwrite a checkpoint file), I think a user would only ever do
426  // this unintentinally
427 #ifdef CH_USE_HDF5
428  check_parameter("plot_prefix", plot_prefix,
429  plot_interval <= 0 || plot_prefix != checkpoint_prefix,
430  "should be different to checkpoint_prefix");
431 #endif
432 
433  check_parameter("output_path", output_path,
434  FilesystemTools::directory_exists(output_path),
435  "should be a valid directory");
436  // pout directory exists - we create it in read_filesystem_params()
437  // can't check hdf5 directory yet - only created after
438 
439  if (boundary_params.reflective_boundaries_exist)
440  {
441  for (int ivar = 0; ivar < NUM_VARS; ++ivar)
442  {
443  std::string name = "vars_parity[c_" +
444  UserVariables::variable_names[ivar] + "]";
445  int var_parity = boundary_params.vars_parity[ivar];
446  check_parameter(name, var_parity,
447  var_parity >= BoundaryConditions::EVEN &&
448  var_parity < BoundaryConditions::UNDEFINED,
449  "parity type undefined");
450  }
451  for (int ivar = 0; ivar < NUM_DIAGNOSTIC_VARS; ++ivar)
452  {
453  std::string name = "vars_parity_diagnostic[c_" +
454  DiagnosticVariables::variable_names[ivar] +
455  "]";
456  int var_parity = boundary_params.vars_parity_diagnostic[ivar];
457  check_parameter(name, var_parity,
458  var_parity >= BoundaryConditions::EVEN &&
459  var_parity <= BoundaryConditions::UNDEFINED,
460  "parity type undefined");
461  }
462  }
463  }
