15 // Computes semi-isotropic Kerr solution as detailed in Liu, Etienne and Shapiro
16 // 2010, arxiv gr-qc/1001.4077
17 template <class data_t> void KerrBH::compute(Cell<data_t> current_cell) const
18 {
19  // set up vars for the metric and extrinsic curvature, shift and lapse in
20  // spherical coords
21  Tensor<2, data_t> spherical_g;
22  Tensor<2, data_t> spherical_K;
23  Tensor<1, data_t> spherical_shift;
24  data_t kerr_lapse;
25 
26  // The cartesian variables and coords
27  Vars<data_t> vars;
28  Coordinates<data_t> coords(current_cell, m_dx, m_params.center);
29 
30  // Compute the components in spherical coords as per 1401.1548
31  compute_kerr(spherical_g, spherical_K, spherical_shift, kerr_lapse, coords);
32 
33  // work out where we are on the grid
34  data_t x = coords.x;
35  double y = coords.y;
36  double z = coords.z;
37 
38  using namespace CoordinateTransformations;
39  // Convert spherical components to cartesian components using coordinate
40  // transforms
41  vars.h = spherical_to_cartesian_LL(spherical_g, x, y, z);
42  vars.A = spherical_to_cartesian_LL(spherical_K, x, y, z);
43  vars.shift = spherical_to_cartesian_U(spherical_shift, x, y, z);
44 
45  using namespace TensorAlgebra;
46  // Convert to BSSN vars
47  data_t deth = compute_determinant(vars.h);
48  auto h_UU = compute_inverse_sym(vars.h);
49  vars.chi = pow(deth, -1. / 3.);
50 
51  // transform extrinsic curvature into A and TrK - note h is still non
52  // conformal version which is what we need here
53  vars.K = compute_trace(vars.A, h_UU);
54  make_trace_free(vars.A, vars.h, h_UU);
55 
56  // Make conformal
57  FOR(i, j)
58  {
59  vars.h[i][j] *= vars.chi;
60  vars.A[i][j] *= vars.chi;
61  }
62 
63  // use a pre collapsed lapse, could also use analytic one
64  // vars.lapse = kerr_lapse;
65  vars.lapse = pow(vars.chi, 0.5);
66 
67  // Populate the variables on the grid
68  // NB We stil need to set Gamma^i which is NON ZERO
69  // but we do this via a separate class/compute function
70  // as we need the gradients of the metric which are not yet available
71  current_cell.store_vars(vars);
72 }

74 template <class data_t>
75 void KerrBH::compute_kerr(Tensor<2, data_t> &spherical_g,
76  Tensor<2, data_t> &spherical_K,
77  Tensor<1, data_t> &spherical_shift,
78  data_t &kerr_lapse,
79  const Coordinates<data_t> coords) const
80 {
81  // Kerr black hole params - mass M and spin a
82  double M = m_params.mass;
83  double a = m_params.spin;
84 
85  // work out where we are on the grid
86  data_t x = coords.x;
87  double y = coords.y;
88  double z = coords.z;
89 
90  // the radius, subject to a floor
91  data_t r = coords.get_radius();
92  data_t r2 = r * r;
93 
94  // the radius in xy plane, subject to a floor
95  data_t rho2 = simd_max(x * x + y * y, 1e-12);
96  data_t rho = sqrt(rho2);
97 
98  // calculate useful position quantities
99  data_t cos_theta = z / r;
100  data_t sin_theta = rho / r;
101  data_t cos_theta2 = cos_theta * cos_theta;
102  data_t sin_theta2 = sin_theta * sin_theta;
103 
104  // calculate useful metric quantities
105  double r_plus = M + sqrt(M * M - a * a);
106  double r_minus = M - sqrt(M * M - a * a);
107 
108  // The Boyer-Lindquist coordinate
109  data_t r_BL = r * pow(1.0 + 0.25 * r_plus / r, 2.0);
110 
111  // Other useful quantities per 1001.4077
112  data_t Sigma = r_BL * r_BL + a * a * cos_theta2;
113  data_t Delta = r_BL * r_BL - 2.0 * M * r_BL + a * a;
114  // In the paper this is just 'A', but not to be confused with A_ij
115  data_t AA = pow(r_BL * r_BL + a * a, 2.0) - Delta * a * a * sin_theta2;
116  // The rr component of the conformal spatial matric
117  data_t gamma_rr =
118  Sigma * pow(r + 0.25 * r_plus, 2.0) / (r * r2 * (r_BL - r_minus));
119 
120  // Metric in semi isotropic Kerr-Schild coordinates, r, theta (t or th), phi
121  // (p)
122  FOR(i, j) { spherical_g[i][j] = 0.0; }
123  spherical_g[0][0] = gamma_rr; // gamma_rr
124  spherical_g[1][1] = Sigma; // gamma_tt
125  spherical_g[2][2] = AA / Sigma * sin_theta2; // gamma_pp
126 
127  // Extrinsic curvature
128  FOR(i, j) { spherical_K[i][j] = 0.0; }
129 
130  // set non zero elements of Krtp - K_rp, K_tp
131  spherical_K[0][2] =
132  a * M * sin_theta2 / (Sigma * sqrt(AA * Sigma)) *
133  (3.0 * pow(r_BL, 4.0) + 2 * a * a * r_BL * r_BL - pow(a, 4.0) -
134  a * a * (r_BL * r_BL - a * a) * sin_theta2) *
135  (1.0 + 0.25 * r_plus / r) / sqrt(r * r_BL - r * r_minus);
136  spherical_K[2][0] = spherical_K[0][2];
137  spherical_K[2][1] = -2.0 * pow(a, 3.0) * M * r_BL * cos_theta * sin_theta *
138  sin_theta2 / (Sigma * sqrt(AA * Sigma)) *
139  (r - 0.25 * r_plus) * sqrt(r_BL / r - r_minus / r);
140  spherical_K[1][2] = spherical_K[2][1];
141 
142  // set the analytic lapse
143  kerr_lapse = sqrt(Delta * Sigma / AA);
144 
145  // set the shift (only the phi component is non zero)
146  spherical_shift[0] = 0.0;
147  spherical_shift[1] = 0.0;
148  spherical_shift[2] = -2.0 * M * a * r_BL / AA;
149 }
