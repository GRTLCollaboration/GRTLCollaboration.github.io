21template <class matter_t>
22template <class data_t>
23void ChiRelaxation<matter_t>::compute(Cell<data_t> current_cell) const
24{
25
26 // copy data from chombo gridpoint into local variable and calculate derivs
27 const auto vars = current_cell.template load_vars<Vars>();
28 const auto d1 = m_deriv.template diff1<Vars>(current_cell);
29 const auto d2 = m_deriv.template diff2<Diff2Vars>(current_cell);
30 const auto advec =
31 m_deriv.template advection<Vars>(current_cell, vars.shift);
32
33 // work out RHS including advection
34 Vars<data_t> rhs;
35 VarsTools::assign(
36 rhs,
37 0.); // All components that are not explicitly set in rhs_equation are 0
38 rhs_equation(rhs, vars, d1, d2, advec);
39
40 // Write the rhs into the output FArrayBox
41 current_cell.store_vars(rhs);
42}

44template <class matter_t>
45template <class data_t>
46void ChiRelaxation<matter_t>::rhs_equation(
47 Vars<data_t> &rhs, const Vars<data_t> &vars,
48 const Vars<Tensor<1, data_t>> &d1, const Diff2Vars<Tensor<2, data_t>> &d2,
49 const Vars<data_t> &advec) const
50{
51
52 using namespace TensorAlgebra;
53
54 const auto h_UU = compute_inverse_sym(vars.h);
55 const auto chris = compute_christoffel(d1.h, h_UU);
56
57 // Calculate elements of the decomposed stress energy tensor and ricci
58 // tensor
59 const auto emtensor = my_matter.compute_emtensor(vars, d1, h_UU, chris.ULL);
60 const auto ricci = CCZ4Geometry::compute_ricci(vars, d1, d2, h_UU, chris);
61 const auto A_UU = raise_all(vars.A, h_UU);
62 const data_t tr_AA = compute_trace(vars.A, A_UU);
63
64 // Calculate the relaxation RHS for chi, all other vars RHS zero
65 // Could have called ConstraintsMatter here, but it is hardly worth it
66 // The division by chi prevents it (usually) from doing chi=0 as a solution
67 rhs.chi =
68 m_relax_speed *
69 (ricci.scalar + (GR_SPACEDIM - 1.) * vars.K * vars.K / GR_SPACEDIM -
70 tr_AA - 16.0 * M_PI * m_G_Newton * emtensor.rho) /
71 vars.chi;
72}
