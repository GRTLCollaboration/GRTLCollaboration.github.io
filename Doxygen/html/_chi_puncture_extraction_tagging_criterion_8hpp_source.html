65 template <class data_t> void compute(Cell<data_t> current_cell) const
66 {
67 // first test the gradients for regions of high curvature
68 const auto d2 = m_deriv.template diff2<Vars>(current_cell);
69 data_t mod_d2_chi = 0;
70 FOR(idir, jdir)
71 {
72 mod_d2_chi += d2.chi[idir][jdir] * d2.chi[idir][jdir];
73 }
74 data_t criterion = m_dx * sqrt(mod_d2_chi);
75
76 // if extracting weyl data at a given radius, enforce a given resolution
77 // there
78 if (m_activate_extraction)
79 {
80 for (int iradius = 0; iradius < m_params.num_extraction_radii;
81 ++iradius)
82 {
83 // regrid if within extraction level and not at required
84 // refinement
85 if (m_level < m_params.extraction_levels[iradius])
86 {
87 const Coordinates<data_t> coords(current_cell, m_dx,
88 m_params.center);
89 const data_t r = coords.get_radius();
90 // add a 20% buffer to extraction zone so not too near to
91 // boundary
92 auto regrid = simd_compare_lt(
93 r, 1.2 * m_params.extraction_radii[iradius]);
94 criterion = simd_conditional(regrid, 100.0, criterion);
95 }
96 }
97 }
98
99 // ensure that the horizons of the punctures are covered
100 // by the max level - for this we need
101 // only check the puncture locations on the top 2 levels
102 // which regrid (ie, max_level - 1 to max_level - 2)
103 // (just the top level would be ok, but doing two ensures
104 // the top levels are well spaced)
105 if ((m_level > (m_max_level - 3)) && (m_track_punctures == 1))
106 {
107 // we want each level to be double the innermost one in size
108 const double factor = pow(2.0, m_max_level - m_level - 1);
109 // loop over puncture masses
110 for (int ipuncture = 0; ipuncture < m_puncture_masses.size();
111 ++ipuncture)
112 {
113 // where am i?
114 const Coordinates<data_t> coords(current_cell, m_dx,
115 m_puncture_coords[ipuncture]);
116 const data_t r = coords.get_radius();
117 // decide whether to tag based on distance to horizon
118 // plus a fudge factor of 1.5
119 auto regrid = simd_compare_lt(
120 r, 1.5 * factor * m_puncture_masses[ipuncture]);
121 criterion = simd_conditional(regrid, 100.0, criterion);
122 }
123 }
124
125 // Write back into the flattened Chombo box
126 current_cell.store_vars(criterion, 0);
127 }
128};
