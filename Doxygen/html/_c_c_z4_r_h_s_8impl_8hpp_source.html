18inline CCZ4RHS<gauge_t, deriv_t>::CCZ4RHS(
19 CCZ4_params_t<typename gauge_t::params_t> a_params, double a_dx,
20 double a_sigma, int a_formulation, double a_cosmological_constant)
21 : m_params(a_params), m_gauge(a_params), m_sigma(a_sigma),
22 m_formulation(a_formulation),
23 m_cosmological_constant(a_cosmological_constant), m_deriv(a_dx)
24{
25 // A user who wants to use BSSN should also have damping paramters = 0
26 if (m_formulation == USE_BSSN)
27 {
28 if ((m_params.kappa1 != 0.) || (m_params.kappa2 != 0.) ||
29 (m_params.kappa3 != 0.))
30 {
31 MayDay::Error("BSSN formulation is selected - CCZ4 kappa values "
32 "should be set to zero in params");
33 }
34 }
35 if (m_formulation > USE_BSSN)
36 MayDay::Error("The requested formulation is not supported");
37}

40template <class data_t>
41void CCZ4RHS<gauge_t, deriv_t>::compute(Cell<data_t> current_cell) const
42{
43 const auto vars = current_cell.template load_vars<Vars>();
44 const auto d1 = m_deriv.template diff1<Vars>(current_cell);
45 const auto d2 = m_deriv.template diff2<Diff2Vars>(current_cell);
46 const auto advec =
47 m_deriv.template advection<Vars>(current_cell, vars.shift);
48
49 Vars<data_t> rhs;
50 rhs_equation(rhs, vars, d1, d2, advec);
51
52 m_deriv.add_dissipation(rhs, current_cell, m_sigma);
53
54 current_cell.store_vars(rhs); // Write the rhs into the output FArrayBox
55}

57template <class gauge_t, class deriv_t>
58template <class data_t, template <typename> class vars_t,
59 template <typename> class diff2_vars_t>
60void CCZ4RHS<gauge_t, deriv_t>::rhs_equation(
61 vars_t<data_t> &rhs, const vars_t<data_t> &vars,
62 const vars_t<Tensor<1, data_t>> &d1,
63 const diff2_vars_t<Tensor<2, data_t>> &d2,
64 const vars_t<data_t> &advec) const
65{
66 using namespace TensorAlgebra;
67
68 auto h_UU = compute_inverse_sym(vars.h);
69 auto chris = compute_christoffel(d1.h, h_UU);
70
71 Tensor<1, data_t> Z_over_chi;
72 Tensor<1, data_t> Z;
73
74 if (m_formulation == USE_BSSN)
75 {
76 FOR(i) Z_over_chi[i] = 0.0;
77 }
78 else
79 {
80 FOR(i) Z_over_chi[i] = 0.5 * (vars.Gamma[i] - chris.contracted[i]);
81 }
82 FOR(i) Z[i] = vars.chi * Z_over_chi[i];
83
84 auto ricci =
85 CCZ4Geometry::compute_ricci_Z(vars, d1, d2, h_UU, chris, Z_over_chi);
86
87 data_t divshift = compute_trace(d1.shift);
88 data_t Z_dot_d1lapse = compute_dot_product(Z, d1.lapse);
89 data_t dlapse_dot_dchi = compute_dot_product(d1.lapse, d1.chi, h_UU);
90
91 Tensor<2, data_t> covdtilde2lapse;
92 Tensor<2, data_t> covd2lapse;
93 FOR(k, l)
94 {
95 covdtilde2lapse[k][l] = d2.lapse[k][l];
96 FOR(m) { covdtilde2lapse[k][l] -= chris.ULL[m][k][l] * d1.lapse[m]; }
97 covd2lapse[k][l] =
98 vars.chi * covdtilde2lapse[k][l] +
99 0.5 * (d1.lapse[k] * d1.chi[l] + d1.chi[k] * d1.lapse[l] -
100 vars.h[k][l] * dlapse_dot_dchi);
101 }
102
103 data_t tr_covd2lapse = -(GR_SPACEDIM / 2.0) * dlapse_dot_dchi;
104 FOR(i)
105 {
106 tr_covd2lapse -= vars.chi * chris.contracted[i] * d1.lapse[i];
107 FOR(j)
108 {
109 tr_covd2lapse += h_UU[i][j] * (vars.chi * d2.lapse[i][j] +
110 d1.lapse[i] * d1.chi[j]);
111 }
112 }
113
114 Tensor<2, data_t> A_UU = raise_all(vars.A, h_UU);
115
116 // A^{ij} A_{ij}. - Note the abuse of the compute trace function.
117 data_t tr_A2 = compute_trace(vars.A, A_UU);
118 rhs.chi = advec.chi +
119 (2.0 / GR_SPACEDIM) * vars.chi * (vars.lapse * vars.K - divshift);
120 FOR(i, j)
121 {
122 rhs.h[i][j] = advec.h[i][j] - 2.0 * vars.lapse * vars.A[i][j] -
123 (2.0 / GR_SPACEDIM) * vars.h[i][j] * divshift;
124 FOR(k)
125 {
126 rhs.h[i][j] +=
127 vars.h[k][i] * d1.shift[k][j] + vars.h[k][j] * d1.shift[k][i];
128 }
129 }
130
131 Tensor<2, data_t> Adot_TF;
132 FOR(i, j)
133 {
134 Adot_TF[i][j] =
135 -covd2lapse[i][j] + vars.chi * vars.lapse * ricci.LL[i][j];
136 }
137 make_trace_free(Adot_TF, vars.h, h_UU);
138
139 FOR(i, j)
140 {
141 rhs.A[i][j] = advec.A[i][j] + Adot_TF[i][j] +
142 vars.A[i][j] * (vars.lapse * (vars.K - 2 * vars.Theta) -
143 (2.0 / GR_SPACEDIM) * divshift);
144 FOR(k)
145 {
146 rhs.A[i][j] +=
147 vars.A[k][i] * d1.shift[k][j] + vars.A[k][j] * d1.shift[k][i];
148 FOR(l)
149 {
150 rhs.A[i][j] -=
151 2 * vars.lapse * h_UU[k][l] * vars.A[i][k] * vars.A[l][j];
152 }
153 }
154 }
155
156 data_t kappa1_times_lapse;
157 if (m_params.covariantZ4)
158 kappa1_times_lapse = m_params.kappa1;
159 else
160 kappa1_times_lapse = m_params.kappa1 * vars.lapse;
161
162 if (m_formulation == USE_BSSN)
163 {
164 rhs.Theta = 0; // ensure the Theta of CCZ4 remains at zero
165 // Use hamiltonian constraint to remove ricci.scalar for BSSN update
166 rhs.K = advec.K + vars.lapse * (tr_A2 + vars.K * vars.K / GR_SPACEDIM) -
167 tr_covd2lapse;
168 rhs.K += -2 * vars.lapse * m_cosmological_constant / (GR_SPACEDIM - 1.);
169 }
170 else
171 {
172 rhs.Theta =
173 advec.Theta +
174 0.5 * vars.lapse *
175 (ricci.scalar - tr_A2 +
176 ((GR_SPACEDIM - 1.0) / (double)GR_SPACEDIM) * vars.K * vars.K -
177 2 * vars.Theta * vars.K) -
178 0.5 * vars.Theta * kappa1_times_lapse *
179 ((GR_SPACEDIM + 1) + m_params.kappa2 * (GR_SPACEDIM - 1)) -
180 Z_dot_d1lapse;
181
182 rhs.Theta += -vars.lapse * m_cosmological_constant;
183 rhs.K =
184 advec.K +
185 vars.lapse * (ricci.scalar + vars.K * (vars.K - 2 * vars.Theta)) -
186 kappa1_times_lapse * GR_SPACEDIM * (1 + m_params.kappa2) *
187 vars.Theta -
188 tr_covd2lapse;
189 rhs.K += -2 * vars.lapse * GR_SPACEDIM / (GR_SPACEDIM - 1.) *
190 m_cosmological_constant;
191 }
192
193 Tensor<1, data_t> Gammadot;
194 FOR(i)
195 {
196 Gammadot[i] = (2.0 / GR_SPACEDIM) *
197 (divshift * (chris.contracted[i] +
198 2 * m_params.kappa3 * Z_over_chi[i]) -
199 2 * vars.lapse * vars.K * Z_over_chi[i]) -
200 2 * kappa1_times_lapse * Z_over_chi[i];
201 FOR(j)
202 {
203 Gammadot[i] +=
204 2 * h_UU[i][j] *
205 (vars.lapse * d1.Theta[j] - vars.Theta * d1.lapse[j]) -
206 2 * A_UU[i][j] * d1.lapse[j] -
207 vars.lapse * ((2 * (GR_SPACEDIM - 1.0) / (double)GR_SPACEDIM) *
208 h_UU[i][j] * d1.K[j] +
209 GR_SPACEDIM * A_UU[i][j] * d1.chi[j] / vars.chi) -
210 (chris.contracted[j] + 2 * m_params.kappa3 * Z_over_chi[j]) *
211 d1.shift[i][j];
212
213 FOR(k)
214 {
215 Gammadot[i] +=
216 2 * vars.lapse * chris.ULL[i][j][k] * A_UU[j][k] +
217 h_UU[j][k] * d2.shift[i][j][k] +
218 ((GR_SPACEDIM - 2.0) / (double)GR_SPACEDIM) * h_UU[i][j] *
219 d2.shift[k][j][k];
220 }
221 }
222 }
223
224 FOR(i) { rhs.Gamma[i] = advec.Gamma[i] + Gammadot[i]; }
225
226 m_gauge.rhs_gauge(rhs, vars, d1, d2, advec);
227}
