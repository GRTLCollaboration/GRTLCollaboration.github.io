140
141 void set_params(D_DECL(double a_radius_x, double a_radius_y,
142 double a_radius_z))
143 {
144 D_TERM(m_radius_x = a_radius_x;, m_radius_y = a_radius_y;
145 , m_radius_z = a_radius_z;)
146 pout() << "Setting Original Guess to ellipsoid=(" D_TERM(
147 << m_radius_x, << ", " << m_radius_y,
148 << ", " << m_radius_z)
149 << ")" << std::endl;
150 }
151
152#if CH_SPACEDIM == 3
153 ALWAYS_INLINE double get_merger_min_distance() const override
154 {
155 return std::max(m_radius_x, std::max(m_radius_y, m_radius_z));
156 };
157
158 ALWAYS_INLINE double get(double u, double v) const override
159 {
160 double sin_u = sin(u);
161 double cos_u = cos(u);
162 double sin_v = sin(v);
163 double cos_v = cos(v);
164 double radius =
165 1. /
166 sqrt(sin_u * sin_u * cos_v * cos_v / (m_radius_x * m_radius_x) +
167 sin_u * sin_u * sin_v * sin_v / (m_radius_y * m_radius_y) +
168 cos_u * cos_u / (m_radius_z * m_radius_z));
169 // pout() << u << "|" << v << "|" << radius << std::endl;
170 return radius;
171 }
172 ALWAYS_INLINE double get_merger_contribution(double u,
173 double v) const override
174 {
175 return get_merger_min_distance(); // use constant
176 // return get(u, v); // use ellipsoid
177 }
178
179#elif CH_SPACEDIM == 2
180 ALWAYS_INLINE double get_merger_min_distance() const override
181 {
182 return std::max(m_radius_x, m_radius_y);
183 };
184
185 ALWAYS_INLINE double get(double u) const override
186 {
187 double sin_u = sin(u);
188 double cos_u = cos(u);
189 // u = 0 corresponds to (x,y)=(-1,0)
190 // u = pi/2 corresponds to (x,y)=(0,1)
191 // u = pi corresponds to (x,y)=(1,0)
192 double radius = 1. / sqrt(cos_u * cos_u / (m_radius_x * m_radius_x) +
193 sin_u * sin_u / (m_radius_y * m_radius_y));
194 // pout() << u << "|" << radius << std::endl;
195 return radius;
196 }
197 ALWAYS_INLINE double get_merger_contribution(double u) const override
198 {
199 return get_merger_min_distance(); // use constant
200 // return get(u); // use ellipsoid
201 }
202#endif
203};
