38 void compute(Cell<double> current_cell) const
39 {
40 bool stop = false;
41 int num_vars = current_cell.get_num_in_vars();
42 for (int ivar = 0; ivar < num_vars; ++ivar)
43 {
44 double val;
45 current_cell.load_vars(val, ivar);
46 if (std::isnan(val) || abs(val) > m_max_abs)
47 stop = true;
48 }
49 if (stop)
50 {
51#pragma omp single
52 {
53 pout() << m_error_info
54 << "::Values have become nan. The current state is: \n";
55 for (int ivar = 0; ivar < num_vars; ++ivar)
56 {
57 pout() << UserVariables::variable_names[ivar] << ": "
58 << current_cell.load_vars(ivar) << std::endl;
59 }
60 pout() << "Integer coordinates: " << current_cell.get_int_vect()
61 << std::endl;
62 if (m_dx > 0)
63 {
64 Coordinates<double> coords{current_cell, m_dx, m_center};
65 pout() << coords << std::endl;
66 }
67 }
68 MayDay::Error("Values have become nan.");
69 }
70 }
