290  template <class data_t, class mask_t>
291  ALWAYS_INLINE data_t advection_term(const double *in_ptr, const int idx,
292  const data_t &vec_comp,
293  const int stride,
294  const mask_t shift_positive) const
295  {
296  const auto in = SIMDIFY<data_t>(in_ptr);
297  const data_t in_left_far = in[idx - 2 * stride];
298  const data_t in_left = in[idx - stride];
299  const data_t in_centre = in[idx];
300  const data_t in_right = in[idx + stride];
301  const data_t in_right_far = in[idx + 2 * stride];
302 
303  data_t weight_0 = +3.33333333333333333333e-2;
304  data_t weight_1 = -4.00000000000000000000e-1;
305  data_t weight_2 = -5.83333333333333333333e-1;
306  data_t weight_3 = +1.33333333333333333333e0;
307  data_t weight_4 = -5.00000000000000000000e-1;
308  data_t weight_5 = +1.33333333333333333333e-1;
309  data_t weight_6 = -1.66666666666666666667e-2;
310 
311  data_t upwind;
312  upwind = vec_comp *
313  (weight_0 * in_left_far + weight_1 * in_left +
314  weight_2 * in_centre + weight_3 * in_right +
315  weight_4 * in_right_far + weight_5 * in[idx + 3 * stride] +
316  weight_6 * in[idx + 4 * stride]) *
317  m_one_over_dx;
318 
319  data_t downwind;
320  downwind = vec_comp *
321  (-weight_6 * in[idx - 4 * stride] -
322  weight_5 * in[idx - 3 * stride] - weight_4 * in_left_far -
323  weight_3 * in_left - weight_2 * in_centre -
324  weight_1 * in_right - weight_0 * in_right_far) *
325  m_one_over_dx;
326 
327  return simd_conditional(shift_positive, upwind, downwind);
328  }
