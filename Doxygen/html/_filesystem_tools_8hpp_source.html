34 static bool mkdir_recursive(const std::string &path)
35 {
36 #ifdef CH_MPI
37  // all processes should get here at the same time
38  // e.g. if doing:
39  // if (!directory_exists(path))
40  // mkdir_recursive(path);
41  // We want to make sure the directory is not created until all
42  // ranks have passed the 'directory_exists'
43  MPI_Barrier(Chombo_MPI::comm);
44 #endif
45 
46  bool success = true;
47  if (procID() == 0)
48  {
49  // For Windows backslash, use delimeters = "/\\"
50  static const std::string delimeter = "/";
51  std::size_t found = path.find_first_of(delimeter);
52 
53  // NB: this would be very beautiful recursively, but let's not do it
54  // because the function involves MPI_Barrier's
55  while (success && found != std::string::npos)
56  {
57  std::string subpath;
58  subpath = path.substr(0, found);
59 
60  if (subpath != "." && subpath != "")
61  {
62  // success if created or if not created because it already
63  // exists
64  success &=
65  (mkdir(subpath.c_str(), 0700) == 0 || errno == EEXIST);
66  }
67  found = path.find_first_of(delimeter, found + 1);
68  }
69  // if path doesn't finish with "/", one more to do
70  if (found != path.size() - 1)
71  success &= (mkdir(path.c_str(), 0700) == 0 || errno == EEXIST);
72  }
73 
74 #ifdef CH_MPI
75  // all processes should wait to make sure directory structure is well set
76  // for everyone
77  MPI_Barrier(Chombo_MPI::comm);
78 #endif
79 
80  return success;
81 }
