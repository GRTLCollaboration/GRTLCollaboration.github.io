36 void read_params(GRParmParse &pp)
37 {
38 // must be before any pout() in the code to setPoutBaseName
39 read_filesystem_params(pp);
40
41 pp.load("verbosity", verbosity, 0);
42 // Grid setup
43 pp.load("max_spatial_derivative_order", max_spatial_derivative_order,
44 4);
45 pp.load("num_ghosts", num_ghosts,
46 (max_spatial_derivative_order == 6) ? 4 : 3);
47 pp.load("tag_buffer_size", tag_buffer_size, 3);
48 pp.load("grid_buffer_size", grid_buffer_size, 8);
49 pp.load("dt_multiplier", dt_multiplier, 0.25);
50 pp.load("fill_ratio", fill_ratio, 0.7);
51
52 // Periodicity and boundaries
53 boundary_params.read_params(pp);
54
55 // L's, N's and center
56 read_grid_params(pp);
57
58#ifdef CH_USE_HDF5
59 pp.load("ignore_checkpoint_name_mismatch",
60 ignore_checkpoint_name_mismatch, false);
61#endif
62
63 pp.load("max_level", max_level, 0);
64 // the reference ratio is hard coded to 2
65 // in principle it can be set to other values, but this is
66 // not recommended since we do not test GRChombo with other
67 // refinement ratios - use other values at your own risk
68 ref_ratios.resize(max_level + 1);
69 ref_ratios.assign(2);
70 pp.getarr("regrid_interval", regrid_interval, 0, max_level);
71 // Regridding on max_level does nothing but Chombo's AMR class
72 // expects this Vector to be of length max_level + 1
73 // so just set the final value to 0.
74 regrid_interval.resize(max_level + 1);
75 regrid_interval[max_level] = 0;
76
77 if (pp.contains("regrid_thresholds"))
78 {
79 pout() << "Using multiple regrid thresholds." << std::endl;
80 // As for regrid_interval, the last element is irrelevant
81 pp.getarr("regrid_thresholds", regrid_thresholds, 0, max_level);
82 regrid_thresholds.resize(max_level + 1);
83 regrid_thresholds[max_level] = regrid_thresholds[max_level - 1];
84 }
85 else
86 {
87 pout() << "Using single regrid threshold." << std::endl;
88 double regrid_threshold;
89 pp.load("regrid_threshold", regrid_threshold, 0.5);
90 regrid_thresholds = Vector<double>(max_level + 1, regrid_threshold);
91 }
92
93 // time stepping outputs and regrid data
94 pp.load("checkpoint_interval", checkpoint_interval, 1);
95 pp.load("plot_interval", plot_interval, 0);
96 pp.load("stop_time", stop_time, 1.0);
97 pp.load("max_steps", max_steps, 1000000);
98#ifdef CH_USE_HDF5
99 pp.load("write_plot_ghosts", write_plot_ghosts, false);
100
101 // load vars to write to plot files
102 UserVariables::load_vars_to_vector(pp, "plot_vars", "num_plot_vars",
103 plot_vars, num_plot_vars);
104#endif
105 // alias the weird chombo names to something more descriptive
106 // for these box params, and default to some reasonable values
107 if (pp.contains("max_grid_size"))
108 {
109 pp.load("max_grid_size", max_grid_size);
110 }
111 else
112 {
113 pp.load("max_box_size", max_grid_size, 64);
114 }
115 if (pp.contains("block_factor"))
116 {
117 pp.load("block_factor", block_factor);
118 }
119 else
120 {
121 pp.load("min_box_size", block_factor, 8);
122 }
123
124 if (pp.contains("check_params"))
125 just_check_params = true;
126
127 pp.load("print_progress_only_to_rank_0", print_progress_only_to_rank_0,
128 false);
129 }

131 void read_filesystem_params(GRParmParse &pp)
132 {
133 // In this function, cannot use default value - it may print a 'default
134 // message' to pout and a 'setPoutBaseName' must happen before
135 restart_from_checkpoint = pp.contains("restart_file");
136#ifdef CH_USE_HDF5
137 if (restart_from_checkpoint)
138 {
139 pp.load("restart_file", restart_file);
140 }
141 pp.load("chk_prefix", checkpoint_prefix);
142 pp.load("plot_prefix", plot_prefix);
143#endif
144
145#ifdef CH_MPI
146 // Again, cannot use default value
147 if (pp.contains("pout_prefix"))
148 pp.load("pout_prefix", pout_prefix);
149 else
150 pout_prefix = "pout";
151#endif
152
153 std::string default_path = "";
154 if (pp.contains("output_path"))
155 pp.load("output_path", output_path);
156 else
157 output_path = default_path;
158
159#ifdef CH_MPI
160 // user sets the 'subpath', we prepend 'output_path'
161 if (pp.contains("pout_subpath"))
162 pp.load("pout_subpath", pout_path);
163 else
164 pout_path = default_path;
165#endif
166
167#ifdef CH_USE_HDF5
168 // user sets the 'subpath', we prepend 'output_path'
169 if (pp.contains("hdf5_subpath"))
170 pp.load("hdf5_subpath", hdf5_path);
171 else
172 hdf5_path = default_path;
173#endif
174
175 // add backslash to paths
176 if (!output_path.empty() && output_path.back() != '/')
177 output_path += "/";
178#ifdef CH_MPI
179 if (!pout_path.empty() && pout_path.back() != '/')
180 pout_path += "/";
181#endif
182#ifdef CH_USE_HDF5
183 if (!hdf5_path.empty() && hdf5_path.back() != '/')
184 hdf5_path += "/";
185#endif
186
187 if (output_path != "./" && !output_path.empty())
188 {
189#ifdef CH_MPI
190 pout_path = output_path + pout_path;
191#endif
192#ifdef CH_USE_HDF5
193 hdf5_path = output_path + hdf5_path;
194 // assume restart_file is an absolute path if it starts with '/'
195 // otherwise assume it is relative to hdf5_path
196 if (restart_from_checkpoint && restart_file.front() != '/')
197 restart_file = hdf5_path + restart_file;
198#endif
199 }
200
201#ifdef CH_MPI
202 // change pout base name!
203 if (!FilesystemTools::directory_exists(pout_path))
204 FilesystemTools::mkdir_recursive(pout_path);
205 setPoutBaseName(pout_path + pout_prefix);
206#endif
207
208 // only create hdf5 directory in setupAMRObject (when it becomes needed)
209 }

211 void read_grid_params(GRParmParse &pp)
212 {
213 // Grid N
214 std::array<int, CH_SPACEDIM> Ni_full;
215 std::array<int, CH_SPACEDIM> Ni;
216 ivN = IntVect::Unit;
217
218 // cannot contain both
219 if ((pp.contains("N_full") && pp.contains("N")))
220 MayDay::Error("Please only provide 'N' or 'N_full', not both");
221
222 int N_full = -1;
223 int N = -1;
224 if (pp.contains("N_full"))
225 pp.load("N_full", N_full);
226 else if (pp.contains("N"))
227 pp.load("N", N);
228
229 // read all options (N, N_full, Ni_full and Ni) and then choose
230 // accordingly
231 FOR(dir)
232 {
233 std::string name = ("N" + std::to_string(dir + 1));
234 std::string name_full = ("N" + std::to_string(dir + 1) + "_full");
235 Ni_full[dir] = -1;
236 Ni[dir] = -1;
237
238 // only one of them exists - this passes if none of the 4 exist, but
239 // that is asserted below
240 if (!((N_full > 0 || N > 0) && !pp.contains(name.c_str()) &&
241 !pp.contains(name_full.c_str())) &&
242 !((N_full < 0 && N < 0) && !(pp.contains(name.c_str()) &&
243 pp.contains(name_full.c_str()))))
244 error("Please provide 'N' or 'N_full' or a set of "
245 "'N1/N1_full', 'N2/N2_full', 'N3/N3_full'");
246
247 if (N_full < 0 && N < 0)
248 {
249 if (pp.contains(name_full.c_str()))
250 pp.load(name_full.c_str(), Ni_full[dir]);
251 else
252 pp.load(name.c_str(), Ni[dir]);
253 }
254 if (N < 0 && N_full < 0 && Ni[dir] < 0 &&
255 Ni_full[dir] < 0) // sanity check
256 error("Please provide 'N' or 'N_full' or a set of "
257 "'N1/N1_full', 'N2/N2_full', 'N3/N3_full'");
258
259 if (N_full > 0)
260 Ni_full[dir] = N_full;
261 else if (N > 0)
262 Ni[dir] = N;
263
264 if (Ni[dir] > 0)
265 {
266 if (boundary_params.lo_boundary[dir] ==
267 BoundaryConditions::REFLECTIVE_BC ||
268 boundary_params.hi_boundary[dir] ==
269 BoundaryConditions::REFLECTIVE_BC)
270
271 Ni_full[dir] = Ni[dir] * 2;
272 else
273 Ni_full[dir] = Ni[dir];
274 }
275 else
276 {
277 if (boundary_params.lo_boundary[dir] ==
278 BoundaryConditions::REFLECTIVE_BC ||
279 boundary_params.hi_boundary[dir] ==
280 BoundaryConditions::REFLECTIVE_BC)
281 {
282 check_parameter("N" + std::to_string(dir) + "_full",
283 Ni_full[dir], Ni_full[dir] % 2 == 0,
284 "must be a multiple of 2");
285
286 Ni[dir] = Ni_full[dir] / 2;
287 }
288 else
289 Ni[dir] = Ni_full[dir];
290 }
291 ivN[dir] = Ni[dir] - 1;
292 }
293 int max_N_full = *std::max_element(Ni_full.begin(), Ni_full.end());
294 int max_N = ivN.max() + 1;
295
296 // Grid L
297 // cannot contain both
298 if ((pp.contains("L_full") && pp.contains("L")))
299 error("Please only provide 'L' or 'L_full', not both");
300
301 double L_full = -1.;
302 if (pp.contains("L_full"))
303 pp.load("L_full", L_full);
304 else
305 pp.load("L", L, 1.0);
306
307 if (L_full > 0.)
308 // necessary for some reflective BC cases, as 'L' is the
309 // length of the longest side of the box
310 L = (L_full * max_N) / max_N_full;
311
312 coarsest_dx = L / max_N;
313 coarsest_dt = coarsest_dx * dt_multiplier;
314
315 // grid spacing params
316 dx.fill(coarsest_dx);
317 origin.fill(coarsest_dx / 2.0);
318
319 // These aren't parameters but used in parameter checks
320 FOR(idir)
321 {
322 reflective_domain_lo[idir] = ((boundary_params.lo_boundary[idir] ==
323 BoundaryConditions::REFLECTIVE_BC)
324 ? -1.0
325 : 0.0) *
326 (ivN[idir] + 1) * coarsest_dx;
327 reflective_domain_hi[idir] = ((boundary_params.hi_boundary[idir] ==
328 BoundaryConditions::REFLECTIVE_BC)
329 ? 2.0
330 : 1.0) *
331 (ivN[idir] + 1) * coarsest_dx;
332 }
333
334 // First work out the default center ignoring reflective BCs
335 // but taking into account different grid lengths in each direction
336 std::array<double, CH_SPACEDIM> default_center;
337#if CH_SPACEDIM == 3
338 default_center = {0.5 * Ni[0] * coarsest_dx, 0.5 * Ni[1] * coarsest_dx,
339 0.5 * Ni[2] * coarsest_dx};
340#elif CH_SPACEDIM == 2
341 default_center = {0.5 * Ni[0] * coarsest_dx, 0.5 * Ni[1] * coarsest_dx};
342#endif
343 // Now take into account reflective BCs
344 FOR(idir)
345 {
346 if ((boundary_params.lo_boundary[idir] ==
347 BoundaryConditions::REFLECTIVE_BC) &&
348 (boundary_params.hi_boundary[idir] !=
349 BoundaryConditions::REFLECTIVE_BC))
350 default_center[idir] = 0.;
351 else if ((boundary_params.hi_boundary[idir] ==
352 BoundaryConditions::REFLECTIVE_BC) &&
353 (boundary_params.lo_boundary[idir] !=
354 BoundaryConditions::REFLECTIVE_BC))
355 default_center[idir] = coarsest_dx * Ni[idir];
356 }
357
358 pp.load("center", center, default_center); // default to center
359 }

361 void check_params()
362 {
363 check_parameter("L", L, L > 0.0, "must be > 0.0");
364 check_parameter("max_level", max_level, max_level >= 0, "must be >= 0");
365 check_parameter("max_spatial_derivative_order",
366 max_spatial_derivative_order,
367 max_spatial_derivative_order == 4 ||
368 max_spatial_derivative_order == 6,
369 "only 4 and 6 are supported");
370 // the following check assumes you will be taking one-sided derivatives
371 // of the order given by max_spatial_derivative_order
372 check_parameter(
373 "num_ghosts", num_ghosts,
374 (num_ghosts >= ((max_spatial_derivative_order == 6) ? 4 : 3)) &&
375 (num_ghosts <= block_factor),
376 "must be >= 3 (4th order derivatives) or 4 (6th order derivatives) "
377 "and <= min_box_size (aka block_factor)");
378 check_parameter("tag_buffer_size", tag_buffer_size,
379 tag_buffer_size >= 0, "must be >= 0");
380 // assume ref_ratio is always 2
381 check_parameter(
382 "grid_buffer_size", grid_buffer_size,
383 grid_buffer_size >= ceil(num_ghosts / 2.0),
384 "must be >= ceil(num_ghosts/max_ref_ratio) for proper nesting");
385
386 // check the restart_file exists and can be read if restarting from a
387 // checkpoint
388#ifdef CH_USE_HDF5
389 if (restart_from_checkpoint)
390 {
391 bool restart_file_exists =
392 (access((restart_file).c_str(), R_OK) == 0);
393 check_parameter("restart_file", restart_file, restart_file_exists,
394 "file cannot be opened for reading");
395 }
396#endif
397
398 check_parameter("dt_multiplier", dt_multiplier, dt_multiplier > 0.0,
399 "must be > 0.0");
400 check_parameter("max_grid_size/max_box_size", max_grid_size,
401 max_grid_size >= 0, "must be >= 0");
402 check_parameter("block_factor/min_box_size", block_factor,
403 block_factor >= 1, "must be >= 1");
404 check_parameter("block_factor/min_box_size", block_factor,
405 Misc::isPower2(block_factor), "must be a power of 2");
406 // note that this also enforces block_factor <= max_grid_size
407 // if max_grid_size > 0
408 check_parameter("block_factor/min_box_size", block_factor,
409 max_grid_size % block_factor == 0,
410 "must divide max_grid_size/max_box_size = " +
411 std::to_string(max_grid_size));
412 FOR(idir)
413 {
414 std::string Ni_string = "N" + std::to_string(idir + 1);
415 std::string invalid_message = "must divide " + Ni_string;
416 if (boundary_params.reflective_boundaries_exist)
417 {
418 invalid_message += " (or " + Ni_string + "_full/2)";
419 }
420 invalid_message += " = " + std::to_string(ivN[idir] + 1);
421 check_parameter("block_factor/min_box_size", block_factor,
422 (ivN[idir] + 1) % block_factor == 0,
423 invalid_message);
424 }
425 check_parameter("fill_ratio", fill_ratio,
426 (fill_ratio > 0.0) && (fill_ratio <= 1.0),
427 "must be > 0 and <= 1");
428 // (MR); while this would technically work (any plot files would just
429 // overwrite a checkpoint file), I think a user would only ever do
430 // this unintentinally
431#ifdef CH_USE_HDF5
432 check_parameter("plot_prefix", plot_prefix,
433 plot_interval <= 0 || plot_prefix != checkpoint_prefix,
434 "should be different to checkpoint_prefix");
435#endif
436
437 check_parameter("output_path", output_path,
438 FilesystemTools::directory_exists(output_path),
439 "should be a valid directory");
440 // pout directory exists - we create it in read_filesystem_params()
441 // can't check hdf5 directory yet - only created after
442
443 if (boundary_params.reflective_boundaries_exist)
444 {
445 for (int ivar = 0; ivar < NUM_VARS; ++ivar)
446 {
447 std::string name = "vars_parity[c_" +
448 UserVariables::variable_names[ivar] + "]";
449 int var_parity = boundary_params.vars_parity[ivar];
450 check_parameter(name, var_parity,
451 var_parity >= BoundaryConditions::EVEN &&
452 var_parity < BoundaryConditions::UNDEFINED,
453 "parity type undefined");
454 }
455 for (int ivar = 0; ivar < NUM_DIAGNOSTIC_VARS; ++ivar)
456 {
457 std::string name = "vars_parity_diagnostic[c_" +
458 DiagnosticVariables::variable_names[ivar] +
459 "]";
460 int var_parity = boundary_params.vars_parity_diagnostic[ivar];
461 check_parameter(name, var_parity,
462 var_parity >= BoundaryConditions::EVEN &&
463 var_parity <= BoundaryConditions::UNDEFINED,
464 "parity type undefined");
465 }
466 }
467 }

542 template <typename T>
543 void check_parameter(const std::string &a_name, T a_value,
544 const bool a_valid,
545 const std::string &a_invalid_explanation)
546 {
547 if (a_valid)
548 return;
549 else
550 {
551 std::ostringstream error_message_ss;
552 error_message_ss << "Parameter: " << a_name << " = " << a_value
553 << " is invalid: " << a_invalid_explanation;
554 error(error_message_ss.str());
555 }
556 }

558 template <typename T>
559 void warn_parameter(const std::string &a_name, T a_value,
560 const bool a_nowarn,
561 const std::string &a_warning_explanation)
562 {
563 if (a_nowarn)
564 return;
565 else
566 {
567 // only print the warning from rank 0
568 if (procID() == 0)
569 {
570 std::ostringstream warning_message_ss;
571 warning_message_ss << "Parameter: " << a_name << " = "
572 << a_value
573 << " warning: " << a_warning_explanation;
574 MayDay::Warning(warning_message_ss.str().c_str());
575 }
576 }
