18// PETSc Solver related params
19// not very relevant to change, but this allows us to in case we want
20// possible the only relevant one is 'snes_maxit' ('AH_SNES_max_iterations')
21// which for highly ellipsoidal AHs sometimes needs to be increased as
22// convergence is rather slow
23// (as of 06/2021, the code outputs this as a recommendation when that happens)
24struct PETSc_params
25{
26 /*
27 (defaults show according to a certain PETSc version,
28 can't assure it is the same for all)
29 atol - absolute convergence tolerance (default 1e-50)
30 rtol - relative convergence tolerance (default 1e-08)
31 stol - convergence tolerance in terms of the norm of the change in the
32 solution between steps (default 1e-08)
33 maxit - maximum number of iterations (default 50)
34 maxf - maximum number of function evaluations (default 10000)
35 divtol - divergence tolerance (default 10000)
36 */
37 PetscReal snes_atol = -1., snes_rtol = -1., snes_stol = -1.;
38 PetscInt snes_maxit = -1, snes_maxf = -1;
39 PetscReal snes_divtol = -1.;
40
41 /*
42 rtol - the relative convergence tolerance (default 1e-05)
43 abstol - the absolute convergence tolerance (default 1e-50)
44 dtol - the divergence tolerance (default 10000)
45 maxits - maximum number of iterations (default 10000)
46 */
47 PetscReal ksp_rtol = -1., ksp_abstol = -1., ksp_dtol = -1.;
48 PetscInt ksp_maxits = -1;
49
50 void read_params(GRParmParse &pp)
51 {
52 // variables to prevent ParmParse to complain about types (only happens
53 // in some clusters)
54 int tmp_int;
55 float tmp_float;
56 if (pp.contains("AH_SNES_absolute_tolerance"))
57 {
58 pp.load("AH_SNES_absolute_tolerance", tmp_float);
59 snes_atol = tmp_float;
60 }
61 if (pp.contains("AH_SNES_relative_tolerance"))
62 {
63 pp.load("AH_SNES_relative_tolerance", tmp_float);
64 snes_rtol = tmp_float;
65 }
66 if (pp.contains("AH_SNES_step_change_tolerance"))
67 {
68 pp.load("AH_SNES_step_change_tolerance", tmp_float);
69 snes_stol = tmp_float;
70 }
71 if (pp.contains("AH_SNES_max_iterations"))
72 {
73 pp.load("AH_SNES_max_iterations", tmp_int);
74 snes_maxit = tmp_int;
75 }
76 if (pp.contains("AH_SNES_max_evaluations"))
77 {
78 pp.load("AH_SNES_max_evaluations", tmp_int);
79 snes_maxf = tmp_int;
80 }
81 if (pp.contains("AH_SNES_divergence_tolerance"))
82 {
83 pp.load("AH_SNES_divergence_tolerance", tmp_float);
84 snes_divtol = tmp_float;
85 }
86
87 if (pp.contains("AH_KSP_relative_tolerance"))
88 {
89 pp.load("AH_KSP_relative_tolerance", tmp_float);
90 ksp_rtol = tmp_float;
91 }
92 if (pp.contains("AH_KSP_absolute_tolerance"))
93 {
94 pp.load("AH_KSP_absolute_tolerance", tmp_float);
95 ksp_abstol = tmp_float;
96 }
97 if (pp.contains("AH_KSP_divergence_tolerance"))
98 {
99 pp.load("AH_KSP_divergence_tolerance", tmp_float);
100 ksp_dtol = tmp_float;
101 }
102 if (pp.contains("AH_KSP_max_evaluations"))
103 {
104 pp.load("AH_KSP_max_evaluations", tmp_int);
105 ksp_maxits = tmp_int;
106 }
107 }
108};

202template <class AHFunction>
203void AHParams_t<AHFunction>::read_params(GRParmParse &pp,
204 const ChomboParameters &a_p)
205{
206 pp.load("AH_num_ranks", num_ranks, 0); // 0 means "all"
207
208 pp.load("AH_num_points_u", num_points_u);
209#if CH_SPACEDIM == 3
210 pp.load("AH_num_points_v", num_points_v);
211#endif
212 int actual_num_ranks = (num_ranks != 0) ? num_ranks : numProc();
213
214 // Let's enforce the below as Katy had problems otherwise
215 ChomboParameters::check_parameter("AH_num_ranks", num_ranks,
216 num_ranks <= numProc(),
217 "must be <= total number of ranks");
218
219 ChomboParameters::check_parameter("AH_num_points_u", num_points_u,
220 num_points_u > 0, "must be >0");
221#if CH_SPACEDIM == 3
222 ChomboParameters::check_parameter("AH_num_points_v", num_points_v,
223 num_points_v > 0, "must be >0");
224 // if box division ends up with side length less than 3, PETSc will
225 // complain - estimate side length using sqrt(num_ranks)
226 ChomboParameters::check_parameter(
227 "AH_num_points_u", num_points_u,
228 num_points_u / sqrt(actual_num_ranks) >= 3,
229 "(num_points_u/sqrt(num_ranks)) must be >3"); // make sure for size 'u'
230 ChomboParameters::check_parameter(
231 "AH_num_points_v", num_points_v,
232 num_points_v / sqrt(actual_num_ranks) >= 3,
233 "(num_points_v/sqrt(num_ranks)) must be >3"); // make sure for size 'v'
234
235 // Not sure if PETSc requires the below but let's just enforce it just
236 // in case
237 ChomboParameters::check_parameter(
238 "AH_num_ranks", num_ranks,
239 (num_points_u * num_points_v) % num_ranks == 0,
240 "must divide AH_num_points_u * AH_num_points_v");
241#elif CH_SPACEDIM == 2
242 ChomboParameters::check_parameter(
243 "AH_num_points_u", num_points_u, num_points_u / actual_num_ranks >= 3,
244 "(num_points_u/size) must be >0"); // make sure for size 'u'
245#endif
246
247 pp.load("AH_solve_interval", solve_interval, 1);
248 pp.load("AH_print_interval", print_interval, 1);
249 // sanity checks
250 ChomboParameters::check_parameter("AH_solve_interval", solve_interval,
251 solve_interval > 0, "must be >0");
252 ChomboParameters::check_parameter("AH_print_interval", print_interval,
253 print_interval > 0, "must be >0");
254 pp.load("AH_track_center", track_center, true);
255 pp.load("AH_predict_origin", predict_origin, track_center);
256 // can't predict if center is not being tracked
257 if (predict_origin)
258 {
259 ChomboParameters::check_parameter(
260 "AH_track_center", track_center, track_center == 1,
261 "can't predict if center is not being tracked");
262 }
263 pp.load("AH_level_to_run", level_to_run, 0);
264 ChomboParameters::check_parameter("AH_level_to_run", level_to_run,
265 level_to_run <= a_p.max_level,
266 "must be <= max_level");
267 ChomboParameters::check_parameter("AH_level_to_run", level_to_run,
268 level_to_run >= 0, "must be >= 0");
269 ChomboParameters::check_parameter("AH_level_to_run", level_to_run,
270 level_to_run > -(a_p.max_level + 1),
271 "must be > -(max_level+1)");
272 if (level_to_run < 0) // if negative, count backwards
273 level_to_run += a_p.max_level + 1;
274
275 pp.load("AH_start_time", start_time, 0.0);
276 pp.load("AH_give_up_time", give_up_time, -1.0);
277
278 pp.load("AH_allow_re_attempt", allow_re_attempt, false);
279 pp.load("AH_max_fails_after_lost", max_fails_after_lost, 0);
280 pp.load("AH_stop_if_max_fails", stop_if_max_fails, false);
281
282 pp.load("AH_verbose", verbose, (int)MIN);
283 pp.load("AH_print_geometry_data", print_geometry_data, false);
284 pp.load("AH_re_solve_at_restart", re_solve_at_restart, false);
285 // load vars to write to coord files
286 num_extra_vars = 0;
287 extra_contain_diagnostic = 0;
288
289 int AH_num_extra_vars;
290 pp.load("AH_num_extra_vars", AH_num_extra_vars, 0);
291 if (AH_num_extra_vars > 0)
292 {
293 std::vector<std::string> AH_extra_var_names(AH_num_extra_vars, "");
294 pp.load("AH_extra_vars", AH_extra_var_names, AH_num_extra_vars);
295 for (const std::string &full_name : AH_extra_var_names)
296 {
297 std::string var_name = full_name;
298
299 // variable names might start with "d1_" or "d2_" to indicate
300 // the user wants derivatives
301 int der_type = 0;
302 std::string der = var_name.substr(0, 3);
303 if (der == "d1_")
304 {
305 der_type = 1;
306 var_name = var_name.substr(3);
307 }
308 else if (der == "d2_")
309 {
310 der_type = 2;
311 var_name = var_name.substr(3);
312 }
313
314 // first assume extra_var is a normal evolution var
315 int var = UserVariables::variable_name_to_enum(var_name);
316 VariableType var_type = VariableType::evolution;
317 if (var < 0)
318 {
319 // if not an evolution var check if it's a diagnostic var
320 var = DiagnosticVariables::variable_name_to_enum(var_name);
321 if (var < 0)
322 {
323 // it's neither :(
324 pout() << "Variable with name " << var_name
325 << " not found.\n";
326 }
327 else
328 {
329 var_type = VariableType::diagnostic;
330 ++extra_contain_diagnostic;
331 }
332 }
333 if (var >= 0)
334 {
335 extra_vars[full_name] =
336 std::tuple<int, VariableType, int>(var, var_type, der_type);
337 if (der_type == 0)
338 num_extra_vars += 1;
339 else if (der_type == 1)
340 num_extra_vars += CH_SPACEDIM;
341 else // if (der_type == 2)
342 num_extra_vars += CH_SPACEDIM * (CH_SPACEDIM + 1) / 2;
343 }
344 }
345 }
346
347 stats_path = a_p.data_path;
348
349 if (pp.contains("AH_coords_subpath"))
350 {
351 pp.load("AH_coords_subpath", coords_path);
352 if (!coords_path.empty() && coords_path.back() != '/')
353 coords_path += "/";
354 if (a_p.output_path != "./" && !a_p.output_path.empty())
355 coords_path = a_p.output_path + coords_path;
356 }
357 else
358 coords_path = stats_path;
359
360 pp.load("AH_stats_prefix", stats_prefix, std::string("stats_AH"));
361 pp.load("AH_coords_prefix", coords_prefix, std::string("coords_AH"));
362
363 pp.load("AH_merger_search_factor", merger_search_factor, 1.);
364 pp.load("AH_merger_pre_factor", merger_pre_factor, 1.);
365
366 func_params.read_params(pp);
367 petsc_params.read_params(pp);
368}
