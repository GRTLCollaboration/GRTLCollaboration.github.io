18 template <class data_t> void BinaryBH::compute(Cell<data_t> current_cell) const
19 {
20  BSSNVars::VarsWithGauge<data_t> vars;
21  VarsTools::assign(vars,
22  0.); // Set only the non-zero components explicitly below
23  Coordinates<data_t> coords(current_cell, m_dx);
24 
25  vars.chi = compute_chi(coords);
26 
27  // Conformal metric is flat
28  FOR(i) vars.h[i][i] = 1.;
29 
30  vars.A = compute_A(vars.chi, coords);
31 
32  switch (m_initial_lapse)

34  case Lapse::ONE:

36  break;

42  break;
43  default:
44  MayDay::Error("BinaryBH::Supplied initial lapse not supported.");
45  }
46

48 }

58 template <class data_t>
59 Tensor<2, data_t> BinaryBH::compute_A(data_t chi,
60  Coordinates<data_t> coords) const
61 {
62 
63  Tensor<2, data_t> Aij1 = bh1.Aij(coords);
64  Tensor<2, data_t> Aij2 = bh2.Aij(coords);
65  Tensor<2, data_t> out;
66 
67  // Aij(CCZ4) = psi^(-6) * Aij(Baumgarte&Shapiro book)
68  FOR(i, j) out[i][j] = pow(chi, 3 / 2.) * (Aij1[i][j] + Aij2[i][j]);
