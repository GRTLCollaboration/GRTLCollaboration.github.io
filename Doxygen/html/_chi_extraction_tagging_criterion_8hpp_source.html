55  template <class data_t> void compute(Cell<data_t> current_cell) const
56  {
57  // first test the gradients for regions of high curvature
58  const auto d2 = m_deriv.template diff2<Vars>(current_cell);
59  data_t mod_d2_chi = 0;
60  FOR(idir, jdir)
61  {
62  mod_d2_chi += d2.chi[idir][jdir] * d2.chi[idir][jdir];
63  }
64  data_t criterion = m_dx * sqrt(mod_d2_chi);
65 
66  // if extracting weyl data at a given radius, enforce a given resolution
67  // there
68  if (m_activate_extraction)
69  {
70  for (int iradius = 0; iradius < m_params.num_extraction_radii;
71  ++iradius)
72  {
73  // regrid if within extraction level and not at required
74  // refinement
75  if (m_level < m_params.extraction_levels[iradius])
76  {
77  const Coordinates<data_t> coords(current_cell, m_dx,
78  m_params.center);
79  const data_t r = coords.get_radius();
80  // add a 20% buffer to extraction zone so not too near to
81  // boundary
82  auto regrid = simd_compare_lt(
83  r, 1.2 * m_params.extraction_radii[iradius]);
84  criterion = simd_conditional(regrid, 100.0, criterion);
85  }
86  }
87  }
88 
89  // Write back into the flattened Chombo box
