81 void add_mode_integrand(
82 int es, int el, int em, const complex_function_t &a_function,
83 std::pair<std::vector<double>, std::vector<double>> &out_integrals,
84 const IntegrationMethod &a_method_theta = IntegrationMethod::simpson,
85 const IntegrationMethod &a_method_phi = IntegrationMethod::trapezium,
86 const bool a_broadcast_integral = false)
87 {
88 auto integrand_re = [center = m_center, &geom = m_geom, es, el, em,
89 &a_function](std::vector<double> &a_data_here,
90 double r, double theta, double phi)
91 {
92 // note that spin_Y_lm requires the coordinates with the center
93 // at the origin
94 double x = geom.get_grid_coord(0, r, theta, phi) - center[0];
95 double y = geom.get_grid_coord(1, r, theta, phi) - center[1];
96 double z = geom.get_grid_coord(2, r, theta, phi) - center[2];
97 SphericalHarmonics::Y_lm_t<double> Y_lm =
98 SphericalHarmonics::spin_Y_lm(x, y, z, es, el, em);
99 auto function_here = a_function(a_data_here, r, theta, phi);
100 return (function_here.first * Y_lm.Real +
101 function_here.second * Y_lm.Im) /
102 (r * r);
103 };
104 add_integrand(integrand_re, out_integrals.first, a_method_theta,
105 a_method_phi, a_broadcast_integral);
106
107 auto integrand_im = [center = m_center, &geom = m_geom, es, el, em,
108 &a_function](std::vector<double> &a_data_here,
109 double r, double theta, double phi)
110 {
111 // note that spin_Y_lm requires the coordinates with the center
112 // at the origin
113 double x = geom.get_grid_coord(0, r, theta, phi) - center[0];
114 double y = geom.get_grid_coord(1, r, theta, phi) - center[1];
115 double z = geom.get_grid_coord(2, r, theta, phi) - center[2];
116 SphericalHarmonics::Y_lm_t<double> Y_lm =
117 SphericalHarmonics::spin_Y_lm(x, y, z, es, el, em);
118 auto function_here = a_function(a_data_here, r, theta, phi);
119 return (function_here.second * Y_lm.Real -
120 function_here.first * Y_lm.Im) /
121 (r * r);
122 };
123 add_integrand(integrand_im, out_integrals.second, a_method_theta,
124 a_method_phi, a_broadcast_integral);
125 }
