37 double val;

40 stop = true;

42 if (stop)

44#pragma omp single

50 pout() << UserVariables::variable_names[ivar] << ": "
51 << current_cell.load_vars(ivar) << std::endl;
52 }
53 pout() << "Integer coordinates: " << current_cell.get_int_vect()
54 << std::endl;
