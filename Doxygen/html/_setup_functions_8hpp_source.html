48void mainSetup(int argc, char *argv[])
49{
50#ifdef CH_MPI
51 int mpi_requested_thread_support = MPI_THREAD_SINGLE;
52#ifdef _OPENMP
53 if (omp_get_max_threads() > 1)
54 {
55 // MR: I don't think we need MPI_THREAD_MULTIPLE but it might be safer..
56 mpi_requested_thread_support = MPI_THREAD_SERIALIZED;
57 }
58#endif
59 int mpi_provided_thread_support;
60 // Start MPI
61 MPI_Init_thread(&argc, &argv, mpi_requested_thread_support,
62 &mpi_provided_thread_support);
63 std::string mpi_provided_thread_support_str;
64 switch (mpi_provided_thread_support)
65 {
66 case MPI_THREAD_SINGLE:
67 mpi_provided_thread_support_str = "single";
68 break;
69 case MPI_THREAD_FUNNELED:
70 mpi_provided_thread_support_str = "funneled";
71 break;
72 case MPI_THREAD_SERIALIZED:
73 mpi_provided_thread_support_str = "serialized";
74 break;
75 case MPI_THREAD_MULTIPLE:
76 default:
77 mpi_provided_thread_support_str = "multiple";
78 break;
79 }
80#ifdef CH_AIX
81 H5dont_atexit();
82#endif
83// setChomboMPIErrorHandler();
84#endif
85
86 int rank, number_procs;
87#ifdef CH_MPI
88 MPI_Comm_rank(Chombo_MPI::comm, &rank);
89 MPI_Comm_size(Chombo_MPI::comm, &number_procs);
90#else
91 rank = 0;
92 number_procs = 1;
93#endif
94
95#ifdef EQUATION_DEBUG_MODE
96 EquationDebugging::check_no_omp();
97 MayDay::Warning("GRChombo is running in equation debug mode. This mode is "
98 "intended only for debugging and leads to significantly "
99 "worse performance.");
100#endif
101
102 if (rank == 0)
103 {
104 std::cout << " number_procs = " << number_procs << endl;
105#ifdef _OPENMP
106 std::cout << " threads = " << omp_get_max_threads() << endl;
107#ifdef CH_MPI
108 std::cout << " mpi provided thread support = "
109 << mpi_provided_thread_support_str << endl;
110#endif
111#endif
112 std::cout << " simd width (doubles) = " << simd_traits<double>::simd_len
113 << endl;
114 }
115
116 const int required_argc = 2;
117 if (argc < required_argc)
118 {
119 cerr << " usage " << argv[0] << " <input_file_name> " << endl;
120 exit(0);
121 }
122}

135void setupAMRObject(GRAMR &gr_amr, AMRLevelFactory &a_factory)
136{
137 // Reread the params - just the base ones
138 // Note that we could have passed these through the function
139 // but this way preserves backwards compatibility
140 GRParmParse pp;
141 ChomboParameters chombo_params(pp);
142
143 // set size of box
144 Box problem_domain(IntVect::Zero, chombo_params.ivN);
145 ProblemDomain physdomain(problem_domain);
146
147 // set periodicity
148 for (int dir = 0; dir < SpaceDim; dir++)
149 {
150 physdomain.setPeriodic(dir,
151 chombo_params.boundary_params.is_periodic[dir]);
152 }
153
154 // Define the AMR object
155 gr_amr.define(chombo_params.max_level, chombo_params.ref_ratios, physdomain,
156 &a_factory);
157
158 // The buffer defines the minimum number of level l cells there have to be
159 // between level l+1 and level l-1
160 // It needs to be at least ceil(num_ghosts/max_ref_ratio) for proper nesting
161 gr_amr.gridBufferSize(chombo_params.grid_buffer_size);
162
163 // set checkpoint and plot intervals and prefixes
164#ifdef CH_USE_HDF5
165 gr_amr.checkpointInterval(chombo_params.checkpoint_interval);
166 gr_amr.checkpointPrefix(chombo_params.hdf5_path +
167 chombo_params.checkpoint_prefix);
168 if (chombo_params.plot_interval != 0)
169 {
170 gr_amr.plotInterval(chombo_params.plot_interval);
171 gr_amr.plotPrefix(chombo_params.hdf5_path + chombo_params.plot_prefix);
172 }
173#endif
174
175 // Number of coarse time steps from one regridding to the next
176 gr_amr.regridIntervals(chombo_params.regrid_interval);
177
178 // max and min box sizes, fill ratio determining accuracy of regrid
179 gr_amr.maxGridSize(chombo_params.max_grid_size);
180 gr_amr.blockFactor(chombo_params.block_factor);
181 gr_amr.fillRatio(chombo_params.fill_ratio);
182
183 // Set verbosity
184 gr_amr.verbosity(chombo_params.verbosity);
185
186 // Set timeEps to half of finest level dt
187 // Chombo sets it to 1.e-6 by default (AMR::setDefaultValues in AMR.cpp)
188 // This is only not enough for >~20 levels
189 double eps = 1.;
190 for (int ilevel = 0; ilevel < chombo_params.max_level; ++ilevel)
191 eps /= chombo_params.ref_ratios[ilevel];
192 gr_amr.timeEps(std::min(1.e-6, eps / 2.));
193
194 // Set up input files
195 if (!chombo_params.restart_from_checkpoint)
196 {
197#ifdef CH_USE_HDF5
198 if (!FilesystemTools::directory_exists(chombo_params.hdf5_path))
199 FilesystemTools::mkdir_recursive(chombo_params.hdf5_path);
200#endif
201
202 gr_amr.setupForNewAMRRun();
203 }
204 else
205 {
206#ifdef CH_USE_HDF5
207 HDF5Handle handle(chombo_params.restart_file, HDF5Handle::OPEN_RDONLY);
208 // read from checkpoint file
209 gr_amr.setupForRestart(handle);
210 handle.close();
211#else
212 MayDay::Error("GRChombo restart only defined with hdf5");
213#endif
214 }
215}
