63 inline double get_grid_coord(int a_dir, double a_radius, double a_theta,
64 double a_phi) const
65 {
66 switch (a_dir)
67 {
68 case (0):
69 return m_center[0] + a_radius * sin(a_theta) * cos(a_phi);
70 case (1):
71 return m_center[1] + a_radius * sin(a_theta) * sin(a_phi);
72 case (2):
73 return m_center[2] + a_radius * cos(a_theta);
74 default:
