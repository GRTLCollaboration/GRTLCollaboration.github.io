44 for (int j = 0; j < CH_SPACEDIM * (CH_SPACEDIM + 1) / 2; ++j)
45 d2[k][j].resize(total);
46
47 query.addComp(var, &d2[k][0][0], Derivative::dxdx, type)
48 .addComp(var, &d2[k][1][0], Derivative::dxdy, type);
49#if CH_SPACEDIM == 3
50 query.addComp(var, &d2[k][2][0], Derivative::dxdz, type)
51 .addComp(var, &d2[k][3][0], Derivative::dydy, type)
52 .addComp(var, &d2[k][4][0], Derivative::dydz, type)
53 .addComp(var, &d2[k][5][0], Derivative::dzdz, type);
54#elif CH_SPACEDIM == 2
55 query.addComp(var, &d2[k][2][0], Derivative::dydy, type);

57 }

63{
64 AHVarsData<key, double> data;
65
66 for (auto &vec : a_data.vars)
67 data.vars[vec.first] = (vec.second[idx]);
68
69 for (auto &vec : a_data.d1)
70 {
71 data.d1[vec.first] = {0.};
72 for (int i = 0; i < CH_SPACEDIM; ++i)
73 data.d1[vec.first][i] = vec.second[i][idx];
74 }
75
76 for (auto &vec : a_data.d2)
77 {
78 data.d2[vec.first] = {0.};
79 for (int i = 0; i < CH_SPACEDIM * (CH_SPACEDIM + 1) / 2; ++i)
80 data.d2[vec.first][i] = vec.second[i][idx];
81 }
82
83 return data;
84}
