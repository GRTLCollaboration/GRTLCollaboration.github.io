97 ALWAYS_INLINE double get_grid_coord(int a_dir, double a_radius,
98 double a_theta, double a_phi = 0.) const
99 {
100 double center = (a_dir < CH_SPACEDIM ? m_center[a_dir] : 0.);
101 switch ((a_dir + 2 - m_up_dir) % 3)
102 {
103 case (0):
104 return center + a_radius * sin(a_theta) * cos(a_phi);
105 case (1):
106 return center + a_radius * sin(a_theta) * sin(a_phi);
107 case (2):
108 return center + a_radius * cos(a_theta);
109 default:
110 MayDay::Error("SphericalGeometry: Direction not supported");
111 }
112 }

114 static const IntegrationMethod &
115 get_recommended_integration_method_u(int a_num_points_theta)
116 {
117 static const IntegrationMethod &simpson = IntegrationMethod::simpson;
118 static const IntegrationMethod &trapezium =
119 IntegrationMethod::trapezium;
120 if (simpson.is_valid(a_num_points_theta, is_u_periodic()))
121 return simpson;
122 MayDay::Warning("Use an odd number of 'u' points to use simpson rule. "
123 "Defaulting to trapezium.");
