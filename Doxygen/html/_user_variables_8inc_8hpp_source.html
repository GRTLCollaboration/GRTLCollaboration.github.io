72// where one has read in a subset of variables with some feature
73// this reads in a set of associated values and assigns it into a full
74// array of all NUM_VARS vars (setting other values to a default value)
75template <class T>
76void load_values_to_array(
77 GRParmParse &pp, const char *a_values_vector_string,
78 const std::vector<std::pair<int, VariableType>> &a_vars_vector,
79 std::array<T, NUM_VARS> &a_values_array, const T a_default_value)
80{
81 // how many values do I need to get?
82 int num_values = a_vars_vector.size();
83 // make a container for them, and load
84 std::vector<T> vars_values(num_values, a_default_value);
85 pp.load(a_values_vector_string, vars_values, num_values, vars_values);
86
87 // populate the values_array for the NUM_VARS values with those read in
88 a_values_array.fill(a_default_value);
89 for (int i = 0; i < num_values; i++)
90 {
91 int icomp = a_vars_vector[i].first;
92 CH_assert(a_vars_vector[i].second == VariableType::evolution);
93 a_values_array[icomp] = vars_values[i];
94 }
95}

97// function to create a vector of enums of vars by reading in their
98// names as strings from the params file and converting it to the enums
99inline void
100load_vars_to_vector(GRParmParse &pp, const char *a_vars_vector_string,
101 const char *a_vector_size_string,
102 std::vector<std::pair<int, VariableType>> &a_vars_vector,
103 int &a_vars_vector_size)
104{
105 int num_values;
106 pp.load(a_vector_size_string, num_values, -1);
107 // only set a_vars_vector and a_var_vector_size if a_vector_size_string
108 // found
109 if (num_values >= 0)
110 {
111 std::vector<std::string> var_names(num_values, "");
112 pp.load(a_vars_vector_string, var_names, num_values, var_names);
113 for (std::string var_name : var_names)
114 {
115 // first assume the variable is a normal evolution var
116 int var = UserVariables::variable_name_to_enum(var_name);
117 VariableType var_type = VariableType::evolution;
118 if (var < 0)
119 {
120 // if not an evolution var check if it's a diagnostic var
121 var = DiagnosticVariables::variable_name_to_enum(var_name);
122 if (var < 0)
123 {
124 // it's neither :(
125 pout() << "Variable with name " << var_name << " not found."
126 << endl;
127 }
128 else
129 {
130 var_type = VariableType::diagnostic;
131 }
132 }
133 if (var >= 0)
134 {
135 a_vars_vector.emplace_back(var, var_type);
136 }
137 }
138 // overwrites read in value if entries have been ignored
139 a_vars_vector_size = a_vars_vector.size();
140 }
141}
