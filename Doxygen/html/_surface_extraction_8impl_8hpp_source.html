15template <class SurfaceGeometry>
16SurfaceExtraction<SurfaceGeometry>::SurfaceExtraction(
17 const SurfaceGeometry &a_geom, const surface_extraction_params_t &a_params,
18 double a_dt, double a_time, bool a_first_step, double a_restart_time)
19 : m_geom(a_geom), m_params(a_params), m_dt(a_dt), m_time(a_time),
20 m_first_step(a_first_step), m_restart_time(a_restart_time),
21 m_num_interp_points((procID() == 0) ? D_TERM(m_params.num_surfaces,
22 *m_params.num_points_u,
23 *m_params.num_points_v)
24 : 0),
25 m_du(m_geom.du(m_params.num_points_u)),
26 m_dv(m_geom.dv(m_params.num_points_v)), m_done_extraction(false)
27{
28 // check folders only in first two timesteps
29 // (or at m_first_step if this is not the first two timesteps)
30 if (m_time < m_restart_time + 1.5 * m_dt || m_first_step)
31 {
32 if (!FilesystemTools::directory_exists(m_params.data_path))
33 FilesystemTools::mkdir_recursive(m_params.data_path);
34
35 if (m_params.write_extraction &&
36 !FilesystemTools::directory_exists(m_params.extraction_path))
37 FilesystemTools::mkdir_recursive(m_params.extraction_path);
38 }
39
40 // only interp points on rank 0
41 if (procID() == 0)
42 {
43 FOR(idir) { m_interp_coords[idir].resize(m_num_interp_points); }
44
45 for (int isurface = 0; isurface < m_params.num_surfaces; ++isurface)
46 {
47 double surface_param_value =
48 m_params.surface_param_values[isurface];
49 for (int iu = 0; iu < m_params.num_points_u; ++iu)
50 {
51 double u = m_geom.u(iu, m_params.num_points_u);
52#if CH_SPACEDIM == 3
53 for (int iv = 0; iv < m_params.num_points_v; ++iv)
54 {
55 double v = m_geom.v(iv, m_params.num_points_v);
56#endif
57 FOR(idir)
58 {
59 int idx = index(D_DECL(isurface, iu, iv));
60 m_interp_coords[idir][idx] = m_geom.get_grid_coord(
61 idir, D_DECL(surface_param_value, u, v));
62 }
63#if CH_SPACEDIM == 3
64 }
65#endif
66 }
67 }
68 }

143template <class SurfaceGeometry>
144template <typename InterpAlgo>
145void SurfaceExtraction<SurfaceGeometry>::extract(
146 AMRInterpolator<InterpAlgo> *a_interpolator)
147{
148 if (a_interpolator == nullptr)
149 {
150 MayDay::Error("SurfaceExtraction: invalid AMRInterpolator pointer");
151 }
152 // m_num_interp_points is 0 on ranks > 0
153 InterpolationQuery query(m_num_interp_points);
154 FOR(idir) { query.setCoords(idir, m_interp_coords[idir].data()); }
155 for (int ivar = 0; ivar < m_vars.size(); ++ivar)
156 {
157 // note the difference in order between the m_vars tuple in this class
158 // and the InterpolationQuery::out_t type
159 query.addComp(std::get<0>(m_vars[ivar]), m_interp_data[ivar].data(),
160 std::get<2>(m_vars[ivar]), std::get<1>(m_vars[ivar]));
161 }
162
163 // submit the query
164 a_interpolator->interp(query);
165 m_done_extraction = true;
166}

171template <class SurfaceGeometry>
172void SurfaceExtraction<SurfaceGeometry>::add_integrand(
173 const integrand_t &a_integrand, std::vector<double> &out_integrals,
174 const IntegrationMethod &a_method_u, const IntegrationMethod &a_method_v,
175 const bool a_broadcast_integral)
176{
177 // if broadcasting integral to all ranks, all ranks need to know about it
178 m_broadcast_integrals.push_back(a_broadcast_integral);
179 if (a_broadcast_integral || procID() == 0)
180 {
181 // resize the out_integrals and store a reference to it
182 out_integrals.resize(m_params.num_surfaces);
183 std::fill(out_integrals.begin(), out_integrals.end(), 0.0);
184 }
185 m_integrals.push_back(std::ref(out_integrals));
186
187 // only rank 0 actually does the integration and needs to know about the
188 // integrand and integration method
189 if (procID() == 0)
190 {
191 // store the integrand
192 m_integrands.push_back(a_integrand);
193
194 // check if integration methods are valid given periodicity and number
195 // of points
196 bool valid_u =
197 a_method_u.is_valid(m_params.num_points_u, m_geom.is_u_periodic());
198 bool valid_v =
199 a_method_v.is_valid(m_params.num_points_v, m_geom.is_v_periodic());
200
201 // default to using the trapezium rule if provided methods are not valid
202 IntegrationMethod method_u = IntegrationMethod::trapezium;
203 IntegrationMethod method_v = IntegrationMethod::trapezium;
204 if (!valid_u)
205 {
206 MayDay::Warning(
207 "SurfaceExtraction<SurfaceGeometry>::integrate: Provided "
208 "IntegrationMethod for u is not valid with\nthis num_points_u; "
209 "reverting to trapezium rule.");
210 }
211 else
212 {
213 method_u = a_method_u;
214 }
215 if (!valid_v)
216 {
217 MayDay::Warning(
218 "SurfaceExtraction<SurfaceGeometry>::integrate: Provided "
219 "IntegrationMethod for v is not valid with\nthis num_points_v; "
220 "reverting to trapezium rule.");
221 }
222 else
223 {
224 method_v = a_method_v;
225 }
226 m_integration_methods.push_back({method_u, method_v});
227 }

247template <class SurfaceGeometry>
248void SurfaceExtraction<SurfaceGeometry>::integrate()
249{
250 CH_assert(m_done_extraction);
251 if (procID() == 0)
252 {
253 // note this condition won't be true on other ranks
254 CH_assert(m_integrands.size() == m_integration_methods.size() &&
255 m_integrals.size() > 0);
256 int num_integrals = m_integrals.size();
257
258 for (int isurface = 0; isurface < m_params.num_surfaces; ++isurface)
259 {
260 double surface_param_value =
261 m_params.surface_param_values[isurface];
262 for (int iu = 0; iu < m_params.num_points_u; ++iu)
263 {
264 double u = m_geom.u(iu, m_params.num_points_u);
265 std::vector<double> inner_integral(num_integrals, 0.0);
266 for (int iv = 0; iv < m_params.num_points_v; ++iv)
267 {
268 double v = m_geom.v(iv, m_params.num_points_v);
269 std::vector<double> data_here(m_vars.size());
270 for (int ivar = 0; ivar < m_vars.size(); ++ivar)
271 {
272 data_here[ivar] =
273 m_interp_data[ivar]
274 [index(D_DECL(isurface, iu, iv))];
275 }
276 for (int iintegral = 0; iintegral < num_integrals;
277 ++iintegral)
278 {
279 auto integrand = m_integrands[iintegral];
280 double integrand_with_area_element =
281 integrand(data_here, surface_param_value, u, v) *
282 m_geom.area_element(surface_param_value, u, v);
283 double weight =
284 m_integration_methods[iintegral][1].weight(
285 iv, m_params.num_points_v,
286 m_geom.is_v_periodic());
287 inner_integral[iintegral] +=
288 weight * m_dv * integrand_with_area_element;
289 }
290 }
291 for (int iintegral = 0; iintegral < num_integrals; ++iintegral)
292 {
293 double weight = m_integration_methods[iintegral][0].weight(
294 iu, m_params.num_points_u, m_geom.is_u_periodic());
295 (m_integrals[iintegral].get())[isurface] +=
296 weight * m_du * inner_integral[iintegral];
297 }
298 }
299 }
300 }
301
302 // now broadcast result to non-zero ranks if requested
303 for (int iintegral = 0; iintegral < m_integrals.size(); ++iintegral)
304 {
305 if (m_broadcast_integrals[iintegral])
306 {
307 Vector<double> broadcast_Vector;
308 if (procID() == 0)
309 broadcast_Vector = m_integrals[iintegral].get();
310 broadcast(broadcast_Vector, 0);
311 if (procID() != 0)
312 {
313 m_integrals[iintegral].get() = broadcast_Vector.stdVector();
314 }
315 }
316 }
317}

325template <class SurfaceGeometry>
326std::vector<double> SurfaceExtraction<SurfaceGeometry>::integrate(
327 integrand_t a_integrand, const IntegrationMethod &a_method_u,
328 const IntegrationMethod &a_method_v, const bool a_broadcast_integral)
329{
330 m_integrands.clear();
331 m_integration_methods.clear();
332 m_integrals.clear();
333
334 std::vector<double> out_integrals(m_params.num_surfaces, 0.0);
335 add_integrand(a_integrand, out_integrals, a_method_u, a_method_v,
336 a_broadcast_integral);
337 integrate();
338
339 return out_integrals;
340}

343template <class SurfaceGeometry>
344void SurfaceExtraction<SurfaceGeometry>::write_extraction(
345 std::string a_file_prefix) const
346{
347 CH_assert(m_done_extraction);
348 if (procID() == 0)
349 {
350 SmallDataIO extraction_file(m_params.extraction_path + a_file_prefix,
351 m_dt, m_time, m_restart_time,
352 SmallDataIO::NEW, m_first_step);
353
354 for (int isurface = 0; isurface < m_params.num_surfaces; ++isurface)
355 {
356 // Write headers
357 std::vector<std::string> header1_strings = {
358 "time = " + std::to_string(m_time) + ",",
359 m_geom.param_name() + " = " +
360 std::to_string(m_params.surface_param_values[isurface])};
361 extraction_file.write_header_line(header1_strings, "");
362 std::vector<std::string> components(m_vars.size());
363 for (int ivar = 0; ivar < m_vars.size(); ++ivar)
364 {
365 if (std::get<2>(m_vars[ivar]) != Derivative::LOCAL)
366 {
367 components[ivar] =
368 Derivative::name(std::get<2>(m_vars[ivar])) + "_";
369 }
370 else
371 {
372 components[ivar] = "";
373 }
374 if (std::get<1>(m_vars[ivar]) == VariableType::evolution)
375 {
376 components[ivar] +=
377 UserVariables::variable_names[std::get<0>(
378 m_vars[ivar])];
379 }
380 else
381 {
382 components[ivar] +=
383 DiagnosticVariables::variable_names[std::get<0>(
384 m_vars[ivar])];
385 }
386 }
387 std::vector<std::string> coords = {m_geom.u_name(),
388 m_geom.v_name()};
389 extraction_file.write_header_line(components, coords);
390
391 // Now the data
392 for (int iu = 0; iu < m_params.num_points_u; ++iu)
393 {
394 double u = m_geom.u(iu, m_params.num_points_u);
395#if CH_SPACEDIM == 3
396 for (int iv = 0; iv < m_params.num_points_v; ++iv)
397 {
398 double v = m_geom.v(iv, m_params.num_points_v);
399
400#endif
401 int idx = index(D_DECL(isurface, iu, iv));
402 std::vector<double> data(m_vars.size());
403 for (int ivar = 0; ivar < m_vars.size(); ++ivar)
404 {
405 data[ivar] = m_interp_data[ivar][idx];
406 }
407#if CH_SPACEDIM == 2
408 extraction_file.write_data_line(data, u);
409#endif
410#if CH_SPACEDIM == 3
411 extraction_file.write_data_line(data, {u, v});
412 }
413#endif
414 }
415 extraction_file.line_break();
416 }
417 }

421template <class SurfaceGeometry>
422void SurfaceExtraction<SurfaceGeometry>::write_integrals(
423 const std::string &a_filename,
424 const std::vector<std::vector<double>> &a_integrals,
425 const std::vector<std::string> &a_labels) const
426{
427 if (procID() == 0)
428 {
429 const int num_integrals_per_surface = a_integrals.size();
430 // if labels are provided there must be the same number of labels as
431 // there are integrals
432 if (!a_labels.empty())
433 {
434 CH_assert(num_integrals_per_surface == a_labels.size());
435 }
436 // each inner vector element of a_integrals must have the same number of
437 // elements as there are surfaces (i.e. one integral per surface)
438 for (auto vect : a_integrals)
439 {
440 CH_assert(vect.size() == m_params.num_surfaces);
441 }
442 // open file for writing
443 SmallDataIO integral_file(m_params.data_path + a_filename, m_dt, m_time,
444 m_restart_time, SmallDataIO::APPEND,
445 m_first_step);
446
447 // remove any duplicate data if this is a restart
448 integral_file.remove_duplicate_time_data();
449
450 if (m_first_step)
451 {
452 // make header strings
453 std::vector<std::string> header1_strings(num_integrals_per_surface *
454 m_params.num_surfaces);
455 std::vector<std::string> header2_strings(num_integrals_per_surface *
456 m_params.num_surfaces);
457 for (int isurface = 0; isurface < m_params.num_surfaces; ++isurface)
458 {
459 for (int iintegral = 0; iintegral < num_integrals_per_surface;
460 ++iintegral)
461 {
462 int idx = isurface * num_integrals_per_surface + iintegral;
463 if (a_labels.empty())
464 header1_strings[idx] = "";
465 else
466 header1_strings[idx] = a_labels[iintegral];
467 header2_strings[idx] =
468 std::to_string(m_params.surface_param_values[isurface]);
469 }
470 }
471 std::string pre_header2_string = m_geom.param_name() + " = ";
472
473 // write headers
474 integral_file.write_header_line(header1_strings);
475 integral_file.write_header_line(header2_strings,
476 pre_header2_string);
477 }
478
479 // make vector of data for writing
480 std::vector<double> data_for_writing(num_integrals_per_surface *
481 m_params.num_surfaces);
482 for (int isurface = 0; isurface < m_params.num_surfaces; ++isurface)
483 {
484 for (int iintegral = 0; iintegral < num_integrals_per_surface;
485 ++iintegral)
486 {
487 int idx = isurface * num_integrals_per_surface + iintegral;
488 data_for_writing[idx] = a_integrals[iintegral][isurface];
489 }
490 }
491
492 // write data
493 integral_file.write_time_data_line(data_for_writing);
494 }
495}
