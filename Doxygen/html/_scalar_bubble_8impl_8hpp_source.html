18 // Compute the value of the initial vars on the grid
19 template <class data_t>
20 void ScalarBubble::compute(Cell<data_t> current_cell) const
21 {
22  MatterCCZ4RHS<ScalarField<>>::Vars<data_t> vars;
23  VarsTools::assign(vars, 0.); // Set only the non-zero components below
24  Coordinates<data_t> coords(current_cell, m_dx, m_params.centerSF);
25 
26  // set the field vars
27  vars.phi = compute_phi(coords);
28  vars.Pi = 0;
29 
30  // start with unit lapse and flat metric (must be relaxed for chi)
31  vars.lapse = 1;
32  vars.chi = 1;
33 
34  // conformal metric is flat
35  FOR(i) vars.h[i][i] = 1.;
36 
37  // Store the initial values of the variables
38  current_cell.store_vars(vars);
39 }

41 // Compute the value of phi at the current point
42 template <class data_t>
43 data_t ScalarBubble::compute_phi(Coordinates<data_t> coords) const
44 {
45  data_t rr = coords.get_radius();
46  data_t rr2 = rr * rr;
47  data_t out_phi = m_params.amplitudeSF * rr2 *
48  exp(-pow((rr - m_params.r_zero) / m_params.widthSF, 2.0));
49 
50  return out_phi;
51 }
