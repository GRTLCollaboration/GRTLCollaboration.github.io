24void MatterConstraints<matter_t>::compute(Cell<data_t> current_cell) const
25{
26 // Load local vars and calculate derivs
27 const auto vars = current_cell.template load_vars<BSSNMatterVars>();
28 const auto d1 = m_deriv.template diff1<BSSNMatterVars>(current_cell);
29 const auto d2 = m_deriv.template diff2<BSSNMatterVars>(current_cell);
30
31 // Get the non matter terms for the constraints
32 Vars<data_t> out = constraint_equations(vars, d1, d2);
33
34 // Inverse metric and Christoffel symbol
35 const auto h_UU = TensorAlgebra::compute_inverse_sym(vars.h);
36 const auto chris = TensorAlgebra::compute_christoffel(d1.h, h_UU);
37
38 // Energy Momentum Tensor
39 const auto emtensor = my_matter.compute_emtensor(vars, d1, h_UU, chris.ULL);
40
41 // Hamiltonian constraint
42 out.Ham += -16.0 * M_PI * m_G_Newton * emtensor.rho;
43
44 // Momentum constraints
45 FOR(i) { out.Mom[i] += -8.0 * M_PI * m_G_Newton * emtensor.Si[i]; }
46
47 // Write the constraints into the output FArrayBox
48 current_cell.store_vars(out);
49}
