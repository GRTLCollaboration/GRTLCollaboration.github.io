48 void mainSetup(int argc, char *argv[])
49 {

51  // Start MPI
52  MPI_Init(&argc, &argv);
53 #ifdef CH_AIX
54  H5dont_atexit();
55 #endif
56 // setChomboMPIErrorHandler();
57 #endif
58 
59  int rank, number_procs;
60 #ifdef CH_MPI
61  MPI_Comm_rank(Chombo_MPI::comm, &rank);
62  MPI_Comm_size(Chombo_MPI::comm, &number_procs);
63 #else
64  rank = 0;
65  number_procs = 1;
66 #endif
67 
68 #ifdef EQUATION_DEBUG_MODE
69  EquationDebugging::check_no_omp();
70  MayDay::Warning("GRChombo is running in equation debug mode. This mode is "
71  "intended only for debugging and leads to significantly "
72  "worse performance.");
73 #endif
74 
75  if (rank == 0)
76  {
77  std::cout << " number_procs = " << number_procs << endl;
78 #ifdef _OPENMP
79  std::cout << " threads = " << omp_get_max_threads() << endl;

82  << endl;
83  }
84 
85  const int required_argc = 2;
86  if (argc < required_argc)
87  {
88  cerr << " usage " << argv[0] << " <input_file_name> " << endl;

104 void setupAMRObject(GRAMR &gr_amr, AMRLevelFactory &a_factory)
105 {
106  // Reread the params - just the base ones
107  // Note that we could have passed these through the function
108  // but this way preserves backwards compatibility
109  GRParmParse pp;
110  ChomboParameters chombo_params(pp);
111 
112  // set size of box
113  Box problem_domain(IntVect::Zero, chombo_params.ivN);
114  ProblemDomain physdomain(problem_domain);
115 
116  // set periodicity
117  for (int dir = 0; dir < SpaceDim; dir++)
118  {
119  physdomain.setPeriodic(dir,
120  chombo_params.boundary_params.is_periodic[dir]);
121  }
122 
123  // Define the AMR object
124  gr_amr.define(chombo_params.max_level, chombo_params.ref_ratios, physdomain,
125  &a_factory);
126 
127  // The buffer defines the minimum number of level l cells there have to be
128  // between level l+1 and level l-1
129  // It needs to be at least ceil(num_ghosts/max_ref_ratio) for proper nesting
130  gr_amr.gridBufferSize(chombo_params.grid_buffer_size);
131 
132  // set checkpoint and plot intervals and prefixes
133 #ifdef CH_USE_HDF5
134  gr_amr.checkpointInterval(chombo_params.checkpoint_interval);
135  gr_amr.checkpointPrefix(chombo_params.hdf5_path +
136  chombo_params.checkpoint_prefix);
137  if (chombo_params.plot_interval != 0)
138  {
139  gr_amr.plotInterval(chombo_params.plot_interval);
140  gr_amr.plotPrefix(chombo_params.hdf5_path + chombo_params.plot_prefix);
141  }
142 #endif
143 
144  // Number of coarse time steps from one regridding to the next
145  gr_amr.regridIntervals(chombo_params.regrid_interval);
146 
147  // max and min box sizes, fill ratio determining accuracy of regrid
148  gr_amr.maxGridSize(chombo_params.max_grid_size);
149  gr_amr.blockFactor(chombo_params.block_factor);
150  gr_amr.fillRatio(chombo_params.fill_ratio);
151 
152  // Set verbosity
153  gr_amr.verbosity(chombo_params.verbosity);
154 
155  // Set timeEps to half of finest level dt
156  // Chombo sets it to 1.e-6 by default (AMR::setDefaultValues in AMR.cpp)
157  // This is only not enough for >~20 levels
158  double eps = 1.;
159  for (int ilevel = 0; ilevel < chombo_params.max_level; ++ilevel)
160  eps /= chombo_params.ref_ratios[ilevel];
161  gr_amr.timeEps(std::min(1.e-6, eps / 2.));
162 
163  // Set up input files
164  if (!chombo_params.restart_from_checkpoint)
165  {
166 #ifdef CH_USE_HDF5
167  if (!FilesystemTools::directory_exists(chombo_params.hdf5_path))
168  FilesystemTools::mkdir_recursive(chombo_params.hdf5_path);
169 #endif
170 
171  gr_amr.setupForNewAMRRun();
172  }
173  else
174  {
175 #ifdef CH_USE_HDF5
176  HDF5Handle handle(chombo_params.hdf5_path + chombo_params.restart_file,
177  HDF5Handle::OPEN_RDONLY);
178  // read from checkpoint file
179  gr_amr.setupForRestart(handle);
180  handle.close();
181 #else
182  MayDay::Error("GRChombo restart only defined with hdf5");
183 #endif
184  }
185 }
