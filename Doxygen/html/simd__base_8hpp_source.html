18// This file provides functionalities for simd<t> whose implementation is
19// architecture independent. Most importantly, all simd functions which have no
20// associated vector intrinsics, e.g. calls like ostream operator <<, are
21// defined in the same way for all architectures and are, therefore, in this
22// file. A more subtle example:
23//*= is an operator which has an architecture dependent vector intrinsic
24// associated to it and is, therefore, not defined in this file. However, the
25// binary operation a*b can be rewritten as { a *= b; return a; } for all
26// architectures and, therefore, is defined in this file.

45 // Note: These binary ops allow us to write e.g. simd<double>+int
46 // or 2*y etc. The "friend function" construction makes
47 // sure that these non-member binary operations
48 // are instantiated at the same time as simd<t>.
49 // The alternative is to template
50 // on the two operand types, however this gives rise to
51 // ambiguity because we have defined
52 // casts between t and simd<t>
53 friend ALWAYS_INLINE simd<t> operator+(const simd<t> &a, const simd<t> &b)

91 template <typename op_t> ALWAYS_INLINE simd<t> foreach (op_t op) const
92 {
93 t in_arr[simd_traits<t>::simd_len];
94 t out_arr[simd_traits<t>::simd_len];
95 simd<t>::store(in_arr, m_value);
96
97#pragma omp simd
98 for (int i = 0; i < simd_traits<t>::simd_len; ++i)
99 {
100 out_arr[i] = op(in_arr[i]);
101 }
102
103 return simd<t>::load(out_arr);
104 }
105
106 template <typename op_t>
107 ALWAYS_INLINE simd<t> foreach (op_t op, simd<t> arg) const
108 {
109 t in_arr[simd_traits<t>::simd_len];
110 t arg_arr[simd_traits<t>::simd_len];
111 t out_arr[simd_traits<t>::simd_len];
112 simd<t>::store(in_arr, m_value);
113 simd<t>::store(arg_arr, arg);
114
115#pragma omp simd
116 for (int i = 0; i < simd_traits<t>::simd_len; ++i)
117 {
118 out_arr[i] = op(in_arr[i], arg_arr[i]);
119 }
120
121 return simd<t>::load(out_arr);
122 }
123};

125#define define_simd_overload(op) \
126 template <typename t> ALWAYS_INLINE simd<t> op(const simd<t> &a) \
127 { \
128 return a.foreach (([&](t x) { return op(x); })); \

131#define define_binary_simd_overload(op) \
132 template <typename t> \
133 ALWAYS_INLINE simd<t> op(const simd<t> &a, const simd<t> &b) \
134 { \
135 return a.foreach (([&](t x, t arg) { return op(x, arg); }), b); \
