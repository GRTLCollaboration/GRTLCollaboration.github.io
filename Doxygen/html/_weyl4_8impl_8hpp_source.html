13 template <class data_t> void Weyl4::compute(Cell<data_t> current_cell) const
14 {
15 
16  // copy data from chombo gridpoint into local variables
17  const auto vars = current_cell.template load_vars<Vars>();
18  const auto d1 = m_deriv.template diff1<Vars>(current_cell);
19  const auto d2 = m_deriv.template diff2<Diff2Vars>(current_cell);
20 
21  // Get the coordinates
22  const Coordinates<data_t> coords(current_cell, m_dx, m_center);
23 
24  // Compute the inverse metric and Christoffel symbols
25  using namespace TensorAlgebra;
26  const auto h_UU = compute_inverse_sym(vars.h);
27  const auto chris = compute_christoffel(d1.h, h_UU);
28 
29  // Compute the spatial volume element
30  const auto epsilon3_LUU = compute_epsilon3_LUU(vars, h_UU);
31 
32  // Compute the E and B fields
33  EBFields_t<data_t> ebfields =
34  compute_EB_fields(vars, d1, d2, epsilon3_LUU, h_UU, chris);
35 
36  // work out the Newman Penrose scalar
37  NPScalar_t<data_t> out =
38  compute_Weyl4(ebfields, vars, d1, d2, h_UU, coords);
39 
40  // Write the rhs into the output FArrayBox
41  current_cell.store_vars(out.Real, c_Weyl4_Re);
42  current_cell.store_vars(out.Im, c_Weyl4_Im);
43 }

45 template <class data_t>
46 Tensor<3, data_t>
47 Weyl4::compute_epsilon3_LUU(const Vars<data_t> &vars,
48  const Tensor<2, data_t> &h_UU) const
49 {
50  // raised normal vector, NB index 3 is time
51  data_t n_U[4];
52  n_U[3] = 1. / vars.lapse;
53  FOR(i) { n_U[i] = -vars.shift[i] / vars.lapse; }
54 
55  // 4D levi civita symbol and 3D levi civita tensor in LLL and LUU form
56  const auto epsilon4 = TensorAlgebra::epsilon4D();
57  Tensor<3, data_t> epsilon3_LLL;
58  Tensor<3, data_t> epsilon3_LUU;
59 
60  // Projection of antisymmentric Tensor onto hypersurface - see 8.3.17,
61  // Alcubierre
62  FOR(i, j, k)
63  {
64  epsilon3_LLL[i][j][k] = 0.0;
65  epsilon3_LUU[i][j][k] = 0.0;
66  }
67  // projection of 4-antisymetric tensor to 3-tensor on hypersurface
68  // note last index contracted as per footnote 86 pg 290 Alcubierre
69  FOR(i, j, k)
70  {
71  for (int l = 0; l < 4; ++l)
72  {
73  epsilon3_LLL[i][j][k] += n_U[l] * epsilon4[i][j][k][l] *
74  vars.lapse / (vars.chi * sqrt(vars.chi));
75  }
76  }
77  // rasing indices
78  FOR(i, j, k)
79  {
80  FOR(m, n)
81  {
82  epsilon3_LUU[i][j][k] += epsilon3_LLL[i][m][n] * h_UU[m][j] *
83  vars.chi * h_UU[n][k] * vars.chi;
84  }
85  }
86 
87  return epsilon3_LUU;
88 }

90 // Calculation of E and B fields, using tetrads from gr-qc/0104063
91 // BSSN expressions from Alcubierre book
92 // CCZ4 expressions calculated by MR and checked with TF see:
93 // https://www.overleaf.com/read/tvqjbyhvqqtp
94 template <class data_t>
95 EBFields_t<data_t> Weyl4::compute_EB_fields(
96  const Vars<data_t> &vars, const Vars<Tensor<1, data_t>> &d1,
97  const Diff2Vars<Tensor<2, data_t>> &d2,
98  const Tensor<3, data_t> &epsilon3_LUU, const Tensor<2, data_t> &h_UU,
99  const chris_t<data_t> &chris) const
100 {
101  EBFields_t<data_t> out;
102 
103  // Extrinsic curvature
104  Tensor<2, data_t> K_tensor;
105  Tensor<3, data_t> d1_K_tensor;
106  Tensor<3, data_t> covariant_deriv_K_tensor;
107 
108  // Compute inverse, Christoffel symbols, Ricci tensor and Z terms
109  // Note that unlike in CCZ4 equations we want R_ij + 0.5(D_iZ_j + D_jZ_i)
110  // rather than R_ij + D_iZ_j + D_jZ_i hence use compute_ricci_Z_general
111  double dZ_coeff = (m_formulation == CCZ4RHS<>::USE_CCZ4) ? 1. : 0.;
112  auto ricci_and_Z_terms = CCZ4Geometry::compute_ricci_Z_general(
113  vars, d1, d2, h_UU, chris, dZ_coeff);
114 
115  // Compute full spatial Christoffel symbols
116  using namespace TensorAlgebra;
117  const Tensor<3, data_t> chris_phys =
118  compute_phys_chris(d1.chi, vars.chi, vars.h, h_UU, chris.ULL);
119 
120  // Extrinsic curvature and corresponding covariant and partial derivatives
121  FOR(i, j)
122  {
123  K_tensor[i][j] = vars.A[i][j] / vars.chi +
124  1. / 3. * (vars.h[i][j] * vars.K) / vars.chi;
125 
126  FOR(k)
127  {
128  d1_K_tensor[i][j][k] = d1.A[i][j][k] / vars.chi -
129  d1.chi[k] / vars.chi * K_tensor[i][j] +
130  1. / 3. * d1.h[i][j][k] * vars.K / vars.chi +
131  1. / 3. * vars.h[i][j] * d1.K[k] / vars.chi;
132  }
133  }
134  // covariant derivative of K
135  FOR(i, j, k)
136  {
137  covariant_deriv_K_tensor[i][j][k] = d1_K_tensor[i][j][k];
138 
139  FOR(l)
140  {
141  covariant_deriv_K_tensor[i][j][k] +=
142  -chris_phys[l][k][i] * K_tensor[l][j] -
143  chris_phys[l][k][j] * K_tensor[i][l];
144  }
145  }
146 
147  // Use 'K-Theta' in CCZ4. Just 'K' in BSSN. Not a mistake, this is not to
148  // confuse with the typical 'K-2*Theta' that appears in the CCZ4 equations
149  data_t K_minus_theta = vars.K;
150  if (m_formulation == CCZ4RHS<>::USE_CCZ4)
151  K_minus_theta -= vars.Theta;
152 
153  // Calculate electric and magnetic fields
154  FOR(i, j)
155  {
156  out.E[i][j] = 0.0;
157  out.B[i][j] = 0.0;
158  }
159 
160  FOR(i, j)
161  {
162  out.E[i][j] +=
163  ricci_and_Z_terms.LL[i][j] + K_minus_theta * K_tensor[i][j];
164 
165  FOR(k, l)
166  {
167  out.E[i][j] +=
168  -K_tensor[i][k] * K_tensor[l][j] * h_UU[k][l] * vars.chi;
169 
170  out.B[i][j] +=
171  epsilon3_LUU[i][k][l] * covariant_deriv_K_tensor[l][j][k];
172  }
173  }
174 
175  // For CCZ4, explicit symmetrization appears naturally;
176  // For BSSN, only extra matter terms appear in the original expression, but
177  // assuming the Momentum constraints are satisfied, we can make the
178  // expression explicitly symmetric, which we enforce below
179  // (see Alcubierre chapter 8.3, from eq. 8.3.15 onwards)
180  TensorAlgebra::make_symmetric(out.B);
181 
182  // For CCZ4, Eij is explicitly trace-free;
183  // For BSSN, only extra matter terms appear in the original expression, but
184  // assuming the Hamiltonian constraint is satisfied, we can make the
185  // expression explicitly trace free, which we enforce below
186  // (see Alcubierre chapter 8.3, from eq. 8.3.15 onwards)
187  TensorAlgebra::make_trace_free(out.E, vars.h, h_UU);
188 
189  return out;
190 }

193 template <class data_t>
194 NPScalar_t<data_t> Weyl4::compute_Weyl4(const EBFields_t<data_t> &ebfields,
195  const Vars<data_t> &vars,
196  const Vars<Tensor<1, data_t>> &d1,
197  const Diff2Vars<Tensor<2, data_t>> &d2,
198  const Tensor<2, data_t> &h_UU,
199  const Coordinates<data_t> &coords) const
200 {
201  NPScalar_t<data_t> out;
202 
203  // Calculate the tetrads
204  const Tetrad_t<data_t> tetrad = compute_null_tetrad(vars, h_UU, coords);
205 
206  // Projection of Electric and magnetic field components using tetrads
207  out.Real = 0.0;
208  out.Im = 0.0;
209  FOR(i, j)
210  {
211  out.Real += 0.5 * (ebfields.E[i][j] * (tetrad.w[i] * tetrad.w[j] -
212  tetrad.v[i] * tetrad.v[j]) -
213  2.0 * ebfields.B[i][j] * tetrad.w[i] * tetrad.v[j]);
214  out.Im += 0.5 * (ebfields.B[i][j] * (-tetrad.w[i] * tetrad.w[j] +
215  tetrad.v[i] * tetrad.v[j]) -
216  2.0 * ebfields.E[i][j] * tetrad.w[i] * tetrad.v[j]);
217  }
218 
219  return out;
220 }

222 // Calculation of the null tetrad
223 // Defintions from gr-qc/0104063
224 // "The Lazarus project: A pragmatic approach to binary black hole evolutions",
225 // Baker et al.
226 template <class data_t>
227 Tetrad_t<data_t>
228 Weyl4::compute_null_tetrad(const Vars<data_t> &vars,
229  const Tensor<2, data_t> &h_UU,
230  const Coordinates<data_t> &coords) const
231 {
232  Tetrad_t<data_t> out;
233 
234  // compute coords
235  const data_t x = coords.x;
236  const double y = coords.y;
237  const double z = coords.z;
238 
239  // the alternating levi civita symbol
240  const Tensor<3, double> epsilon = TensorAlgebra::epsilon();
241 
242  // calculate the tetrad
243  out.u[0] = x;
244  out.u[1] = y;
245  out.u[2] = z;
246 
247  out.v[0] = -y;
248  out.v[1] = x;
249  out.v[2] = 0.0;
250 
251  out.w[0] = 0.0;
252  out.w[1] = 0.0;
253  out.w[2] = 0.0;
254 
255  // floor on chi
256  const data_t chi = simd_max(vars.chi, 1e-4);
257 
258  FOR(i, j, k, m)
259  {
260  out.w[i] += 1. / sqrt(chi) * h_UU[i][j] * epsilon[j][k][m] * out.v[k] *
261  out.u[m];
262  }
263 
264  // Gram Schmitt orthonormalisation
265  // Choice of orthonormalisaion to avoid frame-dragging
266  data_t omega_11 = 0.0;
267  FOR(i, j) { omega_11 += out.v[i] * out.v[j] * vars.h[i][j] / chi; }
268  FOR(i) { out.v[i] = out.v[i] / sqrt(omega_11); }
269 
270  data_t omega_12 = 0.0;
271  FOR(i, j) { omega_12 += out.v[i] * out.u[j] * vars.h[i][j] / chi; }
272  FOR(i) { out.u[i] += -omega_12 * out.v[i]; }
273 
274  data_t omega_22 = 0.0;
275  FOR(i, j) { omega_22 += out.u[i] * out.u[j] * vars.h[i][j] / chi; }
276  FOR(i) { out.u[i] = out.u[i] / sqrt(omega_22); }
277 
278  data_t omega_13 = 0.0;
279  data_t omega_23 = 0.0;
280  FOR(i, j)
281  {
282  omega_13 += out.v[i] * out.w[j] * vars.h[i][j] / chi;
283  omega_23 += out.u[i] * out.w[j] * vars.h[i][j] / chi;
284  }
285  FOR(i) { out.w[i] += -(omega_13 * out.v[i] + omega_23 * out.u[i]); }
286 
287  data_t omega_33 = 0.0;
288  FOR(i, j) { omega_33 += out.w[i] * out.w[j] * vars.h[i][j] / chi; }
289  FOR(i) { out.w[i] = out.w[i] / sqrt(omega_33); }
290 
291  return out;
292 }
