22class CylindricalGeometry
23{
24 private:
25 const std::array<double, CH_SPACEDIM> m_center;
26 const double m_z_length;
27
28 public:
29 CylindricalGeometry(const std::array<double, CH_SPACEDIM> &a_center,
30 double a_z_length)
31 : m_center(a_center), m_z_length(a_z_length)
32 {
33 }
34
35 ALWAYS_INLINE double get_domain_u_min() const { return 0.; }
36 ALWAYS_INLINE double get_domain_u_max() const { return m_z_length; }
37 ALWAYS_INLINE double get_domain_v_min() const { return 0.; }
38 ALWAYS_INLINE double get_domain_v_max() const { return 2. * M_PI; }
39
41 ALWAYS_INLINE double du(int a_num_points_z) const
42 {
43 return (get_domain_u_max() - get_domain_u_min()) / (a_num_points_z - 1);
44 }
45
47 ALWAYS_INLINE double dv(int a_num_points_phi) const
48 {
49 return (get_domain_v_max() - get_domain_v_min()) /
50 ((double)a_num_points_phi);
51 }
52
54 ALWAYS_INLINE double u(int a_iz, int a_num_points_z) const
55 {
56 return a_iz * du(a_num_points_z) -
57 ((get_domain_v_max() - get_domain_v_min()) / 2.0);
58 }
59
61 ALWAYS_INLINE double v(int a_iphi, int a_num_points_phi) const
62 {
63 return a_iphi * dv(a_num_points_phi);
64 }
65
66 ALWAYS_INLINE bool is_u_periodic() const { return false; }
67 ALWAYS_INLINE bool is_v_periodic() const { return true; }
70 ALWAYS_INLINE double get_grid_coord(int a_dir, double a_radius, double a_z,
71 double a_phi) const
72 {
73 switch (a_dir)
74 {
75 case (0):
76 return m_center[0] + a_radius * cos(a_phi);
77 case (1):
78 return m_center[1] + a_radius * sin(a_phi);
79 case (2):
80 return m_center[2] + a_z;
81 default:
82 MayDay::Error("CylindricalGeometry: Direction not supported");
83 }
84 }
85
88 ALWAYS_INLINE double area_element(double a_radius, double a_z,
89 double a_phi) const
90 {
91 return a_radius;
92 }
93
94 ALWAYS_INLINE std::string param_name() const { return "r"; }
95
96 ALWAYS_INLINE std::string u_name() const { return "z"; }
97
98 ALWAYS_INLINE std::string v_name() const { return "phi"; }
99};
