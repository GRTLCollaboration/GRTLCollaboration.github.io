15// Computes semi-isotropic Kerr solution as detailed in Liu, Etienne and Shapiro
16// 2010, arxiv gr-qc/1001.4077
17template <class data_t> void KerrBH::compute(Cell<data_t> current_cell) const
18{
19 using namespace CoordinateTransformations;
20 using namespace TensorAlgebra;
21
22 static const Tensor<1, double> z_dir = {0., 0., 1.};
23 const Tensor<1, double> spin_dir = {m_params.spin_direction[0],
24 m_params.spin_direction[1],
25 m_params.spin_direction[2]};
26
27 // define the rotation matrix needed to transform standard Cartesian (x,y,z)
28 // coordinates into the coordinates of the spin direction
29 Tensor<2, double> R = rotation_matrix(spin_dir, z_dir);
30
31 // set up vars for the metric and extrinsic curvature, shift and lapse in
32 // spherical coords
33 Tensor<2, data_t> spherical_g;
34 Tensor<2, data_t> spherical_K;
35 Tensor<1, data_t> spherical_shift;
36 data_t kerr_lapse;
37
38 // The cartesian variables and coords
39 Vars<data_t> vars;
40 Coordinates<data_t> coords(current_cell, m_dx, m_params.center);
41
42 // rotate point
43 Tensor<1, data_t> xyz = {coords.x, coords.y, coords.z};
44 xyz = transform_vector(xyz, R);
45
46 // Compute the components in spherical coords as per 1401.1548
47 compute_kerr(spherical_g, spherical_K, spherical_shift, kerr_lapse, xyz);
48
49 // work out where we are on the grid
50 data_t x = xyz[0];
51 data_t y = xyz[1];
52 data_t z = xyz[2];
53
54 // Convert spherical components to cartesian components using coordinate
55 // transform_tensor_UU
56 Tensor<2, data_t> cartesian_h =
57 spherical_to_cartesian_LL(spherical_g, x, y, z);
58 Tensor<2, data_t> cartesian_K =
59 spherical_to_cartesian_LL(spherical_K, x, y, z);
60 Tensor<1, data_t> cartesian_shift =
61 spherical_to_cartesian_U(spherical_shift, x, y, z);
62
63 // now rotate tensors back to original coordinates
64 vars.h = transform_tensor_LL(cartesian_h, R);
65 vars.A = transform_tensor_LL(cartesian_K, R);
66 vars.shift = transform_vector(cartesian_shift, R);
67
68 // Convert to BSSN vars
69 data_t deth = compute_determinant(vars.h);
70 auto h_UU = compute_inverse_sym(vars.h);
71 vars.chi = pow(deth, -1. / 3.);
72
73 // transform extrinsic curvature into A and TrK - note h is still non
74 // conformal version which is what we need here
75 vars.K = compute_trace(vars.A, h_UU);
76 make_trace_free(vars.A, vars.h, h_UU);
77
78 // Make conformal
79 FOR(i, j)
80 {
81 vars.h[i][j] *= vars.chi;
82 vars.A[i][j] *= vars.chi;
83 }
84
85 // use a pre collapsed lapse, could also use analytic one
86 // vars.lapse = kerr_lapse;
87 vars.lapse = pow(vars.chi, 0.5);
88
89 // Populate the variables on the grid
90 // NB We stil need to set Gamma^i which is NON ZERO
91 // but we do this via a separate class/compute function
92 // as we need the gradients of the metric which are not yet available
93 current_cell.store_vars(vars);
94}

96template <class data_t>
97void KerrBH::compute_kerr(Tensor<2, data_t> &spherical_g,
98 Tensor<2, data_t> &spherical_K,
99 Tensor<1, data_t> &spherical_shift,
100 data_t &kerr_lapse,
101 const Tensor<1, data_t> &coords) const
102{
103 // Kerr black hole params - mass M and spin a
104 double M = m_params.mass;
105 double a = m_params.spin;
106
107 // work out where we are on the grid
108 data_t x = coords[0];
109 data_t y = coords[1];
110 data_t z = coords[2];
111
112 // the radius, subject to a floor
113 data_t r = sqrt(D_TERM(x * x, +y * y, +z * z));
114 static const double minimum_r = 1e-6;
115 r = simd_max(r, minimum_r);
116
117 data_t r2 = r * r;
118
119 // the radius in xy plane, subject to a floor
120 data_t rho2 = simd_max(x * x + y * y, 1e-12);
121 data_t rho = sqrt(rho2);
122
123 // calculate useful position quantities
124 data_t cos_theta = z / r;
125 data_t sin_theta = rho / r;
126 data_t cos_theta2 = cos_theta * cos_theta;
127 data_t sin_theta2 = sin_theta * sin_theta;
128
129 // calculate useful metric quantities
130 double r_plus = M + sqrt(M * M - a * a);
131 double r_minus = M - sqrt(M * M - a * a);
132
133 // The Boyer-Lindquist coordinate
134 data_t r_BL = r * pow(1.0 + 0.25 * r_plus / r, 2.0);
135
136 // Other useful quantities per 1001.4077
137 data_t Sigma = r_BL * r_BL + a * a * cos_theta2;
138 data_t Delta = r_BL * r_BL - 2.0 * M * r_BL + a * a;
139 // In the paper this is just 'A', but not to be confused with A_ij
140 data_t AA = pow(r_BL * r_BL + a * a, 2.0) - Delta * a * a * sin_theta2;
141 // The rr component of the conformal spatial matric
142 data_t gamma_rr =
143 Sigma * pow(r + 0.25 * r_plus, 2.0) / (r * r2 * (r_BL - r_minus));
144
145 // Metric in semi isotropic Kerr-Schild coordinates, r, theta (t or th), phi
146 // (p)
147 FOR(i, j) { spherical_g[i][j] = 0.0; }
148 spherical_g[0][0] = gamma_rr; // gamma_rr
149 spherical_g[1][1] = Sigma; // gamma_tt
150 spherical_g[2][2] = AA / Sigma * sin_theta2; // gamma_pp
151
152 // Extrinsic curvature
153 FOR(i, j) { spherical_K[i][j] = 0.0; }
154
155 // set non zero elements of Krtp - K_rp, K_tp
156 spherical_K[0][2] =
157 a * M * sin_theta2 / (Sigma * sqrt(AA * Sigma)) *
158 (3.0 * pow(r_BL, 4.0) + 2 * a * a * r_BL * r_BL - pow(a, 4.0) -
159 a * a * (r_BL * r_BL - a * a) * sin_theta2) *
160 (1.0 + 0.25 * r_plus / r) / sqrt(r * r_BL - r * r_minus);
161 spherical_K[2][0] = spherical_K[0][2];
162 spherical_K[2][1] = -2.0 * pow(a, 3.0) * M * r_BL * cos_theta * sin_theta *
163 sin_theta2 / (Sigma * sqrt(AA * Sigma)) *
164 (r - 0.25 * r_plus) * sqrt(r_BL / r - r_minus / r);
165 spherical_K[1][2] = spherical_K[2][1];
166
167 // set the analytic lapse
168 kerr_lapse = sqrt(Delta * Sigma / AA);
169
170 // set the shift (only the phi component is non zero)
171 spherical_shift[0] = 0.0;
172 spherical_shift[1] = 0.0;
173 spherical_shift[2] = -2.0 * M * a * r_BL / AA;
174}
