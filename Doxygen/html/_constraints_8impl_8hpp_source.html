37 template <class data_t, template <typename> class vars_t,
38  template <typename> class diff2_vars_t>
39 Constraints::constraints_t<data_t> Constraints::constraint_equations(
40  const vars_t<data_t> &vars, const vars_t<Tensor<1, data_t>> &d1,
41  const diff2_vars_t<Tensor<2, data_t>> &d2) const
42 {
43  constraints_t<data_t> out;
44 
45  const data_t chi_regularised = simd_max(1e-6, vars.chi);
46 
47  auto h_UU = TensorAlgebra::compute_inverse_sym(vars.h);
48  auto chris = TensorAlgebra::compute_christoffel(d1.h, h_UU);
49 
50  auto ricci = CCZ4Geometry::compute_ricci(vars, d1, d2, h_UU, chris);
51 
52  auto A_UU = TensorAlgebra::raise_all(vars.A, h_UU);
53  data_t tr_A2 = TensorAlgebra::compute_trace(vars.A, A_UU);
54 
55  out.Ham = ricci.scalar +
56  (GR_SPACEDIM - 1.) * vars.K * vars.K / GR_SPACEDIM - tr_A2;
57  out.Ham -= 2 * m_cosmological_constant;
58 
59  Tensor<2, data_t> covd_A[CH_SPACEDIM];
60  FOR3(i, j, k)
61  {
62  covd_A[i][j][k] = d1.A[j][k][i];
63  FOR1(l)
64  {
65  covd_A[i][j][k] += -chris.ULL[l][i][j] * vars.A[l][k] -
66  chris.ULL[l][i][k] * vars.A[l][j];
67  }
68  }
69 
70  FOR1(i) { out.Mom[i] = -(GR_SPACEDIM - 1.) * d1.K[i] / GR_SPACEDIM; }
71  FOR3(i, j, k)
72  {
73  out.Mom[i] += h_UU[j][k] *
74  (covd_A[k][j][i] - GR_SPACEDIM * vars.A[i][j] *
75  d1.chi[k] / (2 * chi_regularised));
76  }
77 
78  return out;
79 }
