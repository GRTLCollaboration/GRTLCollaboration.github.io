15template <class SurfaceGeometry, class AHFunction>
16AHInterpolation_t<SurfaceGeometry, AHFunction>::AHInterpolation_t(
17 const SurfaceGeometry &a_coord_system,
18 AMRInterpolator<Lagrange<4>> *a_interpolator)
19 : m_coord_system(a_coord_system), m_interpolator(a_interpolator)
20{
21 // below:
22 // determine maximum and minimum physical coordinate of the grid
23 // (so that 'is_in_grid' knows when PETSc has diverged out of the grid and
24 // doesn't let him do so,
25 // as it would cause an error in the AMRInterpolator)
26
27 const Box &domainBox = const_cast<AMR &>(m_interpolator->getAMR())
28 .getAMRLevels()[0]
29 ->problemDomain()
30 .domainBox();
31 const IntVect &small_end = domainBox.smallEnd();
32 const IntVect &big_end = domainBox.bigEnd();
33
34 const std::array<double, CH_SPACEDIM> &coarsest_dx =
35 m_interpolator->get_coarsest_dx();
36 const std::array<double, CH_SPACEDIM> &coarsest_origin =
37 m_interpolator->get_coarsest_origin();
38
39 // set coordinate minimum and maximum as ONE cells before the boundary
40 for (unsigned i = 0; i < CH_SPACEDIM; ++i)
41 {
42 if (m_interpolator->get_boundary_reflective(Side::Lo, i))
43 {
44 m_coord_min[i] =
45 -(big_end[i] - 1) * coarsest_dx[i] - coarsest_origin[i];
46 }
47 else if (m_interpolator->get_boundary_periodic(i))
48 {
49 m_coord_min[i] = small_end[i] * coarsest_dx[i];
50 }
51 else
52 {
53 m_coord_min[i] =
54 (small_end[i] + 1) * coarsest_dx[i] + coarsest_origin[i];
55 }
56
57 if (m_interpolator->get_boundary_reflective(Side::Hi, i))
58 {
59 m_coord_max[i] =
60 (2 * big_end[i] - 1) * coarsest_dx[i] + coarsest_origin[i];
61 }
62 else if (m_interpolator->get_boundary_periodic(i))
63 {
64 m_coord_max[i] = (big_end[i] + 1) * coarsest_dx[i];
65 }
66 else
67 {
68 m_coord_max[i] =
69 (big_end[i] - 1) * coarsest_dx[i] + coarsest_origin[i];
70 }
71 // pout() << "m_coord = " << m_coord_min[i] << "\t" << m_coord_max[i]
72 // << " with coarse origin " << coarsest_origin[i] << " and dx "
73 // << coarsest_dx[i] << std::endl;
74 }
75}

112template <class SurfaceGeometry, class AHFunction>
113bool AHInterpolation_t<SurfaceGeometry, AHFunction>::is_in_grid(
114 D_DECL(double &x, double &y, double &z))
115{
116 CH_TIME("AHInterpolation::is_in_grid");
117
118 bool out_of_grid = false;
119
120 // if out of bounds, put back in grid
121 if (x < m_coord_min[0])
122 {
123 out_of_grid = true;
124 x = m_coord_min[0];
125 }
126 if (x > m_coord_max[0])
127 {
128 out_of_grid = true;
129 x = m_coord_max[0];
130 }
131
132 if (y < m_coord_min[1])
133 {
134 out_of_grid = true;
135 y = m_coord_min[1];
136 }
137 if (y > m_coord_max[1])
138 {
139 out_of_grid = true;
140 y = m_coord_max[1];
141 }
142
143#if CH_SPACEDIM == 3
144 if (z < m_coord_min[2])
145 {
146 out_of_grid = true;
147 z = m_coord_min[2];
148 }
149 if (z > m_coord_max[2])
150 {
151 out_of_grid = true;
152 z = m_coord_max[2];
153 }
154#endif
155
156 return out_of_grid;
157}

190template <class SurfaceGeometry, class AHFunction>
191void AHInterpolation_t<SurfaceGeometry, AHFunction>::break_interpolation_loop()
192 const
193{
194 CH_TIME("AMRInterpolator::break_interpolation_loop");
195
196#ifdef CH_MPI
197 // break "keep_interpolating" loop
198 int ZERO = 0;
199 int keep_interpolating = 0;
200 // this is also called in 'interpolate()', breaking the loop

205template <class SurfaceGeometry, class AHFunction>
206int AHInterpolation_t<SurfaceGeometry, AHFunction>::interpolate()
207{
208 CH_TIME("AHInterpolation::interpolate");
209
210 // Code below used to allow PETSc to run in a sub-communicator.
211 // For that, PETSc processes run 'set_coordinates' and 'interpolate', while
212 // non-PETSc processes loop through 'interpolate'. If all cores run
213 // interpolate, 'MPI_Allreduce' will return 'keep_interpolating' as true. To
214 // exit the loop for non-PETSc cores, simply do on them a call to:
215 // MPI_Allreduce(&ZERO, &keep_interpolating, 1, MPI_INT, MPI_LAND,
216 // Chombo_MPI::comm); (this is done in 'break_interpolation_loop()')
217
218 int keep_interpolating = 1;
219#ifdef CH_MPI
220 int ONE = 1;
221 MPI_Allreduce(&ONE, &keep_interpolating, 1, MPI_INT, MPI_LAND,
222 Chombo_MPI::comm);
223#endif
224
225 if (!keep_interpolating)
226 return 0;
227
228 const int n = m_x.size();
229
230 InterpolationQuery query(n);
231 query.setCoords(0, m_x.data()).setCoords(1, m_y.data());
232#if CH_SPACEDIM == 3
233 query.setCoords(2, m_z.data());
234#endif
235
236 for (int i = AHFunction::vars_min(); i <= AHFunction::vars_max(); ++i)
237 m_data.set_vars(query, i, i, VariableType::evolution, n);
238 for (int i = AHFunction::d1_vars_min(); i <= AHFunction::d1_vars_max(); ++i)
239 m_data.set_d1(query, i, i, VariableType::evolution, n);
240 for (int i = AHFunction::d2_vars_min(); i <= AHFunction::d2_vars_max(); ++i)
241 m_data.set_d2(query, i, i, VariableType::evolution, n);
242
243 m_interpolator->interp(query);
244
245 return 1;
246}

248template <class SurfaceGeometry, class AHFunction>
249void AHInterpolation_t<SurfaceGeometry, AHFunction>::refresh_interpolator(
250 bool printing_step,
251 const std::map<std::string, std::tuple<int, VariableType, int>> &extra_vars)
252{
253 CH_TIME("AHInterpolation::refresh_interpolator");
254
255 // brute force filling of all ghosts:
256 // m_interpolator->refresh();
257
258 // OR try to only interpolate the variables needed:
259
260 int min_evolution_var =
261 std::min(AHFunction::vars_min(), std::min(AHFunction::d1_vars_min(),
262 AHFunction::d2_vars_min()));
263 int max_evolution_var =
264 std::max(AHFunction::vars_max(), std::max(AHFunction::d1_vars_max(),
265 AHFunction::d2_vars_max()));
266
267 int min_diagnostic_var = -1;
268 int max_diagnostic_var = -1;
269
270 for (auto &var : extra_vars)
271 {
272 int var_enum = std::get<0>(var.second);
273 VariableType var_type = std::get<1>(var.second);
274
275 if (var_type == VariableType::evolution)
276 {
277 if (var_enum < min_evolution_var)
278 min_evolution_var = var_enum;
279 else if (var_enum > max_evolution_var)
280 max_evolution_var = var_enum;
281 }
282 else if (printing_step) // diagnostic
283 {
284 if (min_diagnostic_var == -1)
285 {
286 min_diagnostic_var = var_enum;
287 max_diagnostic_var = var_enum;
288 }
289 else if (var_enum < min_diagnostic_var)
290 min_diagnostic_var = var_enum;
291 else if (var_enum > max_diagnostic_var)
292 max_diagnostic_var = var_enum;
293 }
294 }
295
296 // fill ghosts manually to minimise communication
297 bool fill_ghosts = false;
298 m_interpolator->refresh(fill_ghosts);
299 m_interpolator->fill_multilevel_ghosts(
300 VariableType::evolution,
301 Interval(min_evolution_var, max_evolution_var));
302 if (printing_step && min_diagnostic_var != -1)
303 {
304 m_interpolator->fill_multilevel_ghosts(
305 VariableType::diagnostic,
306 Interval(min_diagnostic_var, max_diagnostic_var));
307 }
308}

312template <class SurfaceGeometry, class AHFunction>
313bool AHInterpolation_t<SurfaceGeometry, AHFunction>::set_coordinates(
314 const vector<double> &f, const vector<double> &u,
315#if CH_SPACEDIM == 3
316 const vector<double> &v,
317#endif
318 double add_epsilon)
319{
320 CH_TIME("AHInterpolation::set_coordinates");
321
322 CH_assert(f.size() == u.size());
323#if CH_SPACEDIM == 3
324 CH_assert(u.size() == v.size());
325#endif
326
327 const int n = f.size();
328
329 // resize vectors
330 m_f.resize(n);
331 m_u.resize(n);
332#if CH_SPACEDIM == 3
333 m_v.resize(n);
334#endif
335
336 m_x.resize(n);
337 m_y.resize(n);
338#if CH_SPACEDIM == 3
339 m_z.resize(n);
340#endif
341
342 bool out_of_grid = false;
343
344 // Transform to Cartesian
345 for (int i = 0; i < n; ++i)
346 {
347 m_f[i] = f[i] + add_epsilon;
348 m_u[i] = u[i];
349#if CH_SPACEDIM == 3
350 m_v[i] = v[i];
351#endif
352
353 m_x[i] =
354 m_coord_system.get_grid_coord(0, D_DECL(m_f[i], m_u[i], m_v[i]));
355 m_y[i] =
356 m_coord_system.get_grid_coord(1, D_DECL(m_f[i], m_u[i], m_v[i]));
357#if CH_SPACEDIM == 3
358 m_z[i] = m_coord_system.get_grid_coord(2, m_f[i], m_u[i], m_v[i]);
359#endif
360
361 // don't let PETSc diverge to outside of the grid (this can happen if
362 // there is no BH)
363 out_of_grid |= is_in_grid(m_x[i], m_y[i]
364#if CH_SPACEDIM == 3
365 ,
366 m_z[i]
367#endif
368 );
369 }
370
371 return out_of_grid;
372}

413template <class SurfaceGeometry, class AHFunction>
414void AHInterpolation_t<SurfaceGeometry, AHFunction>::interpolate_extra_vars(
415 const std::map<std::string, std::tuple<int, VariableType, int>> &extra_vars)
416{
417 CH_TIME("AHInterpolation::interpolate_extra_vars");
418
419 if (extra_vars.size() == 0)
420 return;
421
422 int n = m_x.size();
423
424 InterpolationQuery query(n);
425 query.setCoords(0, m_x.data()).setCoords(1, m_y.data());
426#if CH_SPACEDIM == 3
427 query.setCoords(2, m_z.data());
428#endif
429
430 for (auto &var : extra_vars)
431 {
432 int var_enum = std::get<0>(var.second);
433 VariableType var_type = std::get<1>(var.second);
434 int der_type = std::get<2>(var.second);
435
436 if (der_type == 0)
437 m_extra.set_vars(query, var_enum, var.first, var_type, n);
438 else if (der_type == 1)
439 m_extra.set_d1(query, var_enum, var.first, var_type, n);
440 else // if(der_type == 2)
441 m_extra.set_d2(query, var_enum, var.first, var_type, n);
442 }
443
444 // ghosts for extra evolution or diagnostic vars already filled in
445 // 'refresh_interpolator'
446 m_interpolator->interp(query);
447}
