22 public:
23 FixedGridsTaggingCriterion(const double dx, const int a_level,
24 const double a_L,
25 const std::array<double, CH_SPACEDIM> a_center)
26 : m_dx(dx), m_L(a_L), m_level(a_level), m_center(a_center){};
27
28 template <class data_t> void compute(Cell<data_t> current_cell) const
29 {
30 data_t criterion = 0.0;
31 // make sure the inner part is regridded around the horizon
32 // take L as the length of full grid, so tag inner 1/2
33 // of it, which means inner \pm L/4
34 double ratio = pow(2.0, -(m_level + 2.0));
35 const Coordinates<data_t> coords(current_cell, m_dx, m_center);
36 const data_t max_abs_xy = simd_max(abs(coords.x), abs(coords.y));
37 const data_t max_abs_xyz = simd_max(max_abs_xy, abs(coords.z));
38 auto regrid = simd_compare_lt(max_abs_xyz, m_L * ratio);
39 criterion = simd_conditional(regrid, 100.0, criterion);
40
41 // Write back into the flattened Chombo box
42 current_cell.store_vars(criterion, 0);
43 }
44};
