53 template <VariableType var_t> void AMRReductions<var_t>::set_domain_volume()
54 {
55  // first check if m_level_data_ptrs has been set
56  CH_assert((m_level_data_ptrs.size() > 0) &&
57  (m_level_data_ptrs[0] != nullptr));
58 
59  // first calculate the volume assuming each cell on the coarsest level has
60  // unit length
61  int cell_volume =
62  m_level_data_ptrs[0]->disjointBoxLayout().physDomain().size().product();
63 
64  // multiply by dx_coarsest to get real volume
65  m_domain_volume =
66  pow(m_coarsest_dx, CH_SPACEDIM) * static_cast<double>(cell_volume);
67 }
